Introduction to WordPress Pages

WordPress Pages are the foundation of your website's static content. Unlike posts, pages are hierarchical, timeless, and typically form the structural backbone of your site - from your homepage to your about page, contact forms, and service descriptions.

üìÑ
Key Insight
Pages are perfect for content that doesn't change frequently and doesn't need to be organized by date. They're excluded from RSS feeds and are meant to be permanent fixtures of your website's architecture.

Creating Pages Programmatically

<?php
// Create a new page programmatically
$page_data = array(
    'post_title'    => 'About Us',
    'post_content'  => 'This is the about us page content.',
    'post_status'   => 'publish',
    'post_type'     => 'page',
    'post_author'   => 1,
    'post_slug'     => 'about-us',
    'page_template' => 'page-templates/full-width.php',
    'menu_order'    => 1,
    'comment_status' => 'closed',
    'ping_status'   => 'closed'
);

// Insert the page
$page_id = wp_insert_post($page_data);

// Set as front page
update_option('page_on_front', $page_id);
update_option('show_on_front', 'page');

// Set as posts page
update_option('page_for_posts', $page_id);

// Update existing page
wp_update_post(array(
    'ID' => $page_id,
    'post_title' => 'About Our Company',
    'post_content' => 'Updated content here'
));

// Create child page
$child_page_data = array(
    'post_title' => 'Our Team',
    'post_content' => 'Meet our amazing team.',
    'post_status' => 'publish',
    'post_type' => 'page',
    'post_parent' => $page_id, // Set parent page
    'menu_order' => 1
);

$child_id = wp_insert_post($child_page_data);

// Add custom fields to page
update_post_meta($page_id, 'hero_image', 'path/to/image.jpg');
update_post_meta($page_id, 'custom_css', '.hero { background: #f0f0f0; }');
update_post_meta($page_id, 'sidebar_position', 'left');
?>

Working with Page Hierarchy

<?php
// Get page hierarchy
function get_page_hierarchy($parent_id = 0, $level = 0) {
    $pages = get_pages(array(
        'parent' => $parent_id,
        'sort_column' => 'menu_order',
        'sort_order' => 'ASC'
    ));
    
    foreach ($pages as $page) {
        echo str_repeat('‚Äî ', $level) . $page->post_title . '<br>';
        
        // Recursively get children
        get_page_hierarchy($page->ID, $level + 1);
    }
}

// Display breadcrumbs
function display_breadcrumbs() {
    global $post;
    
    if (!is_page() || is_front_page()) {
        return;
    }
    
    echo '<nav class="breadcrumbs">';
    echo '<a href="' . home_url() . '">Home</a> > ';
    
    // Get all ancestors
    $ancestors = get_post_ancestors($post->ID);
    $ancestors = array_reverse($ancestors);
    
    foreach ($ancestors as $ancestor) {
        echo '<a href="' . get_permalink($ancestor) . '">';
        echo get_the_title($ancestor) . '</a> > ';
    }
    
    echo get_the_title();
    echo '</nav>';
}

// Get top-level parent page
function get_top_parent_page() {
    global $post;
    
    if ($post->post_parent) {
        $ancestors = get_post_ancestors($post->ID);
        $root = count($ancestors) - 1;
        $parent_id = $ancestors[$root];
    } else {
        $parent_id = $post->ID;
    }
    
    return $parent_id;
}

// List child pages
function list_child_pages() {
    global $post;
    
    $children = wp_list_pages(array(
        'title_li' => '',
        'child_of' => $post->ID,
        'echo' => 0
    ));
    
    if ($children) {
        echo '<ul class="child-pages">' . $children . '</ul>';
    }
}
?>

Special Purpose Pages

Front Page

  • Your site's homepage
  • Can be static page or latest posts
  • Set in Settings ‚Üí Reading
  • Uses front-page.php template
  • Often includes hero sections

Blog Page

  • Displays your blog posts
  • Set in Settings ‚Üí Reading
  • Uses home.php or index.php
  • Cannot edit content directly
  • Inherits archive settings

Setting Up Special Pages

<?php
// Set front page and blog page programmatically
function setup_default_pages() {
    // Create home page
    $home_page = array(
        'post_title' => 'Home',
        'post_content' => 'Welcome to our website',
        'post_status' => 'publish',
        'post_type' => 'page'
    );
    $home_id = wp_insert_post($home_page);
    
    // Create blog page
    $blog_page = array(
        'post_title' => 'Blog',
        'post_content' => '',
        'post_status' => 'publish',
        'post_type' => 'page'
    );
    $blog_id = wp_insert_post($blog_page);
    
    // Set as front page and posts page
    update_option('show_on_front', 'page');
    update_option('page_on_front', $home_id);
    update_option('page_for_posts', $blog_id);
    
    // Create Privacy Policy page
    $privacy_page = array(
        'post_title' => 'Privacy Policy',
        'post_content' => get_default_privacy_policy_content(),
        'post_status' => 'publish',
        'post_type' => 'page'
    );
    $privacy_id = wp_insert_post($privacy_page);
    update_option('wp_page_for_privacy_policy', $privacy_id);
    
    // Create 404 page
    $error_page = array(
        'post_title' => '404 - Page Not Found',
        'post_content' => 'Sorry, the page you are looking for does not exist.',
        'post_status' => 'publish',
        'post_type' => 'page',
        'page_template' => '404-template.php'
    );
    wp_insert_post($error_page);
}

// Check if on specific page
if (is_front_page()) {
    // Homepage specific code
}

if (is_page('about-us')) {
    // About page specific code
}

if (is_page_template('landing.php')) {
    // Landing page template specific code
}

// Custom query for pages
$pages = new WP_Query(array(
    'post_type' => 'page',
    'posts_per_page' => -1,
    'meta_key' => 'featured',
    'meta_value' => 'yes',
    'orderby' => 'menu_order',
    'order' => 'ASC'
));
?>

Popular Page Builders

<?php
// Detect if page builder is active
function is_page_builder_active() {
    // Elementor
    if (defined('ELEMENTOR_VERSION')) {
        return 'elementor';
    }
    
    // Beaver Builder
    if (class_exists('FLBuilder')) {
        return 'beaver-builder';
    }
    
    // Divi
    if (function_exists('et_pb_is_pagebuilder_used')) {
        return 'divi';
    }
    
    // Gutenberg Block Editor
    if (has_blocks()) {
        return 'gutenberg';
    }
    
    return false;
}

// Disable page builder on specific pages
add_filter('elementor/frontend/the_content', function($content) {
    if (is_page('simple-page')) {
        return get_the_content();
    }
    return $content;
});

// Add custom Gutenberg blocks for pages
function register_custom_blocks() {
    register_block_type('custom/hero-section', array(
        'render_callback' => 'render_hero_section'
    ));
    
    register_block_type('custom/features-grid', array(
        'render_callback' => 'render_features_grid'
    ));
}
add_action('init', 'register_custom_blocks');
?>

Page Optimization

<?php
// Add SEO meta tags to pages
add_action('wp_head', function() {
    if (is_page()) {
        global $post;
        
        // Get SEO data
        $meta_description = get_post_meta($post->ID, '_meta_description', true);
        $meta_keywords = get_post_meta($post->ID, '_meta_keywords', true);
        $og_image = get_post_meta($post->ID, '_og_image', true);
        
        if (!$meta_description) {
            $meta_description = wp_trim_words($post->post_content, 30);
        }
        
        echo '<meta name="description" content="' . esc_attr($meta_description) . '">';
        echo '<meta name="keywords" content="' . esc_attr($meta_keywords) . '">';
        
        // Open Graph tags
        echo '<meta property="og:title" content="' . esc_attr($post->post_title) . '">';
        echo '<meta property="og:description" content="' . esc_attr($meta_description) . '">';
        echo '<meta property="og:url" content="' . get_permalink() . '">';
        echo '<meta property="og:type" content="website">';
        
        if ($og_image) {
            echo '<meta property="og:image" content="' . esc_url($og_image) . '">';
        }
        
        // Schema.org markup
        ?>
        <script type="application/ld+json">
        {
            "@context": "https://schema.org",
            "@type": "WebPage",
            "name": "<?php echo esc_js($post->post_title); ?>",
            "description": "<?php echo esc_js($meta_description); ?>",
            "url": "<?php echo get_permalink(); ?>",
            "datePublished": "<?php echo get_the_date('c'); ?>",
            "dateModified": "<?php echo get_the_modified_date('c'); ?>"
        }
        </script>
        <?php
    }
});

// Optimize page loading
add_action('init', function() {
    // Remove unnecessary scripts on specific pages
    if (is_page('simple-page')) {
        remove_action('wp_head', 'print_emoji_detection_script', 7);
        remove_action('wp_print_styles', 'print_emoji_styles');
    }
});

// Add performance monitoring
add_filter('the_content', function($content) {
    if (is_page() && current_user_can('manage_options')) {
        $load_time = timer_stop();
        $queries = get_num_queries();
        
        $content .= '<!-- Page generated in ' . $load_time . ' seconds with ' . $queries . ' queries -->';
    }
    return $content;
});
?>
