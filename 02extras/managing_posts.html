Introduction

Posts are the heart of WordPress's content management system. They power blogs, news sites, and any content that needs to be organized chronologically and categorically. This lesson covers everything you need to know about managing posts effectively.

<?php
// Register a custom block
function register_custom_block() {
    wp_register_script(
        'custom-block-script',

        array('wp-blocks', 'wp-element', 'wp-editor')
    );

        'editor_script' => 'custom-block-script',
        'render_callback' => 'render_custom_block'

    return '<div class="custom-block">' . $content . '</div>';

Working with Post Status

<?php
// Change post status programmatically
function update_post_status($post_id, $new_status) {
    $valid_statuses = array(
        'publish', 'pending', 'draft', 'future',
        'private', 'trash', 'auto-draft', 'inherit'
    );
    
    if (!in_array($new_status, $valid_statuses)) {
        return false;
    }
    
    $post_data = array(
        'ID' => $post_id,
        'post_status' => $new_status
    );
    
    // If scheduling for future
    if ($new_status === 'future') {
        $post_data['post_date'] = '2024-02-01 10:00:00';
        $post_data['post_date_gmt'] = get_gmt_from_date('2024-02-01 10:00:00');
    }
    
    return wp_update_post($post_data);
}

// Hook into status transitions
add_action('transition_post_status', function($new_status, $old_status, $post) {
    if ($new_status === 'publish' && $old_status !== 'publish') {
        // Post just published
        // Send notifications, trigger workflows, etc.
        do_action('my_post_published', $post);
    }
    
    if ($new_status === 'trash') {
        // Post trashed
        // Clean up related data
        delete_post_meta($post->ID, 'view_count');
    }
}, 10, 3);

// Password protect a post
wp_update_post(array(
    'ID' => $post_id,
    'post_password' => 'mypassword'
));

// Make post private
wp_update_post(array(
    'ID' => $post_id,
    'post_status' => 'private'
));

// Check post status in templates
if (get_post_status() === 'private') {
    echo '<div class="private-notice">This is a private post</div>';
}
?>

Custom Bulk Actions

<?php
// Add custom bulk action
add_filter('bulk_actions-edit-post', function($bulk_actions) {
    $bulk_actions['export_csv'] = 'Export to CSV';
    $bulk_actions['reset_meta'] = 'Reset Custom Fields';
    return $bulk_actions;
});

// Handle custom bulk action
add_filter('handle_bulk_actions-edit-post', function($redirect_to, $action, $post_ids) {
    if ($action === 'export_csv') {
        // Export selected posts to CSV
        export_posts_to_csv($post_ids);
        $redirect_to = add_query_arg('exported', count($post_ids), $redirect_to);
    }
    
    if ($action === 'reset_meta') {
        foreach ($post_ids as $post_id) {
            delete_post_meta($post_id, 'custom_field');
        }
        $redirect_to = add_query_arg('reset', count($post_ids), $redirect_to);
    }
    
    return $redirect_to;
}, 10, 3);

// Display admin notices for bulk actions
add_action('admin_notices', function() {
    if (!empty($_REQUEST['exported'])) {
        $count = intval($_REQUEST['exported']);
        printf(
            '<div class="notice notice-success is-dismissible"><p>%d posts exported to CSV.</p></div>',
            $count
        );
    }
});
?>

Post Scheduling

Schedule posts to publish automatically at specific times:

<?php
// Schedule a post for future publication
$future_timestamp = strtotime('+1 week');
$post_data = array(
    'post_title' => 'Future Post',
    'post_content' => 'This will be published next week',
    'post_status' => 'future',
    'post_date' => date('Y-m-d H:i:s', $future_timestamp),
    'post_date_gmt' => gmdate('Y-m-d H:i:s', $future_timestamp)
);

$post_id = wp_insert_post($post_data);

// Reschedule an existing post
wp_update_post(array(
    'ID' => $post_id,
    'post_status' => 'future',
    'post_date' => '2024-02-14 09:00:00',
    'post_date_gmt' => get_gmt_from_date('2024-02-14 09:00:00')
));

// Get all scheduled posts
$scheduled_posts = get_posts(array(
    'post_status' => 'future',
    'posts_per_page' => -1,
    'orderby' => 'post_date',
    'order' => 'ASC'
));

foreach ($scheduled_posts as $post) {
    echo $post->post_title . ' - Scheduled for: ' . $post->post_date . '<br>';
}

// Hook into scheduled post publication
add_action('publish_future_post', function($post_id) {
    // Actions when scheduled post publishes
    // Send notifications, social media, etc.
    $post = get_post($post_id);
    
    // Send email notification
    wp_mail(
        get_option('admin_email'),
        'Scheduled Post Published: ' . $post->post_title,
        'Your scheduled post has been published: ' . get_permalink($post_id)
    );
});

// Create editorial calendar
function get_editorial_calendar($month = null, $year = null) {
    if (!$month) $month = date('m');
    if (!$year) $year = date('Y');
    
    $args = array(
        'post_status' => array('publish', 'future'),
        'year' => $year,
        'monthnum' => $month,
        'posts_per_page' => -1,
        'orderby' => 'date',
        'order' => 'ASC'
    );
    
    return get_posts($args);
}
?>

Working with Revisions

<?php
// Configure revisions in wp-config.php
define('WP_POST_REVISIONS', 10); // Keep only 10 revisions
// Or disable: define('WP_POST_REVISIONS', false);

// Configure autosave interval (default is 60 seconds)
define('AUTOSAVE_INTERVAL', 120); // 2 minutes

// Get post revisions
$revisions = wp_get_post_revisions($post_id, array(
    'posts_per_page' => 10,
    'order' => 'DESC'
));

foreach ($revisions as $revision) {
    echo 'Revision by ' . get_the_author_meta('display_name', $revision->post_author);
    echo ' on ' . get_the_date('', $revision);
    echo '<br>';
}

// Restore a revision
wp_restore_post_revision($revision_id);

// Compare revisions
$old_revision = wp_get_post_revision($old_revision_id);
$new_revision = wp_get_post_revision($new_revision_id);

$title_diff = wp_text_diff(
    $old_revision->post_title,
    $new_revision->post_title
);

$content_diff = wp_text_diff(
    $old_revision->post_content,
    $new_revision->post_content
);

// Clean up old revisions
function cleanup_old_revisions() {
    global $wpdb;
    
    $posts = get_posts(array(
        'post_type' => 'post',
        'posts_per_page' => -1
    ));
    
    foreach ($posts as $post) {
        $revisions = wp_get_post_revisions($post->ID);
        
        if (count($revisions) > 10) {
            $revisions = array_slice($revisions, 10);
            foreach ($revisions as $revision) {
                wp_delete_post_revision($revision->ID);
            }
        }
    }
}

// Hook to save revision metadata
add_action('_wp_put_post_revision', function($revision_id) {
    $parent_id = wp_is_post_revision($revision_id);
    
    if ($parent_id) {
        // Copy custom fields to revision
        $meta_keys = get_post_custom_keys($parent_id);
        
        foreach ($meta_keys as $meta_key) {
            $meta_values = get_post_custom_values($meta_key, $parent_id);
            foreach ($meta_values as $meta_value) {
                add_metadata('post', $revision_id, $meta_key, $meta_value);
            }
        }
    }
});
?>

SEO Best Practices for Posts

<?php
// Add SEO meta tags
add_action('wp_head', function() {
    if (is_single()) {
        global $post;
        
        // Meta description
        $description = get_post_meta($post->ID, '_meta_description', true);
        if (!$description) {
            $description = wp_trim_words($post->post_excerpt ?: $post->post_content, 30);
        }
        echo '<meta name="description" content="' . esc_attr($description) . '">';
        
        // Open Graph tags
        echo '<meta property="og:title" content="' . esc_attr($post->post_title) . '">';
        echo '<meta property="og:description" content="' . esc_attr($description) . '">';
        echo '<meta property="og:url" content="' . get_permalink() . '">';
        echo '<meta property="og:type" content="article">';
        
        if (has_post_thumbnail()) {
            echo '<meta property="og:image" content="' . get_the_post_thumbnail_url($post, 'large') . '">';
        }
        
        // Twitter Card
        echo '<meta name="twitter:card" content="summary_large_image">';
        echo '<meta name="twitter:title" content="' . esc_attr($post->post_title) . '">';
        echo '<meta name="twitter:description" content="' . esc_attr($description) . '">';
        
        // Schema.org Article markup
        $schema = array(
            '@context' => 'https://schema.org',
            '@type' => 'Article',
            'headline' => $post->post_title,
            'description' => $description,
            'datePublished' => get_the_date('c'),
            'dateModified' => get_the_modified_date('c'),
            'author' => array(
                '@type' => 'Person',
                'name' => get_the_author()
            ),
            'publisher' => array(
                '@type' => 'Organization',
                'name' => get_bloginfo('name'),
                'logo' => array(
                    '@type' => 'ImageObject',
                    'url' => get_site_icon_url()
                )
            )
        );
        
        if (has_post_thumbnail()) {
            $schema['image'] = get_the_post_thumbnail_url($post, 'full');
        }
        
        echo '<script type="application/ld+json">' . json_encode($schema) . '</script>';
    }
});

// Generate SEO-friendly slug
function generate_seo_slug($title) {
    $slug = sanitize_title($title);
    
    // Remove stop words
    $stop_words = array('a', 'an', 'the', 'and', 'or', 'but', 'in', 'on', 'at', 'to', 'for');
    $slug_parts = explode('-', $slug);
    $slug_parts = array_diff($slug_parts, $stop_words);
    
    return implode('-', $slug_parts);
}

// Add reading time
function calculate_reading_time($post_id) {
    $content = get_post_field('post_content', $post_id);
    $word_count = str_word_count(strip_tags($content));
    $reading_time = ceil($word_count / 200); // 200 words per minute
    
    update_post_meta($post_id, '_reading_time', $reading_time);
    
    return $reading_time;
}
?>

✅ SEO Checklist for Posts

  • Write compelling meta descriptions (150-160 characters)
  • Use focus keywords in title, URL, and first paragraph
  • Optimize images with alt text and descriptive filenames
  • Include internal and external links
  • Use header tags (H2, H3) for structure
  • Write content that's at least 300 words
  • Enable and respond to comments
  • Update old posts regularly
