WordPress Project Structure for Applications

Organize your WordPress applications for scalability and maintainability

Learning Objectives

By the end of this lesson, you will be able to:

  • Design scalable WordPress project structures
  • Organize files and directories effectively
  • Implement modern development workflows
  • Separate configuration from code
  • Use Composer for dependency management
  • Structure projects for team collaboration

Modern WordPress Project Structure

Root-Level Organization


project-root/
├── .github/                  # GitHub Actions workflows
├── config/                   # Configuration files
│   ├── environments/         # Environment-specific configs
│   ├── application.php       # Main application config
│   └── database.php          # Database configuration
├── public/                   # Web root (document root)
│   ├── wp/                   # WordPress core files
│   ├── app/                  # Custom content directory (wp-content)
│   ├── index.php            # WordPress bootstrap
│   └── wp-config.php        # WordPress configuration
├── resources/               # Source files (uncompiled)
│   ├── assets/             # CSS, JS, images sources
│   ├── views/              # View templates
│   └── languages/          # Translation sources
├── storage/                # Application storage
│   ├── cache/              # Cache files
│   ├── logs/               # Log files
│   └── uploads/            # User uploads
├── tests/                  # Test files
│   ├── unit/               # Unit tests
│   ├── integration/        # Integration tests
│   └── acceptance/         # Acceptance tests
├── vendor/                 # Composer dependencies
├── .env                    # Environment variables
├── .env.example            # Environment template
├── .gitignore             # Git ignore rules
├── composer.json          # PHP dependencies
├── package.json           # Node dependencies
├── phpunit.xml            # PHPUnit configuration
├── webpack.config.js      # Webpack configuration
└── README.md              # Project documentation

Key Principles

  • Separation: Keep WordPress core separate from custom code
  • Security: Move sensitive files above web root
  • Organization: Group related files logically
  • Flexibility: Support multiple environments easily
  • Standards: Follow PSR standards and best practices

Bedrock: Modern WordPress Stack

Bedrock is a modern WordPress stack that helps you get started with the best development tools and practices.

Bedrock Directory Structure


bedrock/
├── config/
│   ├── environments/
│   │   ├── development.php
│   │   ├── staging.php
│   │   └── production.php
│   └── application.php
├── web/                      # Document root
│   ├── app/                  # wp-content equivalent
│   │   ├── mu-plugins/      # Must-use plugins
│   │   ├── plugins/         # Regular plugins
│   │   ├── themes/          # Themes
│   │   └── uploads/         # Media uploads
│   ├── wp/                  # WordPress core (subdirectory)
│   ├── index.php
│   └── wp-config.php
├── vendor/
├── .env
└── composer.json

Setting Up Bedrock


# Install Bedrock
composer create-project roots/bedrock my-project

# Configure environment
cp .env.example .env

# Edit .env file
DB_NAME='database_name'
DB_USER='database_user'
DB_PASSWORD='database_password'
DB_HOST='localhost'

WP_ENV='development'
WP_HOME='http://example.com'
WP_SITEURL="${WP_HOME}/wp"

# Generate salts
wp dotenv salts regenerate

# Install WordPress
composer install

Bedrock Configuration


// config/application.php
use Roots\WPConfig\Config;
use function Env\env;

/**
 * Directory containing all of the site's files
 */
$root_dir = dirname(__DIR__);

/**
 * Document Root
 */
$webroot_dir = $root_dir . '/web';

/**
 * Use Dotenv to set required environment variables
 */
$dotenv = Dotenv\Dotenv::createUnsafeMutable($root_dir);
if (file_exists($root_dir . '/.env')) {
    $dotenv->load();
    $dotenv->required(['WP_HOME', 'WP_SITEURL']);
}

/**
 * Set up our global environment constant
 */
Config::define('WP_ENVIRONMENT_TYPE', env('WP_ENV') ?: 'production');

/**
 * URLs
 */
Config::define('WP_HOME', env('WP_HOME'));
Config::define('WP_SITEURL', env('WP_SITEURL'));

/**
 * Custom Content Directory
 */
Config::define('CONTENT_DIR', '/app');
Config::define('WP_CONTENT_DIR', $webroot_dir . Config::get('CONTENT_DIR'));
Config::define('WP_CONTENT_URL', Config::get('WP_HOME') . Config::get('CONTENT_DIR'));

/**
 * DB settings
 */
Config::define('DB_NAME', env('DB_NAME'));
Config::define('DB_USER', env('DB_USER'));
Config::define('DB_PASSWORD', env('DB_PASSWORD'));
Config::define('DB_HOST', env('DB_HOST') ?: 'localhost');

/**
 * Authentication Keys and Salts
 */
Config::define('AUTH_KEY', env('AUTH_KEY'));
Config::define('SECURE_AUTH_KEY', env('SECURE_AUTH_KEY'));
Config::define('LOGGED_IN_KEY', env('LOGGED_IN_KEY'));
Config::define('NONCE_KEY', env('NONCE_KEY'));

/**
 * Custom Settings
 */
Config::define('AUTOMATIC_UPDATER_DISABLED', true);
Config::define('DISABLE_WP_CRON', env('DISABLE_WP_CRON') ?: false);
Config::define('DISALLOW_FILE_EDIT', true);
Config::define('DISALLOW_FILE_MODS', true);
Config::define('WP_POST_REVISIONS', env('WP_POST_REVISIONS') ?: true);

/**
 * Debugging Settings
 */
Config::define('WP_DEBUG_DISPLAY', false);
Config::define('WP_DEBUG_LOG', env('WP_DEBUG_LOG') ?? false);
Config::define('SCRIPT_DEBUG', false);
ini_set('display_errors', '0');

/**
 * Environment configs
 */
$env_config = __DIR__ . '/environments/' . Config::get('WP_ENVIRONMENT_TYPE') . '.php';

if (file_exists($env_config)) {
    require_once $env_config;
}

Config::apply();

/**
 * Bootstrap WordPress
 */
if (!defined('ABSPATH')) {
    define('ABSPATH', $webroot_dir . '/wp/');
}

Environment-Specific Configuration


// config/environments/development.php
use Roots\WPConfig\Config;

/**
 * Development environment config
 */

// Enable debug mode
Config::define('WP_DEBUG', true);
Config::define('WP_DEBUG_LOG', true);
Config::define('WP_DEBUG_DISPLAY', true);
Config::define('WP_DISABLE_FATAL_ERROR_HANDLER', true);
Config::define('SCRIPT_DEBUG', true);

// Enable all error reporting
ini_set('display_errors', '1');
error_reporting(E_ALL);

// Disable caching
Config::define('WP_CACHE', false);

// Show admin bar
Config::define('WP_ADMIN_BAR', true);

// Enable Save queries for analysis
Config::define('SAVEQUERIES', true);

// config/environments/staging.php
use Roots\WPConfig\Config;

/**
 * Staging environment config
 */

Config::define('WP_DEBUG', false);
Config::define('WP_DEBUG_LOG', true);
Config::define('WP_DEBUG_DISPLAY', false);

// Basic caching
Config::define('WP_CACHE', true);

// Disable indexing
Config::define('WP_ROBOTS_DISALLOW', true);

// config/environments/production.php
use Roots\WPConfig\Config;

/**
 * Production environment config
 */

Config::define('WP_DEBUG', false);
Config::define('WP_DEBUG_LOG', false);
Config::define('WP_DEBUG_DISPLAY', false);

// Enable caching
Config::define('WP_CACHE', true);

// Disable file modifications
Config::define('DISALLOW_FILE_EDIT', true);
Config::define('DISALLOW_FILE_MODS', true);

// Security headers
if (!headers_sent()) {
    header('X-Frame-Options: SAMEORIGIN');
    header('X-Content-Type-Options: nosniff');
    header('X-XSS-Protection: 1; mode=block');
}

Build Tools Configuration

Webpack Configuration


// webpack.config.js
const path = require('path');
const MiniCssExtractPlugin = require('mini-css-extract-plugin');
const { CleanWebpackPlugin } = require('clean-webpack-plugin');
const BrowserSyncPlugin = require('browser-sync-webpack-plugin');

module.exports = (env, argv) => {
    const isDevelopment = argv.mode === 'development';
    
    return {
        entry: {
            app: './resources/scripts/app.js',
            admin: './resources/scripts/admin.js',
            editor: './resources/scripts/editor.js',
        },
        output: {
            path: path.resolve(__dirname, 'dist'),
            filename: 'scripts/[name].[contenthash].js',
        },
        module: {
            rules: [
                {
                    test: /\.js$/,
                    exclude: /node_modules/,
                    use: {
                        loader: 'babel-loader',
                        options: {
                            presets: ['@babel/preset-env'],
                        },
                    },
                },
                {
                    test: /\.(scss|css)$/,
                    use: [
                        isDevelopment ? 'style-loader' : MiniCssExtractPlugin.loader,
                        'css-loader',
                        'postcss-loader',
                        'sass-loader',
                    ],
                },
                {
                    test: /\.(png|jpg|gif|svg)$/,
                    type: 'asset/resource',
                    generator: {
                        filename: 'images/[name].[contenthash][ext]',
                    },
                },
            ],
        },
        plugins: [
            new CleanWebpackPlugin(),
            new MiniCssExtractPlugin({
                filename: 'styles/[name].[contenthash].css',
            }),
            isDevelopment && new BrowserSyncPlugin({
                proxy: 'http://localhost',
                files: [
                    'dist/**/*.css',
                    'dist/**/*.js',
                    'templates/**/*.php',
                    'components/**/*.php',
                ],
            }),
        ].filter(Boolean),
        devtool: isDevelopment ? 'source-map' : false,
        optimization: {
            splitChunks: {
                chunks: 'all',
                cacheGroups: {
                    vendor: {
                        test: /[\\/]node_modules[\\/]/,
                        name: 'vendor',
                        priority: 10,
                    },
                },
            },
        },
    };
};

Summary

Key Takeaways

  • Modern WordPress projects benefit from structured organization
  • Separation of concerns improves maintainability
  • Environment-based configuration adds flexibility
  • Composer manages PHP dependencies effectively
  • Build tools optimize asset management
  • Testing structure ensures code quality

Best Practices

  • Keep WordPress core in a subdirectory
  • Use environment variables for configuration
  • Implement PSR-4 autoloading
  • Separate source from compiled assets
  • Version control everything except dependencies
  • Document your structure decisions
