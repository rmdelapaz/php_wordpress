REST API Integration for CPTs

Enable and customize REST API support for custom post types

Learning Objectives

By the end of this lesson, you will be able to:

  • Enable REST API support for custom post types
  • Create custom REST API endpoints
  • Customize API responses and fields
  • Implement authentication and permissions
  • Handle custom meta fields in the API
  • Build headless WordPress applications

Enabling REST API Support

Basic REST API Registration


// Enable REST API support when registering CPT
function register_cpt_with_rest_api() {
    register_post_type('product', [
        'labels' => [
            'name' => 'Products',
            'singular_name' => 'Product',
        ],
        'public' => true,
        'show_in_rest' => true, // Enable REST API
        'rest_base' => 'products', // Custom endpoint base
        'rest_controller_class' => 'WP_REST_Posts_Controller', // Controller class
        'rest_namespace' => 'wp/v2', // API namespace
        'supports' => ['title', 'editor', 'thumbnail', 'custom-fields'],
    ]);
    
    // Register taxonomy with REST support
    register_taxonomy('product_category', 'product', [
        'labels' => [
            'name' => 'Product Categories',
            'singular_name' => 'Product Category',
        ],
        'hierarchical' => true,
        'show_in_rest' => true,
        'rest_base' => 'product-categories',
        'rest_controller_class' => 'WP_REST_Terms_Controller',
    ]);
}
add_action('init', 'register_cpt_with_rest_api');

Custom REST Controller


// Create a custom REST controller for more control
class Product_REST_Controller extends WP_REST_Posts_Controller {
    
    public function __construct() {
        $this->namespace = 'myapi/v1';
        $this->rest_base = 'products';
        $this->post_type = 'product';
    }
    
    // Override methods as needed
    public function get_items($request) {
        // Custom logic before getting items
        $params = $request->get_params();
        
        // Add custom filtering
        if (isset($params['min_price'])) {
            add_filter('posts_where', function($where) use ($params) {
                global $wpdb;
                $where .= $wpdb->prepare(
                    " AND {$wpdb->postmeta}.meta_key = 'price' 
                      AND CAST({$wpdb->postmeta}.meta_value AS DECIMAL) >= %d",
                    $params['min_price']
                );
                return $where;
            });
        }
        
        return parent::get_items($request);
    }
    
    public function get_item($request) {
        $response = parent::get_item($request);
        
        // Add custom data to single item response
        if (!is_wp_error($response)) {
            $data = $response->get_data();
            $data['custom_field'] = get_post_meta($data['id'], 'custom_field', true);
            $response->set_data($data);
        }
        
        return $response;
    }
    
    // Custom permission callback
    public function get_items_permissions_check($request) {
        // Allow public access to list
        return true;
    }
    
    public function create_item_permissions_check($request) {
        // Only allow authenticated users to create
        return current_user_can('edit_posts');
    }
}

Creating Custom Endpoints

Register Custom Routes


// Register custom REST API routes
class Custom_Product_Endpoints {
    
    public function __construct() {
        add_action('rest_api_init', [$this, 'register_routes']);
    }
    
    public function register_routes() {
        // Get featured products
        register_rest_route('myapi/v1', '/products/featured', [
            'methods' => WP_REST_Server::READABLE,
            'callback' => [$this, 'get_featured_products'],
            'permission_callback' => '__return_true',
            'args' => [
                'per_page' => [
                    'default' => 10,
                    'sanitize_callback' => 'absint',
                ],
                'page' => [
                    'default' => 1,
                    'sanitize_callback' => 'absint',
                ],
            ],
        ]);
        
        // Get product by SKU
        register_rest_route('myapi/v1', '/products/sku/(?P[a-zA-Z0-9-]+)', [
            'methods' => WP_REST_Server::READABLE,
            'callback' => [$this, 'get_product_by_sku'],
            'permission_callback' => '__return_true',
            'args' => [
                'sku' => [
                    'required' => true,
                    'validate_callback' => function($param) {
                        return is_string($param) && !empty($param);
                    },
                ],
            ],
        ]);
        
        // Product search with filters
        register_rest_route('myapi/v1', '/products/search', [
            'methods' => WP_REST_Server::READABLE,
            'callback' => [$this, 'search_products'],
            'permission_callback' => '__return_true',
            'args' => [
                'keyword' => [
                    'required' => false,
                    'sanitize_callback' => 'sanitize_text_field',
                ],
                'category' => [
                    'required' => false,
                    'sanitize_callback' => 'absint',
                ],
                'min_price' => [
                    'required' => false,
                    'sanitize_callback' => 'absint',
                ],
                'max_price' => [
                    'required' => false,
                    'sanitize_callback' => 'absint',
                ],
                'sort' => [
                    'default' => 'date',
                    'enum' => ['date', 'price', 'title', 'popularity'],
                ],
                'order' => [
                    'default' => 'DESC',
                    'enum' => ['ASC', 'DESC'],
                ],
            ],
        ]);
        
        // Batch update products
        register_rest_route('myapi/v1', '/products/batch', [
            'methods' => WP_REST_Server::CREATABLE,
            'callback' => [$this, 'batch_update_products'],
            'permission_callback' => [$this, 'batch_update_permission'],
            'args' => [
                'products' => [
                    'required' => true,
                    'type' => 'array',
                    'items' => [
                        'type' => 'object',
                        'properties' => [
                            'id' => [
                                'type' => 'integer',
                                'required' => true,
                            ],
                            'price' => [
                                'type' => 'number',
                            ],
                            'stock' => [
                                'type' => 'integer',
                            ],
                        ],
                    ],
                ],
            ],
        ]);
    }
    
    public function get_featured_products($request) {
        $per_page = $request->get_param('per_page');
        $page = $request->get_param('page');
        
        $args = [
            'post_type' => 'product',
            'posts_per_page' => $per_page,
            'paged' => $page,
            'meta_query' => [
                [
                    'key' => 'is_featured',
                    'value' => 'yes',
                    'compare' => '=',
                ],
            ],
        ];
        
        $query = new WP_Query($args);
        $products = [];
        
        if ($query->have_posts()) {
            while ($query->have_posts()) {
                $query->the_post();
                $products[] = $this->prepare_product_for_response(get_post());
            }
            wp_reset_postdata();
        }
        
        return new WP_REST_Response([
            'products' => $products,
            'total' => $query->found_posts,
            'pages' => $query->max_num_pages,
            'current_page' => $page,
        ], 200);
    }
    
    public function get_product_by_sku($request) {
        $sku = $request->get_param('sku');
        
        $args = [
            'post_type' => 'product',
            'posts_per_page' => 1,
            'meta_query' => [
                [
                    'key' => 'sku',
                    'value' => $sku,
                    'compare' => '=',
                ],
            ],
        ];
        
        $query = new WP_Query($args);
        
        if (!$query->have_posts()) {
            return new WP_Error(
                'product_not_found',
                'Product with SKU ' . $sku . ' not found',
                ['status' => 404]
            );
        }
        
        $query->the_post();
        $product = $this->prepare_product_for_response(get_post());
        wp_reset_postdata();
        
        return new WP_REST_Response($product, 200);
    }
    
    public function search_products($request) {
        $args = [
            'post_type' => 'product',
            'posts_per_page' => 20,
            'post_status' => 'publish',
        ];
        
        // Keyword search
        if ($keyword = $request->get_param('keyword')) {
            $args['s'] = $keyword;
        }
        
        // Category filter
        if ($category = $request->get_param('category')) {
            $args['tax_query'] = [
                [
                    'taxonomy' => 'product_category',
                    'field' => 'term_id',
                    'terms' => $category,
                ],
            ];
        }
        
        // Price range filter
        $meta_query = [];
        if ($min_price = $request->get_param('min_price')) {
            $meta_query[] = [
                'key' => 'price',
                'value' => $min_price,
                'compare' => '>=',
                'type' => 'NUMERIC',
            ];
        }
        
        if ($max_price = $request->get_param('max_price')) {
            $meta_query[] = [
                'key' => 'price',
                'value' => $max_price,
                'compare' => '<=',
                'type' => 'NUMERIC',
            ];
        }
        
        if (!empty($meta_query)) {
            $args['meta_query'] = $meta_query;
        }
        
        // Sorting
        $sort = $request->get_param('sort');
        $order = $request->get_param('order');
        
        switch ($sort) {
            case 'price':
                $args['meta_key'] = 'price';
                $args['orderby'] = 'meta_value_num';
                break;
            case 'title':
                $args['orderby'] = 'title';
                break;
            case 'popularity':
                $args['meta_key'] = 'views';
                $args['orderby'] = 'meta_value_num';
                break;
            default:
                $args['orderby'] = 'date';
        }
        
        $args['order'] = $order;
        
        $query = new WP_Query($args);
        $products = [];
        
        if ($query->have_posts()) {
            while ($query->have_posts()) {
                $query->the_post();
                $products[] = $this->prepare_product_for_response(get_post());
            }
            wp_reset_postdata();
        }
        
        return new WP_REST_Response([
            'products' => $products,
            'total' => $query->found_posts,
        ], 200);
    }
    
    public function batch_update_products($request) {
        $products = $request->get_param('products');
        $updated = [];
        $errors = [];
        
        foreach ($products as $product_data) {
            $post_id = $product_data['id'];
            
            // Check if product exists
            if (get_post_type($post_id) !== 'product') {
                $errors[] = [
                    'id' => $post_id,
                    'error' => 'Product not found',
                ];
                continue;
            }
            
            // Update price
            if (isset($product_data['price'])) {
                update_post_meta($post_id, 'price', $product_data['price']);
            }
            
            // Update stock
            if (isset($product_data['stock'])) {
                update_post_meta($post_id, 'stock', $product_data['stock']);
            }
            
            $updated[] = $post_id;
        }
        
        return new WP_REST_Response([
            'updated' => $updated,
            'errors' => $errors,
        ], 200);
    }
    
    public function batch_update_permission($request) {
        return current_user_can('edit_posts');
    }
    
    private function prepare_product_for_response($post) {
        $featured_image = get_the_post_thumbnail_url($post->ID, 'full');
        
        return [
            'id' => $post->ID,
            'title' => $post->post_title,
            'slug' => $post->post_name,
            'content' => $post->post_content,
            'excerpt' => $post->post_excerpt,
            'featured_image' => $featured_image ?: null,
            'price' => get_post_meta($post->ID, 'price', true),
            'sku' => get_post_meta($post->ID, 'sku', true),
            'stock' => get_post_meta($post->ID, 'stock', true),
            'categories' => wp_get_post_terms($post->ID, 'product_category', ['fields' => 'names']),
            'link' => get_permalink($post->ID),
            'date' => $post->post_date,
            'modified' => $post->post_modified,
        ];
    }
}

// Register meta fields for REST API
class REST_Meta_Fields {
    
    public function __construct() {
        add_action('init', [$this, 'register_meta_fields']);
        add_action('rest_api_init', [$this, 'register_rest_fields']);
    }
    
    public function register_meta_fields() {
        // Register meta with REST API support
        register_post_meta('product', 'price', [
            'type' => 'number',
            'single' => true,
            'show_in_rest' => true,
            'auth_callback' => function() {
                return current_user_can('edit_posts');
            },
        ]);
        
        register_post_meta('product', 'sku', [
            'type' => 'string',
            'single' => true,
            'show_in_rest' => [
                'schema' => [
                    'type' => 'string',
                    'format' => 'text-field',
                    'description' => 'Product SKU',
                ],
            ],
        ]);
        
        register_post_meta('product', 'specifications', [
            'type' => 'object',
            'single' => true,
            'show_in_rest' => [
                'schema' => [
                    'type' => 'object',
                    'properties' => [
                        'weight' => [
                            'type' => 'string',
                        ],
                        'dimensions' => [
                            'type' => 'object',
                            'properties' => [
                                'length' => ['type' => 'number'],
                                'width' => ['type' => 'number'],
                                'height' => ['type' => 'number'],
                            ],
                        ],
                        'materials' => [
                            'type' => 'array',
                            'items' => [
                                'type' => 'string',
                            ],
                        ],
                    ],
                ],
            ],
        ]);
    }
    
    public function register_rest_fields() {
        // Add custom fields to REST response
        register_rest_field('product', 'formatted_price', [
            'get_callback' => [$this, 'get_formatted_price'],
            'update_callback' => null,
            'schema' => [
                'description' => 'Formatted price with currency',
                'type' => 'string',
                'context' => ['view', 'edit'],
            ],
        ]);
        
        register_rest_field('product', 'gallery', [
            'get_callback' => [$this, 'get_gallery_images'],
            'update_callback' => [$this, 'update_gallery_images'],
            'schema' => [
                'description' => 'Product gallery images',
                'type' => 'array',
                'items' => [
                    'type' => 'object',
                    'properties' => [
                        'id' => ['type' => 'integer'],
                        'url' => ['type' => 'string', 'format' => 'uri'],
                        'alt' => ['type' => 'string'],
                    ],
                ],
            ],
        ]);
        
        register_rest_field('product', 'related_products', [
            'get_callback' => [$this, 'get_related_products'],
            'schema' => [
                'description' => 'Related product IDs',
                'type' => 'array',
                'items' => [
                    'type' => 'integer',
                ],
            ],
        ]);
        
        // Add computed field
        register_rest_field('product', 'availability', [
            'get_callback' => [$this, 'get_availability_status'],
            'schema' => [
                'description' => 'Product availability status',
                'type' => 'object',
                'properties' => [
                    'status' => [
                        'type' => 'string',
                        'enum' => ['in_stock', 'out_of_stock', 'low_stock'],
                    ],
                    'quantity' => ['type' => 'integer'],
                    'message' => ['type' => 'string'],
                ],
            ],
        ]);
    }
    
    public function get_formatted_price($post, $field_name, $request) {
        $price = get_post_meta($post['id'], 'price', true);
        return $price ? '$' . number_format($price, 2) : null;
    }
    
    public function get_gallery_images($post, $field_name, $request) {
        $gallery_ids = get_post_meta($post['id'], 'gallery_images', true);
        
        if (!$gallery_ids) {
            return [];
        }
        
        $images = [];
        foreach ($gallery_ids as $image_id) {
            $images[] = [
                'id' => $image_id,
                'url' => wp_get_attachment_url($image_id),
                'alt' => get_post_meta($image_id, '_wp_attachment_image_alt', true),
            ];
        }
        
        return $images;
    }
    
    public function update_gallery_images($value, $post, $field_name) {
        if (!is_array($value)) {
            return new WP_Error(
                'invalid_gallery',
                'Gallery must be an array of image IDs',
                ['status' => 400]
            );
        }
        
        $image_ids = array_map('intval', $value);
        update_post_meta($post->ID, 'gallery_images', $image_ids);
        
        return true;
    }
    
    public function get_related_products($post, $field_name, $request) {
        $related_ids = get_post_meta($post['id'], 'related_products', true);
        
        if (!$related_ids) {
            // Auto-generate based on category
            $terms = wp_get_post_terms($post['id'], 'product_category', ['fields' => 'ids']);
            
            if (!empty($terms)) {
                $args = [
                    'post_type' => 'product',
                    'posts_per_page' => 4,
                    'post__not_in' => [$post['id']],
                    'tax_query' => [
                        [
                            'taxonomy' => 'product_category',
                            'field' => 'term_id',
                            'terms' => $terms,
                        ],
                    ],
                    'fields' => 'ids',
                ];
                
                $related_ids = get_posts($args);
            }
        }
        
        return $related_ids ?: [];
    }
    
    public function get_availability_status($post, $field_name, $request) {
        $stock = (int) get_post_meta($post['id'], 'stock', true);
        
        if ($stock <= 0) {
            return [
                'status' => 'out_of_stock',
                'quantity' => 0,
                'message' => 'Out of stock',
            ];
        } elseif ($stock <= 5) {
            return [
                'status' => 'low_stock',
                'quantity' => $stock,
                'message' => "Only {$stock} left in stock",
            ];
        } else {
            return [
                'status' => 'in_stock',
                'quantity' => $stock,
                'message' => 'In stock',
            ];
        }
    }
}

Authentication and Permissions

Custom Authentication


// Implement custom authentication for REST API
class REST_Authentication {
    
    public function __construct() {
        add_filter('rest_authentication_errors', [$this, 'check_authentication']);
        add_filter('rest_pre_dispatch', [$this, 'pre_dispatch'], 10, 3);
        add_action('rest_api_init', [$this, 'register_auth_routes']);
    }
    
    public function check_authentication($error) {
        // Allow authentication error to pass through
        if (!empty($error)) {
            return $error;
        }
        
        // Check for API key authentication
        if ($this->is_api_key_auth()) {
            $api_key = $this->get_api_key();
            $user = $this->validate_api_key($api_key);
            
            if (!$user) {
                return new WP_Error(
                    'invalid_api_key',
                    'Invalid API key provided',
                    ['status' => 401]
                );
            }
            
            // Set current user
            wp_set_current_user($user->ID);
            return true;
        }
        
        // Check for JWT authentication
        if ($this->is_jwt_auth()) {
            $token = $this->get_jwt_token();
            $user_id = $this->validate_jwt($token);
            
            if (!$user_id) {
                return new WP_Error(
                    'invalid_token',
                    'Invalid or expired token',
                    ['status' => 401]
                );
            }
            
            wp_set_current_user($user_id);
            return true;
        }
        
        return $error;
    }
    
    public function pre_dispatch($result, $server, $request) {
        // Log API requests
        $this->log_api_request($request);
        
        // Rate limiting
        if (!$this->check_rate_limit($request)) {
            return new WP_Error(
                'rate_limit_exceeded',
                'Too many requests',
                ['status' => 429]
            );
        }
        
        return $result;
    }
    
    public function register_auth_routes() {
        // Login endpoint
        register_rest_route('auth/v1', '/login', [
            'methods' => WP_REST_Server::CREATABLE,
            'callback' => [$this, 'login'],
            'permission_callback' => '__return_true',
            'args' => [
                'username' => [
                    'required' => true,
                    'sanitize_callback' => 'sanitize_text_field',
                ],
                'password' => [
                    'required' => true,
                ],
            ],
        ]);
        
        // Token refresh endpoint
        register_rest_route('auth/v1', '/refresh', [
            'methods' => WP_REST_Server::CREATABLE,
            'callback' => [$this, 'refresh_token'],
            'permission_callback' => '__return_true',
        ]);
        
        // API key generation
        register_rest_route('auth/v1', '/generate-key', [
            'methods' => WP_REST_Server::CREATABLE,
            'callback' => [$this, 'generate_api_key'],
            'permission_callback' => function() {
                return is_user_logged_in();
            },
        ]);
    }
    
    public function login($request) {
        $username = $request->get_param('username');
        $password = $request->get_param('password');
        
        $user = wp_authenticate($username, $password);
        
        if (is_wp_error($user)) {
            return new WP_Error(
                'invalid_credentials',
                'Invalid username or password',
                ['status' => 401]
            );
        }
        
        // Generate JWT token
        $token = $this->generate_jwt($user->ID);
        
        return new WP_REST_Response([
            'token' => $token,
            'user' => [
                'id' => $user->ID,
                'email' => $user->user_email,
                'name' => $user->display_name,
            ],
            'expires_in' => 3600,
        ], 200);
    }
    
    public function refresh_token($request) {
        $token = $this->get_jwt_token();
        
        if (!$token) {
            return new WP_Error(
                'no_token',
                'No token provided',
                ['status' => 400]
            );
        }
        
        $user_id = $this->validate_jwt($token, true); // Allow expired for refresh
        
        if (!$user_id) {
            return new WP_Error(
                'invalid_token',
                'Invalid token',
                ['status' => 401]
            );
        }
        
        // Generate new token
        $new_token = $this->generate_jwt($user_id);
        
        return new WP_REST_Response([
            'token' => $new_token,
            'expires_in' => 3600,
        ], 200);
    }
    
    public function generate_api_key($request) {
        $user_id = get_current_user_id();
        $api_key = wp_generate_password(32, false);
        
        // Store API key (hashed)
        update_user_meta($user_id, 'api_key', wp_hash($api_key));
        update_user_meta($user_id, 'api_key_created', time());
        
        return new WP_REST_Response([
            'api_key' => $api_key,
            'message' => 'Store this key securely. It will not be shown again.',
        ], 200);
    }
    
    private function is_api_key_auth() {
        return isset($_SERVER['HTTP_X_API_KEY']);
    }
    
    private function get_api_key() {
        return $_SERVER['HTTP_X_API_KEY'] ?? '';
    }
    
    private function validate_api_key($api_key) {
        if (empty($api_key)) {
            return false;
        }
        
        // Find user with this API key
        $users = get_users([
            'meta_key' => 'api_key',
            'meta_value' => wp_hash($api_key),
            'number' => 1,
        ]);
        
        return !empty($users) ? $users[0] : false;
    }
    
    private function is_jwt_auth() {
        return isset($_SERVER['HTTP_AUTHORIZATION']);
    }
    
    private function get_jwt_token() {
        $auth_header = $_SERVER['HTTP_AUTHORIZATION'] ?? '';
        
        if (preg_match('/Bearer\s+(.*)$/i', $auth_header, $matches)) {
            return $matches[1];
        }
        
        return null;
    }
    
    private function generate_jwt($user_id) {
        // Simple JWT implementation (use a proper library in production)
        $header = json_encode(['typ' => 'JWT', 'alg' => 'HS256']);
        $payload = json_encode([
            'user_id' => $user_id,
            'exp' => time() + 3600,
            'iat' => time(),
        ]);
        
        $base64Header = str_replace(['+', '/', '='], ['-', '_', ''], base64_encode($header));
        $base64Payload = str_replace(['+', '/', '='], ['-', '_', ''], base64_encode($payload));
        
        $signature = hash_hmac('sha256', $base64Header . '.' . $base64Payload, AUTH_KEY, true);
        $base64Signature = str_replace(['+', '/', '='], ['-', '_', ''], base64_encode($signature));
        
        return $base64Header . '.' . $base64Payload . '.' . $base64Signature;
    }
    
    private function validate_jwt($token, $allow_expired = false) {
        $parts = explode('.', $token);
        
        if (count($parts) !== 3) {
            return false;
        }
        
        $payload = json_decode(base64_decode($parts[1]), true);
        
        if (!$allow_expired && $payload['exp'] < time()) {
            return false;
        }
        
        // Verify signature
        $signature = hash_hmac('sha256', $parts[0] . '.' . $parts[1], AUTH_KEY, true);
        $base64Signature = str_replace(['+', '/', '='], ['-', '_', ''], base64_encode($signature));
        
        if ($base64Signature !== $parts[2]) {
            return false;
        }
        
        return $payload['user_id'];
    }
    
    private function check_rate_limit($request) {
        $identifier = $this->get_client_identifier();
        $key = 'api_rate_limit_' . $identifier;
        $limit = 100; // 100 requests per hour
        
        $count = get_transient($key);
        
        if ($count === false) {
            set_transient($key, 1, HOUR_IN_SECONDS);
            return true;
        }
        
        if ($count >= $limit) {
            return false;
        }
        
        set_transient($key, $count + 1, HOUR_IN_SECONDS);
        return true;
    }
    
    private function get_client_identifier() {
        if (is_user_logged_in()) {
            return 'user_' . get_current_user_id();
        }
        
        return 'ip_' . $_SERVER['REMOTE_ADDR'];
    }
    
    private function log_api_request($request) {
        $log_data = [
            'timestamp' => current_time('mysql'),
            'method' => $request->get_method(),
            'route' => $request->get_route(),
            'params' => $request->get_params(),
            'user' => get_current_user_id(),
            'ip' => $_SERVER['REMOTE_ADDR'],
        ];
        
        // Log to custom table or file
        error_log('API Request: ' . json_encode($log_data));
    }
}

// JavaScript client for consuming WordPress REST API
class WordPressAPI {
    constructor(baseURL) {
        this.baseURL = baseURL;
        this.token = localStorage.getItem('wp_token');
    }
    
    async request(endpoint, options = {}) {
        const url = `${this.baseURL}/wp-json${endpoint}`;
        
        const headers = {
            'Content-Type': 'application/json',
            ...options.headers,
        };
        
        if (this.token) {
            headers['Authorization'] = `Bearer ${this.token}`;
        }
        
        try {
            const response = await fetch(url, {
                ...options,
                headers,
            });
            
            if (!response.ok) {
                throw new Error(`HTTP error! status: ${response.status}`);
            }
            
            return await response.json();
        } catch (error) {
            console.error('API request failed:', error);
            throw error;
        }
    }
    
    // Authentication
    async login(username, password) {
        const response = await this.request('/auth/v1/login', {
            method: 'POST',
            body: JSON.stringify({ username, password }),
        });
        
        this.token = response.token;
        localStorage.setItem('wp_token', this.token);
        
        return response;
    }
    
    async logout() {
        this.token = null;
        localStorage.removeItem('wp_token');
    }
    
    // Products
    async getProducts(params = {}) {
        const queryString = new URLSearchParams(params).toString();
        return this.request(`/wp/v2/products?${queryString}`);
    }
    
    async getProduct(id) {
        return this.request(`/wp/v2/products/${id}`);
    }
    
    async createProduct(data) {
        return this.request('/wp/v2/products', {
            method: 'POST',
            body: JSON.stringify(data),
        });
    }
    
    async updateProduct(id, data) {
        return this.request(`/wp/v2/products/${id}`, {
            method: 'PUT',
            body: JSON.stringify(data),
        });
    }
    
    async deleteProduct(id) {
        return this.request(`/wp/v2/products/${id}`, {
            method: 'DELETE',
        });
    }
    
    // Custom endpoints
    async getFeaturedProducts(perPage = 10) {
        return this.request(`/myapi/v1/products/featured?per_page=${perPage}`);
    }
    
    async searchProducts(filters) {
        const queryString = new URLSearchParams(filters).toString();
        return this.request(`/myapi/v1/products/search?${queryString}`);
    }
    
    async getProductBySKU(sku) {
        return this.request(`/myapi/v1/products/sku/${sku}`);
    }
}

// React component using WordPress REST API
import React, { useState, useEffect } from 'react';
import WordPressAPI from './WordPressAPI';

const ProductList = () => {
    const [products, setProducts] = useState([]);
    const [loading, setLoading] = useState(true);
    const [filters, setFilters] = useState({
        category: '',
        min_price: '',
        max_price: '',
    });
    
    const api = new WordPressAPI(process.env.REACT_APP_WP_URL);
    
    useEffect(() => {
        fetchProducts();
    }, [filters]);
    
    const fetchProducts = async () => {
        setLoading(true);
        try {
            const response = await api.searchProducts(filters);
            setProducts(response.products);
        } catch (error) {
            console.error('Failed to fetch products:', error);
        } finally {
            setLoading(false);
        }
    };
    
    const handleFilterChange = (e) => {
        const { name, value } = e.target;
        setFilters(prev => ({
            ...prev,
            [name]: value,
        }));
    };
    
    if (loading) {
        return <div className="loading">Loading products...</div>;
    }
    
    return (
        <div className="product-list">
            <div className="filters">
                <input
                    type="number"
                    name="min_price"
                    placeholder="Min Price"
                    value={filters.min_price}
                    onChange={handleFilterChange}
                />
                <input
                    type="number"
                    name="max_price"
                    placeholder="Max Price"
                    value={filters.max_price}
                    onChange={handleFilterChange}
                />
            </div>
            
            <div className="products">
                {products.map(product => (
                    <ProductCard key={product.id} product={product} />
                ))}
            </div>
        </div>
    );
};

const ProductCard = ({ product }) => {
    return (
        <div className="product-card">
            {product.featured_image && (
                <img src={product.featured_image} alt={product.title} />
            )}
            <h3>{product.title}</h3>
            <p className="price">{product.formatted_price}</p>
            <p className="availability">
                {product.availability.message}
            </p>
            <button>View Details</button>
        </div>
    );
};

export default ProductList;
