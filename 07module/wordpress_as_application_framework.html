WordPress as an Application Framework

Transform WordPress into a powerful application platform

Learning Objectives

By the end of this lesson, you will be able to:

  • Understand WordPress beyond traditional content management

WordPress Beyond Content Management

While WordPress began as a blogging platform and evolved into a full-featured CMS, it has become much more. Today, WordPress powers:

  • E-commerce platforms (WooCommerce powers 30% of online stores)
  • Learning Management Systems (LMS)
  • Customer Relationship Management (CRM) systems
  • Social networks and communities
  • SaaS applications
  • Mobile app backends
  • Enterprise applications

2. Library Pattern

Create a shared library for common functionality:


// wp-content/mu-plugins/app-library.php
namespace AppLibrary;

class Application {
    private static $instance = null;
    private $modules = [];
    
    public static function getInstance() {
        if (self::$instance === null) {
            self::$instance = new self();
        }
        return self::$instance;
    }
    
    public function registerModule($name, $module) {
        $this->modules[$name] = $module;
    }
    
    public function getModule($name) {
        return $this->modules[$name] ?? null;
    }
}

3. Service Container Pattern

Implement dependency injection for better testability:


class Container {
    private $services = [];
    private $factories = [];
    
    public function register($name, callable $factory) {
        $this->factories[$name] = $factory;
    }
    
    public function get($name) {
        if (!isset($this->services[$name])) {
            if (isset($this->factories[$name])) {
                $this->services[$name] = $this->factories[$name]($this);
            }
        }
        return $this->services[$name] ?? null;
    }
}

// Usage
$container = new Container();
$container->register('database', function($c) {
    return new DatabaseService();
});
$container->register('api', function($c) {
    return new ApiService($c->get('database'));
});

Model-View-Controller (MVC) Concepts in WordPress

While WordPress doesn't follow strict MVC, you can implement MVC patterns:

Model Layer

Handle data and business logic:


// models/class-product-model.php
namespace App\Models;

class ProductModel {
    private $table_name;
    
    public function __construct() {
        global $wpdb;
        $this->table_name = $wpdb->prefix . 'products';
    }
    
    public function find($id) {
        global $wpdb;
        return $wpdb->get_row($wpdb->prepare(
            "SELECT * FROM {$this->table_name} WHERE id = %d",
            $id
        ));
    }
    
    public function all($args = []) {
        global $wpdb;
        $defaults = [
            'orderby' => 'name',
            'order' => 'ASC',
            'limit' => 10
        ];
        $args = wp_parse_args($args, $defaults);
        
        return $wpdb->get_results($wpdb->prepare(
            "SELECT * FROM {$this->table_name} 
             ORDER BY %s %s LIMIT %d",
            $args['orderby'],
            $args['order'],
            $args['limit']
        ));
    }
    
    public function save($data) {
        global $wpdb;
        
        if (isset($data['id'])) {
            return $wpdb->update(
                $this->table_name,
                $data,
                ['id' => $data['id']]
            );
        } else {
            return $wpdb->insert($this->table_name, $data);
        }
    }
}

View Layer

Separate presentation logic:


// views/class-view.php
namespace App\Views;

class View {
    private $template_path;
    
    public function __construct($template_path) {
        $this->template_path = $template_path;
    }
    
    public function render($template, $data = []) {
        extract($data);
        include $this->template_path . '/' . $template . '.php';
    }
    
    public function get($template, $data = []) {
        ob_start();
        $this->render($template, $data);
        return ob_get_clean();
    }
}

Controller Layer

Handle requests and coordinate between Model and View:


// controllers/class-product-controller.php
namespace App\Controllers;

use App\Models\ProductModel;
use App\Views\View;

class ProductController {
    private $model;
    private $view;
    
    public function __construct() {
        $this->model = new ProductModel();
        $this->view = new View(plugin_dir_path(__FILE__) . '../views/templates');
        
        add_action('init', [$this, 'registerRoutes']);
    }
    
    public function registerRoutes() {
        add_rewrite_rule(
            '^products/?$',
            'index.php?app_route=products',
            'top'
        );
        
        add_filter('query_vars', function($vars) {
            $vars[] = 'app_route';
            return $vars;
        });
        
        add_action('template_redirect', [$this, 'handleRequest']);
    }
    
    public function handleRequest() {
        $route = get_query_var('app_route');
        
        if ($route === 'products') {
            $this->index();
            exit;
        }
    }
    
    public function index() {
        $products = $this->model->all();
        
        get_header();
        echo $this->view->get('product-list', [
            'products' => $products
        ]);
        get_footer();
    }
}

Separating Business Logic from Presentation

Service Layer Pattern

Create services to encapsulate business logic:


// src/Services/OrderService.php
namespace App\Services;

class OrderService {
    private $repository;
    private $emailService;
    private $inventoryService;
    
    public function __construct(
        OrderRepository $repository,
        EmailService $emailService,
        InventoryService $inventoryService
    ) {
        $this->repository = $repository;
        $this->emailService = $emailService;
        $this->inventoryService = $inventoryService;
    }
    
    public function createOrder(array $data) {
        // Business logic
        $order = $this->validateOrderData($data);
        
        // Check inventory
        if (!$this->inventoryService->checkAvailability($order['items'])) {
            throw new \Exception('Items not available');
        }
        
        // Save order
        $orderId = $this->repository->create($order);
        
        // Update inventory
        $this->inventoryService->decrementStock($order['items']);
        
        // Send confirmation email
        $this->emailService->sendOrderConfirmation($orderId);
        
        // Trigger WordPress action
        do_action('app_order_created', $orderId, $order);
        
        return $orderId;
    }
    
    private function validateOrderData($data) {
        // Validation logic
        return $data;
    }
}

Building Reusable Components

Component-Based Architecture

Create modular, reusable components:


// src/Components/DataTable.php
namespace App\Components;

class DataTable {
    private $columns = [];
    private $data = [];
    private $options = [];
    
    public function __construct($options = []) {
        $this->options = wp_parse_args($options, [
            'sortable' => true,
            'searchable' => true,
            'paginated' => true,
            'items_per_page' => 10
        ]);
    }
    
    public function addColumn($key, $label, $options = []) {
        $this->columns[$key] = [
            'label' => $label,
            'sortable' => $options['sortable'] ?? true,
            'callback' => $options['callback'] ?? null
        ];
        return $this;
    }
    
    public function setData($data) {
        $this->data = $data;
        return $this;
    }
    
    public function render() {
        ob_start();
        ?>
        <div class="app-data-table">
            <?php if ($this->options['searchable']): ?>
                <input type="text" class="table-search" placeholder="Search...">
            <?php endif; ?>
            
            <table class="wp-list-table widefat fixed striped">
                <thead>
                    <tr>
                        <?php foreach ($this->columns as $key => $column): ?>
                            <th class="<?php echo $column['sortable'] ? 'sortable' : ''; ?>">
                                <?php echo esc_html($column['label']); ?>
                            </th>
                        <?php endforeach; ?>
                    </tr>
                </thead>
                <tbody>
                    <?php foreach ($this->data as $row): ?>
                        <tr>
                            <?php foreach ($this->columns as $key => $column): ?>
                                <td>
                                    <?php 
                                    if ($column['callback']) {
                                        echo call_user_func($column['callback'], $row, $key);
                                    } else {
                                        echo esc_html($row->$key ?? '');
                                    }
                                    ?>
                                </td>
                            <?php endforeach; ?>
                        </tr>
                    <?php endforeach; ?>
                </tbody>
            </table>
            
            <?php if ($this->options['paginated']): ?>
                <div class="table-pagination">
                    <!-- Pagination controls -->
                </div>
            <?php endif; ?>
        </div>
        <?php
        return ob_get_clean();
    }
}

// Usage
$table = new DataTable(['searchable' => true]);
$table->addColumn('name', 'Product Name')

          'callback' => function($row) {
              return '$' . number_format($row->price, 2);
          }
      ])

Trait for Common Functionality

Use traits to share functionality across classes:


// src/Traits/Singleton.php
namespace App\Traits;

trait Singleton {
    private static $instance = null;
    
    public static function getInstance() {
        if (self::$instance === null) {
            self::$instance = new static();
        }
        return self::$instance;
    }
    
    private function __construct() {
        $this->init();
    }
    
    private function __clone() {}
    private function __wakeup() {}
    
    abstract protected function init();
}

// Usage
class Application {
    use Singleton;
    
    protected function init() {
        // Initialize application
    }
}

Let's build a simple task management application structure:


// Main plugin file: task-manager.php
<?php
/**
 * Plugin Name: Task Manager Application
 * Description: A WordPress-powered task management application
 */

namespace TaskManager;

// Autoload classes
spl_autoload_register(function ($class) {
    $prefix = 'TaskManager\\';
    $base_dir = __DIR__ . '/src/';
    
    $len = strlen($prefix);
    if (strncmp($prefix, $class, $len) !== 0) {
        return;
    }
    
    $relative_class = substr($class, $len);
    $file = $base_dir . str_replace('\\', '/', $relative_class) . '.php';
    
    if (file_exists($file)) {
        require $file;
    }
});

// Initialize application
add_action('plugins_loaded', function() {
    $app = new Application();
    $app->run();
});

// src/Application.php
namespace TaskManager;

class Application {
    private $services = [];
    
    public function run() {
