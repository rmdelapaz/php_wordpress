Field Registration and Visibility

Master advanced field management in WordPress REST API

Learning Objectives

By the end of this lesson, you will be able to:

  • Register complex field schemas
  • Control field visibility by context
  • Implement dynamic field registration
  • Create computed and virtual fields
  • Handle field dependencies

Advanced Field Registration

Complex Field Schemas


/**
 * Advanced field registration with complex schemas
 */
class Advanced_Field_Registration {
    
    public function __construct() {
        add_action('rest_api_init', [$this, 'register_complex_fields']);
    }
    
    /**
     * Register fields with complex schemas
     */
    public function register_complex_fields() {
        // Nested object field
        register_rest_field('post', 'metadata', [
            'get_callback' => [$this, 'get_metadata'],
            'update_callback' => [$this, 'update_metadata'],
            'schema' => $this->get_metadata_schema(),
        ]);
        
        // Array of objects field
        register_rest_field('post', 'gallery', [
            'get_callback' => [$this, 'get_gallery'],
            'update_callback' => [$this, 'update_gallery'],
            'schema' => [
                'description' => 'Image gallery',
                'type' => 'array',
                'context' => ['view', 'edit'],
                'items' => [
                    'type' => 'object',
                    'properties' => [
                        'id' => [
                            'type' => 'integer',
                            'description' => 'Attachment ID',
                        ],
                        'url' => [
                            'type' => 'string',
                            'format' => 'uri',
                            'description' => 'Image URL',
                        ],
                        'alt' => [
                            'type' => 'string',
                            'description' => 'Alt text',
                        ],
                        'caption' => [
                            'type' => 'string',
                            'description' => 'Image caption',
                        ],
                        'sizes' => [
                            'type' => 'object',
                            'properties' => [
                                'thumbnail' => [
                                    'type' => 'string',
                                    'format' => 'uri',
                                ],
                                'medium' => [
                                    'type' => 'string',
                                    'format' => 'uri',
                                ],
                                'large' => [
                                    'type' => 'string',
                                    'format' => 'uri',
                                ],
                                'full' => [
                                    'type' => 'string',
                                    'format' => 'uri',
                                ],
                            ],
                        ],
                    ],
                ],
            ],
        ]);
        
        // Polymorphic field (can be multiple types)
        register_rest_field('post', 'flexible_content', [
            'get_callback' => [$this, 'get_flexible_content'],
            'update_callback' => [$this, 'update_flexible_content'],
            'schema' => [
                'description' => 'Flexible content blocks',
                'type' => 'array',
                'context' => ['view', 'edit'],
                'items' => [
                    'oneOf' => [
                        [
                            'type' => 'object',
                            'properties' => [
                                'type' => [
                                    'type' => 'string',
                                    'enum' => ['text'],
                                ],
                                'content' => [
                                    'type' => 'string',
                                ],
                            ],
                        ],
                        [
                            'type' => 'object',
                            'properties' => [
                                'type' => [
                                    'type' => 'string',
                                    'enum' => ['image'],
                                ],
                                'id' => [
                                    'type' => 'integer',
                                ],
                                'url' => [
                                    'type' => 'string',
                                    'format' => 'uri',
                                ],
                            ],
                        ],
                        [
                            'type' => 'object',
                            'properties' => [
                                'type' => [
                                    'type' => 'string',
                                    'enum' => ['video'],
                                ],
                                'url' => [
                                    'type' => 'string',
                                    'format' => 'uri',

                                    'enum' => ['youtube', 'vimeo', 'self'],

        $post_types = get_post_types(['public' => true]);

                'get_callback' => function($post) use ($post_type) {

        }
    }
    
    /**
     * Get metadata schema
     */
    private function get_metadata_schema() {
        return [
            'description' => 'Post metadata',
            'type' => 'object',
            'context' => ['view', 'edit'],
            'properties' => [
                'seo' => [
                    'type' => 'object',
                    'properties' => [
                        'title' => [

                        'unique_visitors' => ['type' => 'integer', 'minimum' => 0],

                    ],

                            ],
                        ],
                    ],
                ],
            ],
            'additionalProperties' => false,
        ];
    }
    
    /**
     * Field callbacks
     */
    public function get_metadata($post) {
        $metadata = get_post_meta($post['id'], '_metadata', true);
        
        if (!$metadata) {
            // Return default structure
            return [
                'seo' => [
                    'title' => '',
                    'description' => '',
                    'keywords' => [],
                    'og_image' => 0,
                    'robots' => [
                        'index' => true,
                        'follow' => true,
                        'archive' => true,
                    ],
                ],
                'analytics' => [
                    'views' => 0,
                    'unique_visitors' => 0,
                    'avg_time' => 0,
                    'bounce_rate' => 0,
                ],
                'social' => [
                    'shares' => [
                        'facebook' => 0,
                        'twitter' => 0,
                        'linkedin' => 0,
                    ],
                ],
            ];
        }
        
        return $metadata;
    }
    
    public function update_metadata($value, $post) {
        // Validate nested structure
        $validated = $this->validate_nested_data($value, $this->get_metadata_schema());
        
        if (is_wp_error($validated)) {
            return false;
        }
        
        return update_post_meta($post->ID, '_metadata', $validated);
    }
    
    /**
     * Validate nested data against schema
     */
    private function validate_nested_data($data, $schema) {
        if ($schema['type'] === 'object' && isset($schema['properties'])) {
            $validated = [];
            
            foreach ($schema['properties'] as $key => $prop_schema) {
                if (isset($data[$key])) {
                    if ($prop_schema['type'] === 'object') {
                        $validated[$key] = $this->validate_nested_data($data[$key], $prop_schema);
                    } elseif ($prop_schema['type'] === 'array') {
                        $validated[$key] = $this->validate_array_data($data[$key], $prop_schema);
                    } else {
                        $validated[$key] = $this->validate_scalar_data($data[$key], $prop_schema);
                    }
                }
            }
            
            return $validated;
        }
        
        return $data;
    }
    
    private function validate_array_data($data, $schema) {
        if (!is_array($data)) {
            return [];
        }
        
        $validated = [];
        
        foreach ($data as $item) {
            if (isset($schema['items'])) {
                if ($schema['items']['type'] === 'object') {
                    $validated[] = $this->validate_nested_data($item, $schema['items']);
                } else {
                    $validated[] = $this->validate_scalar_data($item, $schema['items']);
                }
            } else {
                $validated[] = $item;
            }
        }
        
        // Apply array constraints
        if (isset($schema['maxItems']) && count($validated) > $schema['maxItems']) {
            $validated = array_slice($validated, 0, $schema['maxItems']);
        }
        
        if (isset($schema['uniqueItems']) && $schema['uniqueItems']) {
            $validated = array_unique($validated);
        }
        
        return $validated;
    }
    
    private function validate_scalar_data($data, $schema) {
        switch ($schema['type']) {
            case 'string':
                $data = (string) $data;
                if (isset($schema['maxLength'])) {
                    $data = substr($data, 0, $schema['maxLength']);
                }
                if (isset($schema['enum']) && !in_array($data, $schema['enum'])) {
                    $data = $schema['enum'][0];
                }
                break;
                
            case 'integer':
                $data = (int) $data;
                if (isset($schema['minimum'])) {
                    $data = max($data, $schema['minimum']);
                }
                if (isset($schema['maximum'])) {
                    $data = min($data, $schema['maximum']);
                }
                break;
                
            case 'number':
                $data = (float) $data;
                if (isset($schema['minimum'])) {
                    $data = max($data, $schema['minimum']);
                }
                if (isset($schema['maximum'])) {
                    $data = min($data, $schema['maximum']);
                }
                break;
                
            case 'boolean':
                $data = (bool) $data;
                break;
        }
        
        return $data;
    }
}

Context-Based Field Visibility

Dynamic Context Management


/**
 * Context-aware field visibility
 */
class Context_Field_Visibility {
    
    public function __construct() {
        add_action('rest_api_init', [$this, 'register_context_fields']);
    }
    
    /**
     * Register fields with context-based visibility
     */
    public function register_context_fields() {
        // Register field that changes based on context
        register_rest_field('post', 'contextual_data', [
            'get_callback' => [$this, 'get_contextual_data'],
            'schema' => [
                'description' => 'Context-dependent data',
                'type' => 'object',
                'context' => ['view', 'edit', 'embed'],
                'properties' => [
                    'summary' => [
                        'type' => 'string',
                        'context' => ['view', 'embed'],
                    ],
                    'full_content' => [
                        'type' => 'string',
                        'context' => ['view'],
                    ],
                    'edit_data' => [
                        'type' => 'object',
                        'context' => ['edit'],
                    ],
                ],
            ],
        ]);
        
        // Multi-context field handler
        register_rest_field('post', 'adaptive_field', [
            'get_callback' => [$this, 'get_adaptive_field'],
            'schema' => [
                'description' => 'Adaptive field based on context',
                'type' => ['string', 'object', 'array'],
                'context' => ['view', 'edit', 'embed'],
            ],
        ]);
    }
    
    /**
     * Get data based on context
     */
    public function get_contextual_data($post, $field_name, $request) {
        $context = $request['context'] ?? 'view';
        $data = [];
        
        // Always include basic data
        if (in_array($context, ['view', 'edit', 'embed'])) {
            $data['summary'] = wp_trim_words(get_the_content(null, false, $post['id']), 20);
        }
        
        // Include full content only in view context
        if ($context === 'view') {
            $data['full_content'] = get_the_content(null, false, $post['id']);
            $data['related_links'] = $this->get_related_links($post['id']);
            $data['media_attachments'] = $this->get_media_attachments($post['id']);
        }
        
        // Include edit data only in edit context
        if ($context === 'edit' && current_user_can('edit_post', $post['id'])) {
            $data['edit_data'] = [
                'last_modified_by' => get_post_meta($post['id'], '_edit_last', true),
                'revision_count' => wp_get_post_revisions($post['id'], ['fields' => 'ids']),
                'locked_by' => wp_check_post_lock($post['id']),
                'autosave' => wp_get_post_autosave($post['id'], get_current_user_id()),
            ];
        }
        
        // Minimal data for embed context
        if ($context === 'embed') {
            // Remove unnecessary data for embeds
            unset($data['full_content']);
            unset($data['edit_data']);
        }
        
        return $data;
    }
    
    /**
     * Adaptive field that returns different data types based on context
     */
    public function get_adaptive_field($post, $field_name, $request) {
        $context = $request['context'] ?? 'view';
        
        switch ($context) {
            case 'embed':
                // Return simple string for embeds
                return get_the_title($post['id']);
                
            case 'view':
                // Return object for view context
                return [
                    'title' => get_the_title($post['id']),
                    'excerpt' => get_the_excerpt($post['id']),
                    'thumbnail' => get_the_post_thumbnail_url($post['id']),
                ];
                
            case 'edit':
                // Return detailed array for edit context
                return [
                    'raw' => [
                        'title' => get_post_field('post_title', $post['id'], 'raw'),
                        'content' => get_post_field('post_content', $post['id'], 'raw'),
                    ],
                    'rendered' => [
                        'title' => get_the_title($post['id']),
                        'content' => apply_filters('the_content', get_the_content(null, false, $post['id'])),
                    ],
                    'meta' => get_post_meta($post['id']),
                ];
                
            default:
                return null;
        }
    }
    
    /**
     * Helper methods
     */
    private function get_related_links($post_id) {
        $links = [];
        
        // Get internal links
        $content = get_post_field('post_content', $post_id);
        preg_match_all('/]+href=["\'](https?:\/\/[^"\']+)["\'][^>]*>/i', $content, $matches);
        
        if (!empty($matches[1])) {
            foreach ($matches[1] as $url) {
                $links[] = [
                    'url' => $url,
                    'type' => $this->determine_link_type($url),
                ];
            }
        }
        
        return $links;
    }
    
    private function get_media_attachments($post_id) {
        $attachments = get_children([
            'post_parent' => $post_id,
            'post_type' => 'attachment',
            'posts_per_page' => -1,
        ]);
        
        $media = [];
        
        foreach ($attachments as $attachment) {
            $media[] = [
                'id' => $attachment->ID,
                'type' => $attachment->post_mime_type,
                'url' => wp_get_attachment_url($attachment->ID),
                'title' => $attachment->post_title,
            ];
        }
        
        return $media;
    }
    
    private function determine_link_type($url) {
        $site_url = get_site_url();
        
        if (strpos($url, $site_url) === 0) {
            return 'internal';
        }
        
        return 'external';
    }
}

Dynamic Field Registration

Runtime Field Registration


/**
 * Dynamic field registration based on conditions
 */
class Dynamic_Field_Registration {
    
    private $dynamic_fields = [];
    
    public function __construct() {
        add_action('init', [$this, 'determine_fields']);
        add_action('rest_api_init', [$this, 'register_dynamic_fields']);
    }
    
    /**
     * Determine which fields to register based on conditions
     */
    public function determine_fields() {
        // Register fields based on active plugins
        if (is_plugin_active('woocommerce/woocommerce.php')) {
            $this->dynamic_fields['product'] = [
                'price' => [
                    'type' => 'number',
                    'description' => 'Product price',
                ],
                'stock' => [
                    'type' => 'integer',
                    'description' => 'Stock quantity',
                ],
            ];
        }
        
        // Register fields based on theme support
        if (current_theme_supports('post-formats')) {
            $this->dynamic_fields['post_format_data'] = [
                'format' => [
                    'type' => 'string',
                    'enum' => get_post_format_strings(),
                ],
                'format_meta' => [
                    'type' => 'object',
                ],
            ];
        }
        
        // Register fields based on user capabilities
        if (current_user_can('manage_options')) {
            $this->dynamic_fields['admin_only'] = [
                'internal_notes' => [
                    'type' => 'string',
                ],
                'priority' => [
                    'type' => 'integer',
                    'minimum' => 1,
                    'maximum' => 10,
                ],
            ];
        }
        
        // Register fields based on settings
        $api_settings = get_option('rest_api_settings', []);
        
        if (!empty($api_settings['enable_analytics'])) {
            $this->dynamic_fields['analytics'] = [
                'pageviews' => ['type' => 'integer'],
                'unique_visitors' => ['type' => 'integer'],
                'engagement_rate' => ['type' => 'number'],
            ];
        }
    }
    
    /**
     * Register the dynamic fields
     */
    public function register_dynamic_fields() {
        foreach ($this->dynamic_fields as $field_name => $field_config) {
            $this->register_dynamic_field($field_name, $field_config);
        }
        
        // Register fields from database
        $this->register_database_fields();
        
        // Register fields from configuration files
        $this->register_config_fields();
    }
    
    /**
     * Register a single dynamic field
     */
    private function register_dynamic_field($field_name, $config) {
        $post_types = get_post_types(['public' => true]);
        
        foreach ($post_types as $post_type) {
            register_rest_field($post_type, $field_name, [
                'get_callback' => function($post) use ($field_name, $config) {
                    return $this->get_dynamic_field_value($post, $field_name, $config);
                },
                'update_callback' => function($value, $post) use ($field_name, $config) {
                    return $this->update_dynamic_field_value($value, $post, $field_name, $config);
                },
                'schema' => [
                    'description' => $config['description'] ?? 'Dynamic field',
                    'type' => $this->determine_schema_type($config),
                    'properties' => $this->build_schema_properties($config),
                ],
            ]);
        }
    }
    
    /**
     * Register fields from database
     */
    private function register_database_fields() {
        global $wpdb;
        
        // Get custom field definitions from database
        $fields = $wpdb->get_results(
            "SELECT * FROM {$wpdb->prefix}rest_field_definitions WHERE active = 1"
        );
        
        foreach ($fields as $field) {
            $config = json_decode($field->config, true);
            
            register_rest_field($field->object_type, $field->field_name, [
                'get_callback' => function($object) use ($field) {
                    return get_metadata($field->object_type, $object['id'], $field->meta_key, true);
                },
                'update_callback' => function($value, $object) use ($field) {
                    return update_metadata($field->object_type, $object->ID, $field->meta_key, $value);
                },
                'schema' => $config['schema'] ?? [],
            ]);
        }
    }
    
    /**
     * Register fields from configuration files
     */
    private function register_config_fields() {
        $config_file = WP_CONTENT_DIR . '/rest-fields.json';
        
        if (file_exists($config_file)) {
            $config = json_decode(file_get_contents($config_file), true);
            
            foreach ($config['fields'] as $field) {
                register_rest_field($field['object_type'], $field['name'], [
                    'get_callback' => $this->create_get_callback($field),
                    'update_callback' => $this->create_update_callback($field),
                    'schema' => $field['schema'],
                ]);
            }
        }
    }
    
    /**
     * Create dynamic get callback
     */
    private function create_get_callback($field_config) {
        return function($object) use ($field_config) {
            $value = null;
            
            switch ($field_config['source']) {
                case 'meta':
                    $value = get_post_meta($object['id'], $field_config['meta_key'], true);
                    break;
                    
                case 'option':
                    $value = get_option($field_config['option_key']);
                    break;
                    
                case 'callback':
                    if (is_callable($field_config['callback'])) {
                        $value = call_user_func($field_config['callback'], $object);
                    }
                    break;
                    
                case 'computed':
                    $value = $this->compute_field_value($object, $field_config);
                    break;
            }
            
            return $value;
        };
    }
    
    /**
     * Create dynamic update callback
     */
    private function create_update_callback($field_config) {
        return function($value, $object) use ($field_config) {
            switch ($field_config['source']) {
                case 'meta':
                    return update_post_meta($object->ID, $field_config['meta_key'], $value);
                    
                case 'option':
                    return update_option($field_config['option_key'], $value);
                    
                case 'callback':
                    if (is_callable($field_config['update_callback'])) {
                        return call_user_func($field_config['update_callback'], $value, $object);
                    }
                    break;
            }
            
            return false;
        };
    }
    
    /**
     * Compute field value dynamically
     */
    private function compute_field_value($object, $config) {
        $formula = $config['formula'];
        
        // Replace placeholders with actual values
        preg_match_all('/\{([^}]+)\}/', $formula, $matches);
        
        foreach ($matches[1] as $placeholder) {
            $value = get_post_meta($object['id'], $placeholder, true);
            $formula = str_replace('{' . $placeholder . '}', $value, $formula);
        }
        
        // Safely evaluate the formula
        try {
            return eval('return ' . $formula . ';');
        } catch (Exception $e) {
            return null;
        }
    }
}

Non-Persistent Fields


/**
 * Virtual and computed fields that don't persist
 */
class Virtual_Computed_Fields {
    
    public function __construct() {
        add_action('rest_api_init', [$this, 'register_virtual_fields']);
    }
    
    /**
     * Register virtual fields
     */
    public function register_virtual_fields() {
        // Computed statistics field
        register_rest_field('post', 'statistics', [
            'get_callback' => [$this, 'compute_statistics'],
            'schema' => [
                'description' => 'Computed post statistics',
                'type' => 'object',
                'readonly' => true,
                'context' => ['view'],
                'properties' => [
                    'word_count' => ['type' => 'integer'],
                    'reading_time' => ['type' => 'integer'],
                    'sentence_count' => ['type' => 'integer'],
                    'paragraph_count' => ['type' => 'integer'],
                    'image_count' => ['type' => 'integer'],
                    'link_count' => ['type' => 'integer'],
                    'readability_score' => ['type' => 'number'],
                ],
            ],
        ]);
        
        // Aggregated data field
        register_rest_field('user', 'activity_summary', [
            'get_callback' => [$this, 'aggregate_user_activity'],
            'schema' => [
                'description' => 'User activity summary',
                'type' => 'object',
                'readonly' => true,
                'context' => ['view', 'edit'],
            ],
        ]);
        
        // Calculated relationships field
        register_rest_field('post', 'relationships', [
            'get_callback' => [$this, 'calculate_relationships'],
            'schema' => [
                'description' => 'Post relationships',
                'type' => 'object',
                'readonly' => true,
                'context' => ['view'],
            ],
        ]);
        
        // Dynamic permissions field
        register_rest_field('post', 'permissions', [
            'get_callback' => [$this, 'get_permissions'],
            'schema' => [
                'description' => 'Current user permissions for this post',
                'type' => 'object',
                'readonly' => true,
                'context' => ['view', 'edit'],
            ],
        ]);
    }
    
    /**
     * Compute post statistics
     */
    public function compute_statistics($post) {
        $content = get_post_field('post_content', $post['id']);
        $plain_text = wp_strip_all_tags($content);
        
        return [
            'word_count' => str_word_count($plain_text),
            'reading_time' => ceil(str_word_count($plain_text) / 200),
            'sentence_count' => preg_match_all('/[.!?]+/', $plain_text),
            'paragraph_count' => substr_count($content, '
'),
            'image_count' => substr_count($content, ' substr_count($content, ' $this->calculate_readability($plain_text),
        ];
    }
    
    /**
     * Calculate readability score (Flesch Reading Ease)
     */
    private function calculate_readability($text) {
        $words = str_word_count($text);
        $sentences = preg_match_all('/[.!?]+/', $text, $matches);
        $syllables = $this->count_syllables($text);
        
        if ($words == 0 || $sentences == 0) {
            return 0;
        }
        
        $score = 206.835 - 1.015 * ($words / $sentences) - 84.6 * ($syllables / $words);
        
        return round(max(0, min(100, $score)), 2);
    }
    
    /**
     * Count syllables in text
     */
    private function count_syllables($text) {
        $words = str_word_count($text, 1);
        $syllables = 0;
        
        foreach ($words as $word) {
            $syllables += max(1, preg_match_all('/[aeiouAEIOU]/', $word, $matches));
        }
        
        return $syllables;
    }
    
    /**
     * Aggregate user activity
     */
    public function aggregate_user_activity($user) {
        $user_id = $user['id'];
        
        return [
            'total_posts' => count_user_posts($user_id),
            'recent_posts' => get_posts([
                'author' => $user_id,
                'posts_per_page' => 5,
                'fields' => 'ids',
            ]),
            'total_comments' => get_comments([
                'user_id' => $user_id,
                'count' => true,
            ]),
            'recent_comments' => get_comments([
                'user_id' => $user_id,
                'number' => 5,
                'fields' => 'ids',
            ]),
            'last_login' => get_user_meta($user_id, 'last_login', true),
            'registration_age_days' => $this->get_registration_age($user_id),
        ];
    }
    
    /**
     * Calculate relationships
     */
    public function calculate_relationships($post) {
        $post_id = $post['id'];
        
        return [
            'parent' => wp_get_post_parent_id($post_id),
            'children' => get_children([
                'post_parent' => $post_id,
                'post_type' => 'any',
                'fields' => 'ids',
            ]),
            'siblings' => $this->get_sibling_posts($post_id),
            'related' => $this->get_related_posts_by_taxonomy($post_id),
            'references' => $this->get_post_references($post_id),
            'referenced_by' => $this->get_posts_referencing($post_id),
        ];
    }
    
    /**
     * Get current user permissions for post
     */
    public function get_permissions($post) {
        $post_id = $post['id'];
        $post_obj = get_post($post_id);
        
        return [
            'read' => current_user_can('read_post', $post_id),
            'edit' => current_user_can('edit_post', $post_id),
            'delete' => current_user_can('delete_post', $post_id),
            'publish' => current_user_can('publish_post', $post_id),
            'edit_others' => current_user_can('edit_others_posts'),
            'is_author' => get_current_user_id() == $post_obj->post_author,
        ];
    }
    
    /**
     * Helper methods
     */
    private function get_registration_age($user_id) {
        $user = get_user_by('id', $user_id);
        $registered = strtotime($user->user_registered);
        
        return floor((time() - $registered) / DAY_IN_SECONDS);
    }
    
    private function get_sibling_posts($post_id) {
        $parent_id = wp_get_post_parent_id($post_id);
        
        if (!$parent_id) {
            return [];
        }
        
        return get_children([
            'post_parent' => $parent_id,
            'post_type' => 'any',
            'exclude' => $post_id,
            'fields' => 'ids',
        ]);
    }
    
    private function get_related_posts_by_taxonomy($post_id) {
        $taxonomies = get_post_taxonomies($post_id);
        $related = [];
        
        foreach ($taxonomies as $taxonomy) {
            $terms = wp_get_post_terms($post_id, $taxonomy, ['fields' => 'ids']);
            
            if (!empty($terms)) {
                $posts = get_posts([
                    'tax_query' => [
                        [
                            'taxonomy' => $taxonomy,
                            'field' => 'term_id',
                            'terms' => $terms,
                        ],
                    ],
                    'post__not_in' => [$post_id],
                    'posts_per_page' => 5,
                    'fields' => 'ids',
                ]);
                
                $related = array_merge($related, $posts);
            }
        }
        
        return array_unique($related);
    }
    
    private function get_post_references($post_id) {
        $content = get_post_field('post_content', $post_id);
        preg_match_all('/wp:post-link[^}]+}"id":(\d+)/', $content, $matches);
        
        return array_unique($matches[1]);
    }
    
    private function get_posts_referencing($post_id) {
        global $wpdb;
        
        $results = $wpdb->get_col($wpdb->prepare(
            "SELECT ID FROM {$wpdb->posts} 
            WHERE post_content LIKE %s 
            AND post_status = 'publish'",
            '%"id":' . $post_id . '%'
        ));
        
        return $results;
    }
}
