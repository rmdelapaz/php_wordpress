Unit Testing in WordPress

Write reliable, testable WordPress code with PHPUnit

Learning Objectives

By the end of this lesson, you will be able to:

  • Set up PHPUnit for WordPress testing
  • Write unit tests for plugins and themes
  • Mock WordPress functions and database
  • Implement test-driven development (TDD)
  • Use fixtures and data providers
  • Generate code coverage reports

Setting Up PHPUnit for WordPress

Installation and Configuration


# Install PHPUnit via Composer
composer require --dev phpunit/phpunit ^9.5
composer require --dev wp-phpunit/wp-phpunit
composer require --dev yoast/phpunit-polyfills

# Install WordPress test suite
bash bin/install-wp-tests.sh wordpress_test root '' localhost latest

# Project structure
my-plugin/
├── src/
│   └── MyPlugin.php
├── tests/

│   ├── test-sample.php
│   └── fixtures/
├── bin/

Bootstrap File (tests/bootstrap.php)


<?php
/**
 * PHPUnit bootstrap file for WordPress plugin testing
 */

// Get WordPress test instance path from environment or default
$_tests_dir = getenv('WP_TESTS_DIR');

if (!$_tests_dir) {
    $_tests_dir = rtrim(sys_get_temp_dir(), '/\\') . '/wordpress-tests-lib';
}

// Forward custom PHPUnit parameters
if (!file_exists($_tests_dir . '/includes/functions.php')) {
    echo "Could not find $_tests_dir/includes/functions.php, have you run bin/install-wp-tests.sh?" . PHP_EOL;
    exit(1);
}

// Give access to tests_add_filter() function
require_once $_tests_dir . '/includes/functions.php';

/**
 * Manually load the plugin being tested
 */
function _manually_load_plugin() {
    require dirname(dirname(__FILE__)) . '/my-plugin.php';
}

// Hook plugin loading to muplugins_loaded
tests_add_filter('muplugins_loaded', '_manually_load_plugin');

// Load the WordPress test environment
require $_tests_dir . '/includes/bootstrap.php';

// Load test case base classes
require_once dirname(__FILE__) . '/class-test-case.php';

Writing Unit Tests

Basic Test Class


<?php
/**
 * Class Test_My_Plugin
 *
 * @package My_Plugin
 */

use PHPUnit\Framework\TestCase;
use WP_UnitTestCase;

/**
 * Plugin test case
 */
class Test_My_Plugin extends WP_UnitTestCase {
    
    /**
     * Plugin instance
     */
    private $plugin;
    
    /**
     * Set up test fixtures
     */
    public function setUp(): void {
        parent::setUp();
        
        // Initialize plugin
        $this->plugin = new My_Plugin();
        
        // Create test data
        $this->factory = new WP_UnitTest_Factory();
        
        // Create test user
        $this->user_id = $this->factory->user->create([
            'role' => 'administrator',
            'user_login' => 'testuser',
            'user_email' => 'test@example.com'
        ]);
        
        // Set current user
        wp_set_current_user($this->user_id);
    }
    
    /**
     * Tear down test fixtures
     */
    public function tearDown(): void {
        parent::tearDown();
        
        // Clean up
        wp_delete_user($this->user_id);
        
        // Reset globals
        $GLOBALS['current_user'] = null;
        wp_set_current_user(0);
    }
    
    /**
     * Test plugin initialization
     */
    public function test_plugin_initialization() {
        $this->assertInstanceOf(My_Plugin::class, $this->plugin);
        $this->assertTrue(class_exists('My_Plugin'));
    }
    
    /**
     * Test hooks registration
     */
    public function test_hooks_registered() {
        // Test action hooks
        $this->assertNotFalse(has_action('init', [$this->plugin, 'init']));
        $this->assertNotFalse(has_action('wp_enqueue_scripts', [$this->plugin, 'enqueue_scripts']));
        
        // Test filter hooks
        $this->assertNotFalse(has_filter('the_content', [$this->plugin, 'filter_content']));
        
        // Test hook priority
        $this->assertEquals(10, has_action('init', [$this->plugin, 'init']));
    }
    
    /**
     * Test database operations
     */
    public function test_database_operations() {
        global $wpdb;
        
        // Create test post
        $post_id = $this->factory->post->create([
            'post_title' => 'Test Post',
            'post_content' => 'Test content',
            'post_status' => 'publish'
        ]);
        
        // Test post creation
        $this->assertGreaterThan(0, $post_id);
        
        // Test post retrieval
        $post = get_post($post_id);
        $this->assertEquals('Test Post', $post->post_title);
        
        // Test meta data
        update_post_meta($post_id, 'test_meta', 'test_value');
        $this->assertEquals('test_value', get_post_meta($post_id, 'test_meta', true));
        
        // Test custom table operations
        $table_name = $wpdb->prefix . 'my_custom_table';
        $result = $wpdb->insert(
            $table_name,
            ['column1' => 'value1', 'column2' => 'value2'],
            ['%s', '%s']
        );
        
        $this->assertEquals(1, $result);
        
        // Clean up
        wp_delete_post($post_id, true);
    }
    
    /**
     * Test with data provider
     * 
     * @dataProvider provide_content_filter_data
     */
    public function test_content_filter($input, $expected) {
        $result = $this->plugin->filter_content($input);
        $this->assertEquals($expected, $result);
    }
    
    /**
     * Data provider for content filter test
     */
    public function provide_content_filter_data() {
        return [
            'empty_content' => ['', ''],
            'simple_text' => ['Hello World', 'Hello World'],
            'with_shortcode' => ['[myshortcode]', 'Processed Shortcode'],
            'with_html' => ['
Test
', '
Test
'],
            'special_chars' => ['Test & < > "', 'Test & < > "']
        ];
    }
    
    /**
     * Test AJAX handler
     */
    public function test_ajax_handler() {
        // Set up AJAX environment
        define('DOING_AJAX', true);
        $_POST['nonce'] = wp_create_nonce('my_ajax_nonce');
        $_POST['action'] = 'my_ajax_action';
        $_POST['data'] = 'test_data';
        
        // Capture output
        ob_start();
        $this->plugin->ajax_handler();
        $output = ob_get_clean();
        
        // Parse JSON response
        $response = json_decode($output, true);
        
        // Assertions
        $this->assertTrue($response['success']);
        $this->assertEquals('test_data', $response['data']['received']);
    }
    
    /**
     * Test exception handling
     */
    public function test_exception_handling() {
        $this->expectException(InvalidArgumentException::class);
        $this->expectExceptionMessage('Invalid input');
        
        $this->plugin->process_data(-1);
    }
    
    /**
     * Test with mock objects
     */
    public function test_with_mock() {
        // Create mock
        $mock = $this->getMockBuilder(External_API::class)
            ->setMethods(['fetch_data'])
            ->getMock();
        
        // Set expectations
        $mock->expects($this->once())
            ->method('fetch_data')
            ->with('test_endpoint')
            ->willReturn(['status' => 'success', 'data' => 'mocked_data']);
        
        // Inject mock
        $this->plugin->set_api($mock);
        
        // Test
        $result = $this->plugin->get_external_data('test_endpoint');
        $this->assertEquals('mocked_data', $result);
    }
}

Testing WordPress Components

Testing Custom Post Types


<?php
/**
 * Test Custom Post Type
 */
class Test_Custom_Post_Type extends WP_UnitTestCase {
    
    public function setUp(): void {
        parent::setUp();
        
        // Register custom post type
        register_post_type('book', [
            'public' => true,
            'label' => 'Books',
            'supports' => ['title', 'editor', 'thumbnail']
        ]);
    }
    
    /**
     * Test post type registration
     */
    public function test_post_type_exists() {
        $this->assertTrue(post_type_exists('book'));
    }
    
    /**
     * Test post type capabilities
     */
    public function test_post_type_capabilities() {
        $post_type = get_post_type_object('book');
        
        $this->assertNotNull($post_type);
        $this->assertTrue($post_type->public);
        $this->assertEquals('Books', $post_type->label);
        $this->assertTrue(post_type_supports('book', 'title'));
        $this->assertTrue(post_type_supports('book', 'editor'));
        $this->assertTrue(post_type_supports('book', 'thumbnail'));
    }
    
    /**
     * Test creating custom post
     */
    public function test_create_custom_post() {
        $post_id = wp_insert_post([
            'post_title' => 'Test Book',
            'post_content' => 'Book content',
            'post_type' => 'book',
            'post_status' => 'publish'
        ]);
        
        $this->assertIsInt($post_id);
        $this->assertGreaterThan(0, $post_id);
        
        $post = get_post($post_id);
        $this->assertEquals('book', $post->post_type);
        $this->assertEquals('Test Book', $post->post_title);
    }
    
    /**
     * Test custom taxonomies
     */
    public function test_custom_taxonomy() {
        // Register taxonomy
        register_taxonomy('genre', 'book', [
            'label' => 'Genres',
            'hierarchical' => true
        ]);
        
        $this->assertTrue(taxonomy_exists('genre'));
        
        // Create terms
        $term = wp_insert_term('Fiction', 'genre');
        $this->assertIsArray($term);
        $this->assertArrayHasKey('term_id', $term);
        
        // Assign term to post
        $post_id = $this->factory->post->create(['post_type' => 'book']);
        wp_set_object_terms($post_id, $term['term_id'], 'genre');
        
        $terms = wp_get_object_terms($post_id, 'genre');
        $this->assertCount(1, $terms);
        $this->assertEquals('Fiction', $terms[0]->name);
    }
}

/**
 * Test WordPress Hooks
 */
class Test_WordPress_Hooks extends WP_UnitTestCase {
    
    /**
     * Test action hooks
     */
    public function test_action_execution() {
        $test_value = '';
        
        // Add action
        add_action('test_action', function($value) use (&$test_value) {
            $test_value = $value;
        });
        
        // Execute action
        do_action('test_action', 'test_data');
        
        $this->assertEquals('test_data', $test_value);
    }
    
    /**
     * Test filter hooks
     */
    public function test_filter_application() {
        // Add filter
        add_filter('test_filter', function($value) {
            return strtoupper($value);
        });
        
        // Apply filter
        $result = apply_filters('test_filter', 'hello');
        
        $this->assertEquals('HELLO', $result);
    }
    
    /**
     * Test hook priorities
     */
    public function test_hook_priorities() {
        $result = [];
        
        add_action('priority_test', function() use (&$result) {
            $result[] = 'second';
        }, 20);
        
        add_action('priority_test', function() use (&$result) {
            $result[] = 'first';
        }, 10);
        
        add_action('priority_test', function() use (&$result) {
            $result[] = 'third';
        }, 30);
        
        do_action('priority_test');
        
        $this->assertEquals(['first', 'second', 'third'], $result);
    }
    
    /**
     * Test removing hooks
     */
    public function test_remove_hook() {
        $callback = function($value) {
            return 'modified';
        };
        
        add_filter('removable_filter', $callback, 10);
        $this->assertEquals('modified', apply_filters('removable_filter', 'original'));
        
        remove_filter('removable_filter', $callback, 10);
        $this->assertEquals('original', apply_filters('removable_filter', 'original'));
    }
}

/**
 * Test REST API Endpoints
 */
class Test_REST_API extends WP_UnitTestCase {
    
    /**
     * REST Server
     */
    protected $server;
    
    public function setUp(): void {
        parent::setUp();
        
        // Initialize REST server
        global $wp_rest_server;
        $this->server = $wp_rest_server = new WP_REST_Server();
        
        // Register routes
        do_action('rest_api_init');
        
        // Create test user
        $this->user_id = $this->factory->user->create([
            'role' => 'administrator'
        ]);
    }
    
    /**
     * Test REST endpoint registration
     */
    public function test_endpoint_registration() {
        $routes = $this->server->get_routes();
        
        $this->assertArrayHasKey('/myplugin/v1/items', $routes);
        $this->assertArrayHasKey('/myplugin/v1/items/(?P[\d]+)', $routes);
    }
    
    /**
     * Test GET request
     */
    public function test_get_items() {
        wp_set_current_user($this->user_id);
        
        $request = new WP_REST_Request('GET', '/myplugin/v1/items');
        $response = $this->server->dispatch($request);
        
        $this->assertEquals(200, $response->get_status());
        
        $data = $response->get_data();
        $this->assertIsArray($data);
    }
    
    /**
     * Test POST request
     */
    public function test_create_item() {
        wp_set_current_user($this->user_id);
        
        $request = new WP_REST_Request('POST', '/myplugin/v1/items');
        $request->set_body_params([
            'title' => 'Test Item',
            'content' => 'Test content'
        ]);
        
        $response = $this->server->dispatch($request);
        
        $this->assertEquals(201, $response->get_status());
        
        $data = $response->get_data();
        $this->assertEquals('Test Item', $data['title']);
    }
    
    /**
     * Test authentication
     */
    public function test_unauthorized_access() {
        // No user set
        wp_set_current_user(0);
        
        $request = new WP_REST_Request('POST', '/myplugin/v1/items');
        $response = $this->server->dispatch($request);
        
        $this->assertEquals(401, $response->get_status());
    }
    
    /**
     * Test validation
     */
    public function test_validation_error() {
        wp_set_current_user($this->user_id);
        
        $request = new WP_REST_Request('POST', '/myplugin/v1/items');
        $request->set_body_params([
            'title' => '' // Empty title should fail validation
        ]);
        
        $response = $this->server->dispatch($request);
        
        $this->assertEquals(400, $response->get_status());
    }
}

Test Doubles and Mocking

Creating and Using Mocks


<?php
/**
 * Advanced Mocking Techniques
 */
class Test_With_Mocks extends WP_UnitTestCase {
    
    /**
     * Mock WordPress functions
     */
    public function test_mock_wp_functions() {
        // Mock wp_remote_get
        add_filter('pre_http_request', function($preempt, $args, $url) {
            if ($url === 'https://api.example.com/data') {
                return [
                    'response' => ['code' => 200],
                    'body' => json_encode(['status' => 'success', 'data' => 'test'])
                ];
            }
            return $preempt;
        }, 10, 3);
        
        // Test API call
        $response = wp_remote_get('https://api.example.com/data');
        $body = json_decode(wp_remote_retrieve_body($response), true);
        
        $this->assertEquals('success', $body['status']);
        $this->assertEquals('test', $body['data']);
    }
    
    /**
     * Mock database operations
     */
    public function test_mock_database() {
        global $wpdb;
        
        // Create mock for $wpdb
        $wpdb_mock = $this->getMockBuilder(stdClass::class)
            ->setMethods(['get_results', 'insert', 'update', 'delete'])
            ->getMock();
        
        // Set up expectations
        $wpdb_mock->expects($this->once())
            ->method('get_results')
            ->with($this->stringContains('SELECT * FROM'))
            ->willReturn([
                (object)['id' => 1, 'name' => 'Test 1'],
                (object)['id' => 2, 'name' => 'Test 2']
            ]);
        
        // Use mock in test
        $original_wpdb = $wpdb;
        $wpdb = $wpdb_mock;
        
        // Test code that uses database
        $results = $wpdb->get_results("SELECT * FROM test_table");
        $this->assertCount(2, $results);
        $this->assertEquals('Test 1', $results[0]->name);
        
        // Restore original
        $wpdb = $original_wpdb;
    }
    
    /**
     * Mock external dependencies
     */
    public function test_mock_external_service() {
        // Create service mock
        $service_mock = $this->createMock(External_Service::class);
        
        // Configure mock
        $service_mock->method('authenticate')
            ->willReturn(true);
        
        $service_mock->method('fetch_user_data')
            ->with('user123')
            ->willReturn([
                'id' => 'user123',
                'name' => 'Test User',
                'email' => 'test@example.com'
            ]);
        
        // Test with mock
        $plugin = new My_Plugin($service_mock);
        $user_data = $plugin->get_user_info('user123');
        
        $this->assertEquals('Test User', $user_data['name']);
    }
    
    /**
     * Stub complex objects
     */
    public function test_with_stubs() {
        // Create stub
        $stub = $this->createStub(Payment_Gateway::class);
        
        // Configure stub
        $stub->method('process_payment')
            ->willReturnMap([
                [100, 'USD', ['status' => 'success', 'transaction_id' => 'TXN123']],
                [0, 'USD', ['status' => 'error', 'message' => 'Invalid amount']],
            ]);
        
        // Test different scenarios
        $result1 = $stub->process_payment(100, 'USD');
        $this->assertEquals('success', $result1['status']);
        
        $result2 = $stub->process_payment(0, 'USD');
        $this->assertEquals('error', $result2['status']);
    }
    
    /**
     * Spy on method calls
     */
    public function test_with_spy() {
        // Create spy
        $logger_spy = $this->getMockBuilder(Logger::class)
            ->setMethods(['log'])
            ->getMock();
        
        // Expect method to be called
        $logger_spy->expects($this->exactly(3))
            ->method('log')
            ->withConsecutive(
                [$this->equalTo('info'), $this->stringContains('Started')],
                [$this->equalTo('debug'), $this->anything()],
                [$this->equalTo('info'), $this->stringContains('Completed')]
            );
        
        // Use spy in test
        $processor = new Data_Processor($logger_spy);
        $processor->process(['test' => 'data']);
    }
}

/**
 * Test Fixtures and Factories

class Test_With_Fixtures extends WP_UnitTestCase {
    
    /**
     * Use factory to create test data
     */
    public function test_with_factory_data() {
        // Create multiple users
        $users = $this->factory->user->create_many(5, [
            'role' => 'subscriber'
        ]);
        
        $this->assertCount(5, $users);
        
        // Create posts with specific attributes
        $post_ids = $this->factory->post->create_many(10, [
            'post_author' => $users[0],
            'post_status' => 'publish',
            'post_type' => 'post'
        ]);
        
        // Create hierarchical pages
        $parent_id = $this->factory->post->create([
            'post_type' => 'page',
            'post_title' => 'Parent Page'
        ]);
        
        $child_ids = $this->factory->post->create_many(3, [
            'post_type' => 'page',
            'post_parent' => $parent_id
        ]);
        
        // Test hierarchy
        $children = get_children($parent_id);
        $this->assertCount(3, $children);
        
        // Create terms and assign to posts
        $term_id = $this->factory->term->create([
            'taxonomy' => 'category',
            'name' => 'Test Category'
        ]);
        
        foreach ($post_ids as $post_id) {
            wp_set_post_terms($post_id, [$term_id], 'category');
        }
        
        // Test term assignment
        $posts_in_category = get_posts([
            'category' => $term_id,
            'posts_per_page' => -1
        ]);
        
        $this->assertCount(10, $posts_in_category);
    }
    
    /**
     * Load fixtures from files
     */
    public function test_with_file_fixtures() {
        // Load JSON fixture
        $fixture_data = json_decode(
            file_get_contents(__DIR__ . '/fixtures/test_data.json'),
            true
        );
        
        foreach ($fixture_data['posts'] as $post_data) {
            $post_id = wp_insert_post($post_data);
            $this->assertIsInt($post_id);
        }
        
        // Load SQL fixture
        global $wpdb;
        $sql = file_get_contents(__DIR__ . '/fixtures/test_data.sql');
        $wpdb->query($sql);
        
        // Verify data loaded
        $results = $wpdb->get_results("SELECT * FROM {$wpdb->prefix}test_table");
        $this->assertNotEmpty($results);
    }
}

Code Coverage Analysis

Generating Coverage Reports

vendor/bin/phpunit --coverage-clover coverage/clover.xml

vendor/bin/phpunit --coverage-xml coverage/xml

vendor/bin/phpunit --coverage-text --coverage-filter src/ --min=80

Coverage Annotations


<?php
/**
 * Class with coverage annotations
 * 
 * @covers My_Plugin
 * @uses Helper_Class
 */
class Test_My_Plugin_Coverage extends WP_UnitTestCase {
    
    /**
     * @covers My_Plugin::process_data
     * @covers My_Plugin::validate_input
     */
    public function test_process_data() {
        // Test implementation
    }
    
    /**
     * @covers ::global_function
     * @uses Database_Helper
     */
    public function test_global_function() {
        // Test implementation
    }
    
    /**
     * @coversNothing
     */
    public function test_integration() {
        // Integration test that doesn't count toward coverage
    }
    
    /**
     * @codeCoverageIgnore
     */
    private function helper_method() {
        // This method is ignored in coverage
    }
}

/**
 * Class with inline coverage directives
 */
class Example_Class {
    
    public function covered_method() {
        // This is covered
        return true;
    }
    
    // @codeCoverageIgnoreStart
    public function ignored_method() {
        // This entire method is ignored
        return false;
    }
    // @codeCoverageIgnoreEnd
    
    public function partial_coverage() {
        $result = $this->covered_method();
        
        // @codeCoverageIgnoreStart
        if (defined('SPECIAL_CONSTANT')) {
            // This block is ignored
            $result = false;
        }
        // @codeCoverageIgnoreEnd
        
        return $result;
    }
}

Summary

Key Takeaways

  • PHPUnit provides comprehensive testing for WordPress
  • WP_UnitTestCase offers WordPress-specific assertions
  • Factories simplify test data creation
  • Mocking isolates code under test
  • Code coverage identifies untested code
  • TDD improves code quality and design

Testing Best Practices

  • Write tests before implementation (TDD)
  • Keep tests focused and isolated
  • Use descriptive test names
  • Mock external dependencies
  • Maintain high code coverage (>80%)
  • Run tests in CI/CD pipeline
  • Test edge cases and errors
