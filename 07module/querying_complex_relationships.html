Querying Complex Relationships

Master advanced WordPress queries for complex data relationships

Learning Objectives

By the end of this lesson, you will be able to:

  • Build complex WP_Query instances
  • Combine multiple meta queries and tax queries
  • Create custom SQL queries when needed
  • Query posts with multiple relationships
  • Optimize complex queries for performance
  • Implement advanced search functionality

class Advanced_Query_Builder {
    
    /**
     * Query posts with complex meta relationships
     */
    public static function complex_meta_query() {
        $args = [
            'post_type' => 'product',
            'posts_per_page' => 20,
            'meta_query' => [
                'relation' => 'AND',
                // Price range
                [
                    'relation' => 'AND',
                    [
                        'key' => '_price',
                        'value' => 100,
                        'compare' => '>=',
                        'type' => 'NUMERIC'
                    ],
                    [
                        'key' => '_price',
                        'value' => 500,
                        'compare' => '<=',
                        'type' => 'NUMERIC'
                    ]
                ],
                // Stock status
                [
                    'relation' => 'OR',
                    [
                        'key' => '_stock_status',
                        'value' => 'instock',
                        'compare' => '='
                    ],
                    [
                        'relation' => 'AND',
                        [
                            'key' => '_backorders',
                            'value' => 'yes',
                            'compare' => '='
                        ],
                        [
                            'key' => '_stock',
                            'value' => 0,
                            'compare' => '>',
                            'type' => 'NUMERIC'
                        ]
                    ]
                ],
                // Featured products with rating
                [
                    'relation' => 'OR',
                    [
                        'key' => '_featured',
                        'value' => 'yes',
                        'compare' => '='
                    ],
                    [
                        'key' => '_average_rating',
                        'value' => 4,
                        'compare' => '>=',
                        'type' => 'NUMERIC'
                    ]
                ]
            ],
            // Order by multiple meta values
            'orderby' => [
                'featured_clause' => 'DESC',
                'price_clause' => 'ASC',
                'rating_clause' => 'DESC'
            ],
            'meta_query' => [
                'featured_clause' => [
                    'key' => '_featured',
                    'compare' => 'EXISTS'
                ],
                'price_clause' => [
                    'key' => '_price',
                    'type' => 'NUMERIC'
                ],
                'rating_clause' => [
                    'key' => '_average_rating',
                    'type' => 'NUMERIC'
                ]
            ]
        ];
        
        return new WP_Query($args);
    }
    
    /**
     * Date-based meta queries
     */
    public static function date_meta_query() {
        $args = [
            'post_type' => 'event',
            'posts_per_page' => -1,
            'meta_query' => [
                'relation' => 'AND',
                // Events starting in the future
                [
                    'key' => '_event_start_date',
                    'value' => current_time('mysql'),
                    'compare' => '>',
                    'type' => 'DATETIME'
                ],
                // But within next 30 days
                [
                    'key' => '_event_start_date',
                    'value' => date('Y-m-d H:i:s', strtotime('+30 days')),
                    'compare' => '<=',
                    'type' => 'DATETIME'
                ],
                // Not cancelled
                [
                    'key' => '_event_status',
                    'value' => 'cancelled',
                    'compare' => '!='
                ]
            ],
            'orderby' => 'meta_value',
            'meta_key' => '_event_start_date',
            'order' => 'ASC'
        ];
        
        return new WP_Query($args);
    }
    
    /**
     * Geolocation-based queries
     */
    public static function geo_location_query($lat, $lng, $radius = 50) {
        global $wpdb;
        
        // First, get posts within radius using custom SQL
        $sql = $wpdb->prepare("
            SELECT DISTINCT p.ID, 
                   ( 6371 * acos( cos( radians(%f) ) * cos( radians( lat.meta_value ) ) 
                   * cos( radians( lng.meta_value ) - radians(%f) ) 
                   + sin( radians(%f) ) * sin( radians( lat.meta_value ) ) ) ) AS distance
            FROM {$wpdb->posts} p
            INNER JOIN {$wpdb->postmeta} lat ON p.ID = lat.post_id AND lat.meta_key = '_latitude'
            INNER JOIN {$wpdb->postmeta} lng ON p.ID = lng.post_id AND lng.meta_key = '_longitude'
            WHERE p.post_type = 'location'
            AND p.post_status = 'publish'
            HAVING distance < %d
            ORDER BY distance ASC
        ", $lat, $lng, $lat, $radius);
        
        $post_ids = $wpdb->get_col($sql);
        
        if (empty($post_ids)) {
            return new WP_Query(['post__in' => [0]]);
        }
        
        // Then query with additional filters
        $args = [
            'post_type' => 'location',
            'post__in' => $post_ids,
            'orderby' => 'post__in',
            'posts_per_page' => 20
        ];
        
        return new WP_Query($args);
    }

     * Query posts that have ALL specified terms

    /**
     * Exclude posts with certain term combinations
     */
    public static function exclude_term_combinations() {
        global $wpdb;
        
        // First get posts to exclude (discontinued products in certain categories)
        $excluded_sql = "
            SELECT DISTINCT tr1.object_id
            FROM {$wpdb->term_relationships} tr1
            INNER JOIN {$wpdb->term_taxonomy} tt1 ON tr1.term_taxonomy_id = tt1.term_taxonomy_id
            INNER JOIN {$wpdb->terms} t1 ON tt1.term_id = t1.term_id
            INNER JOIN {$wpdb->term_relationships} tr2 ON tr1.object_id = tr2.object_id
            INNER JOIN {$wpdb->term_taxonomy} tt2 ON tr2.term_taxonomy_id = tt2.term_taxonomy_id
            INNER JOIN {$wpdb->terms} t2 ON tt2.term_id = t2.term_id
            WHERE tt1.taxonomy = 'product_status' AND t1.slug = 'discontinued'
            AND tt2.taxonomy = 'product_category' AND t2.slug IN ('electronics', 'computers')
        ";
        
        $excluded_ids = $wpdb->get_col($excluded_sql);
        
        $args = [
            'post_type' => 'product',
            'posts_per_page' => 30,
            'post__not_in' => $excluded_ids,
            'tax_query' => [
                [
                    'taxonomy' => 'product_category',
                    'field' => 'slug',
                    'terms' => ['electronics', 'computers', 'accessories'],
                    'operator' => 'IN'
                ]
            ]
        ];
        
        return new WP_Query($args);
    }

// Combining meta and taxonomy queries
class Combined_Complex_Queries {
    
    /**
     * Products filtered by category, price, and features
     */
    public static function filtered_products_query($filters) {
        $args = [
            'post_type' => 'product',
            'posts_per_page' => 20,
            'paged' => get_query_var('paged') ?: 1
        ];
        
        // Build tax query
        $tax_query = ['relation' => 'AND'];
        
        if (!empty($filters['category'])) {
            $tax_query[] = [
                'taxonomy' => 'product_category',
                'field' => 'slug',
                'terms' => $filters['category'],
                'include_children' => true
            ];
        }
        
        if (!empty($filters['brand'])) {
            $tax_query[] = [
                'taxonomy' => 'brand',
                'field' => 'slug',
                'terms' => $filters['brand']
            ];
        }
        
        if (!empty($filters['features'])) {
            $tax_query[] = [
                'taxonomy' => 'features',
                'field' => 'slug',
                'terms' => $filters['features'],
                'operator' => 'AND' // Must have all features
            ];
        }
        
        if (count($tax_query) > 1) {
            $args['tax_query'] = $tax_query;
        }
        
        // Build meta query
        $meta_query = ['relation' => 'AND'];
        
        if (!empty($filters['min_price']) || !empty($filters['max_price'])) {
            $price_query = ['relation' => 'AND'];
            
            if (!empty($filters['min_price'])) {
                $price_query[] = [
                    'key' => '_price',
                    'value' => $filters['min_price'],
                    'compare' => '>=',
                    'type' => 'NUMERIC'
                ];
            }
            
            if (!empty($filters['max_price'])) {
                $price_query[] = [
                    'key' => '_price',
                    'value' => $filters['max_price'],
                    'compare' => '<=',
                    'type' => 'NUMERIC'
                ];
            }
            
            $meta_query[] = $price_query;
        }
        
        if (!empty($filters['in_stock'])) {
            $meta_query[] = [
                'key' => '_stock',
                'value' => 0,
                'compare' => '>',
                'type' => 'NUMERIC'
            ];
        }
        
        if (!empty($filters['on_sale'])) {
            $meta_query[] = [
                'key' => '_sale_price',
                'value' => '',
                'compare' => '!='
            ];
        }
        
        if (!empty($filters['min_rating'])) {
            $meta_query[] = [
                'key' => '_average_rating',
                'value' => $filters['min_rating'],
                'compare' => '>=',
                'type' => 'NUMERIC'
            ];
        }
        
        if (count($meta_query) > 1) {
            $args['meta_query'] = $meta_query;
        }
        
        // Sorting
        if (!empty($filters['sort'])) {
            switch ($filters['sort']) {
                case 'price_asc':
                    $args['orderby'] = 'meta_value_num';
                    $args['meta_key'] = '_price';
                    $args['order'] = 'ASC';
                    break;
                    
                case 'price_desc':
                    $args['orderby'] = 'meta_value_num';
                    $args['meta_key'] = '_price';
                    $args['order'] = 'DESC';
                    break;
                    
                case 'popularity':
                    $args['orderby'] = 'meta_value_num';
                    $args['meta_key'] = '_sales_count';
                    $args['order'] = 'DESC';
                    break;
                    
                case 'rating':
                    $args['orderby'] = 'meta_value_num';
                    $args['meta_key'] = '_average_rating';
                    $args['order'] = 'DESC';
                    break;
                    
                case 'newest':
                    $args['orderby'] = 'date';
                    $args['order'] = 'DESC';
                    break;
            }
        }
        
        return new WP_Query($args);
    }
    
    /**
     * Related products query
     */
    public static function get_related_products($product_id, $limit = 6) {
        // Get current product's terms
        $categories = wp_get_post_terms($product_id, 'product_category', ['fields' => 'ids']);
        $tags = wp_get_post_terms($product_id, 'product_tag', ['fields' => 'ids']);
        $brand = wp_get_post_terms($product_id, 'brand', ['fields' => 'ids']);
        
        // Get current product's price for similar price range
        $price = get_post_meta($product_id, '_price', true);
        $price_range = 0.3; // 30% range
        
        $args = [
            'post_type' => 'product',
            'posts_per_page' => $limit,
            'post__not_in' => [$product_id],
            'tax_query' => [
                'relation' => 'OR',
                // Same category
                [
                    'taxonomy' => 'product_category',
                    'field' => 'term_id',
                    'terms' => $categories,
                    'operator' => 'IN'
                ],
                // Same brand
                [
                    'taxonomy' => 'brand',
                    'field' => 'term_id',
                    'terms' => $brand,
                    'operator' => 'IN'
                ],
                // Same tags
                [
                    'taxonomy' => 'product_tag',
                    'field' => 'term_id',
                    'terms' => $tags,
                    'operator' => 'IN'
                ]
            ],
            'meta_query' => [
                'relation' => 'AND',
                // Similar price range
                [
                    'key' => '_price',
                    'value' => [$price * (1 - $price_range), $price * (1 + $price_range)],
                    'compare' => 'BETWEEN',
                    'type' => 'NUMERIC'
                ],
                // In stock
                [
                    'key' => '_stock_status',
                    'value' => 'instock',
                    'compare' => '='
                ]
            ],
            // Order by relevance score
            'orderby' => 'meta_value_num',
            'meta_key' => '_popularity_score',
            'order' => 'DESC'
        ];
        
        return new WP_Query($args);
    }
}

// Custom SQL for complex relationships
class Custom_SQL_Queries {
    
    /**
     * Get posts with complex relationships using custom SQL
     */
    public static function complex_relationship_query() {
        global $wpdb;
        
        // Query posts that have specific meta AND taxonomy combinations
        // with aggregated data
        $sql = "
            SELECT 
                p.ID,
                p.post_title,
                p.post_date,
                GROUP_CONCAT(DISTINCT t.name ORDER BY t.name SEPARATOR ', ') as categories,
                AVG(CAST(pm_rating.meta_value AS DECIMAL(3,2))) as avg_rating,
                COUNT(DISTINCT c.comment_ID) as comment_count,
                SUM(CAST(pm_views.meta_value AS UNSIGNED)) as total_views
            FROM {$wpdb->posts} p
            
            -- Join taxonomies
            LEFT JOIN {$wpdb->term_relationships} tr ON p.ID = tr.object_id
            LEFT JOIN {$wpdb->term_taxonomy} tt ON tr.term_taxonomy_id = tt.term_taxonomy_id
            LEFT JOIN {$wpdb->terms} t ON tt.term_id = t.term_id
            
            -- Join meta
            LEFT JOIN {$wpdb->postmeta} pm_rating ON p.ID = pm_rating.post_id 
                AND pm_rating.meta_key = '_rating'
            LEFT JOIN {$wpdb->postmeta} pm_views ON p.ID = pm_views.post_id 
                AND pm_views.meta_key = '_view_count'
            
            -- Join comments
            LEFT JOIN {$wpdb->comments} c ON p.ID = c.comment_post_ID 
                AND c.comment_approved = '1'
            
            WHERE p.post_type = 'product'
            AND p.post_status = 'publish'
            AND tt.taxonomy = 'product_category'
            
            GROUP BY p.ID
            HAVING avg_rating >= 4
            AND comment_count >= 5
            
            ORDER BY total_views DESC, avg_rating DESC
            LIMIT 20
        ";
        
        return $wpdb->get_results($sql);
    }
    
    /**
     * Posts with cross-referenced relationships
     */
    public static function cross_reference_query() {
        global $wpdb;
        
        // Find posts that share multiple relationships
        $sql = "
            SELECT 
                p1.ID as post1_id,
                p1.post_title as post1_title,
                p2.ID as post2_id,
                p2.post_title as post2_title,
                COUNT(DISTINCT tt.taxonomy) as shared_taxonomies,
                COUNT(DISTINCT t.term_id) as shared_terms
            FROM {$wpdb->posts} p1
            
            -- Self join to find related posts
            INNER JOIN {$wpdb->term_relationships} tr1 ON p1.ID = tr1.object_id
            INNER JOIN {$wpdb->term_relationships} tr2 ON tr1.term_taxonomy_id = tr2.term_taxonomy_id
            INNER JOIN {$wpdb->posts} p2 ON tr2.object_id = p2.ID
            
            -- Get taxonomy info
            INNER JOIN {$wpdb->term_taxonomy} tt ON tr1.term_taxonomy_id = tt.term_taxonomy_id
            INNER JOIN {$wpdb->terms} t ON tt.term_id = t.term_id
            
            WHERE p1.post_type = 'product'
            AND p2.post_type = 'product'
            AND p1.ID != p2.ID
            AND p1.post_status = 'publish'
            AND p2.post_status = 'publish'
            AND p1.ID = %d
            
            GROUP BY p1.ID, p2.ID
            HAVING shared_terms >= 3
            
            ORDER BY shared_terms DESC, shared_taxonomies DESC
            LIMIT 10
        ";
        
        $post_id = get_the_ID();
        return $wpdb->get_results($wpdb->prepare($sql, $post_id));
    }
    
    /**
     * Hierarchical data with recursive query
     */
    public static function hierarchical_category_posts() {
        global $wpdb;
        
        $parent_term_id = 5; // Example parent category
        
        // MySQL 8.0+ recursive CTE
        $sql = "
            WITH RECURSIVE category_tree AS (
                -- Anchor: start with parent category
                SELECT term_id, name, slug, parent, 0 as level
                FROM {$wpdb->terms} t
                INNER JOIN {$wpdb->term_taxonomy} tt ON t.term_id = tt.term_id
                WHERE term_id = %d
                AND tt.taxonomy = 'product_category'
                
                UNION ALL
                
                -- Recursive: get all children
                SELECT t.term_id, t.name, t.slug, tt.parent, ct.level + 1
                FROM {$wpdb->terms} t
                INNER JOIN {$wpdb->term_taxonomy} tt ON t.term_id = tt.term_id
                INNER JOIN category_tree ct ON tt.parent = ct.term_id
                WHERE tt.taxonomy = 'product_category'
            )
            SELECT DISTINCT
                p.ID,
                p.post_title,
                ct.name as category_name,
                ct.level as category_level
            FROM {$wpdb->posts} p
            INNER JOIN {$wpdb->term_relationships} tr ON p.ID = tr.object_id
            INNER JOIN {$wpdb->term_taxonomy} tt ON tr.term_taxonomy_id = tt.term_taxonomy_id
            INNER JOIN category_tree ct ON tt.term_id = ct.term_id
            WHERE p.post_type = 'product'
            AND p.post_status = 'publish'
            ORDER BY ct.level, p.post_title
        ";
        
        // Fallback for older MySQL versions
        if (version_compare($wpdb->db_version(), '8.0', '<')) {
            $sql = self::hierarchical_fallback_query($parent_term_id);
        }
        
        return $wpdb->get_results($wpdb->prepare($sql, $parent_term_id));
    }
    
    /**
     * Fallback for older MySQL versions
     */
    private static function hierarchical_fallback_query($parent_id) {
        global $wpdb;
        
        // Get all child categories recursively
        $categories = self::get_term_children_recursive($parent_id, 'product_category');
        $categories[] = $parent_id; // Include parent
        
        $placeholders = implode(',', array_fill(0, count($categories), '%d'));
        
        $sql = "
            SELECT DISTINCT
                p.ID,
                p.post_title,
                t.name as category_name
            FROM {$wpdb->posts} p
            INNER JOIN {$wpdb->term_relationships} tr ON p.ID = tr.object_id
            INNER JOIN {$wpdb->term_taxonomy} tt ON tr.term_taxonomy_id = tt.term_taxonomy_id
            INNER JOIN {$wpdb->terms} t ON tt.term_id = t.term_id
            WHERE p.post_type = 'product'
            AND p.post_status = 'publish'
            AND tt.taxonomy = 'product_category'
            AND t.term_id IN ($placeholders)
            ORDER BY p.post_title
        ";
        
        return $wpdb->prepare($sql, $categories);
    }
    
    /**
     * Recursive function to get all term children
     */
    private static function get_term_children_recursive($parent_id, $taxonomy) {
        $children = get_term_children($parent_id, $taxonomy);
        $all_children = $children;
        
        foreach ($children as $child_id) {
            $grandchildren = self::get_term_children_recursive($child_id, $taxonomy);
            $all_children = array_merge($all_children, $grandchildren);
        }
        
        return array_unique($all_children);
    }

class Query_Optimization {
    
    /**
     * Optimized query with selective field loading
     */
    public static function optimized_field_query() {
        global $wpdb;
        
        // Only select needed fields
        add_filter('posts_fields', function($fields, $query) {

            if ($query->get('optimize_fields')) {
                $fields = "{$wpdb->posts}.ID, {$wpdb->posts}.post_title, {$wpdb->posts}.post_name";
            }
            
            return $fields;
        }, 10, 2);
        
        // Remove unnecessary JOINs
        add_filter('posts_join', function($join, $query) {
            if ($query->get('optimize_fields')) {
                // Remove JOINs if not needed
            }
            return $join;
        }, 10, 2);
        
        $args = [
            'post_type' => 'product',
            'posts_per_page' => 100,
            'optimize_fields' => true,
            'no_found_rows' => true, // Skip pagination count
            'update_post_meta_cache' => false,
            'update_post_term_cache' => false,
            'fields' => 'ids' // Get only IDs first
        ];
        
        $query = new WP_Query($args);
        
        // Then get specific data as needed
        if ($query->posts) {
            foreach ($query->posts as $post_id) {
                // Selective data loading
                $title = get_the_title($post_id);
                $price = get_post_meta($post_id, '_price', true);
                // Process...
            }
        }
        
        return $query;
    }
    
    /**
     * Cached complex query
     */
    public static function cached_complex_query($args) {
        $cache_key = 'complex_query_' . md5(serialize($args));
        $cached = wp_cache_get($cache_key, 'queries');
        
        if (false !== $cached) {
            return $cached;
        }
        
        // Perform expensive query
        $query = new WP_Query($args);
        
        // Cache for 1 hour
        wp_cache_set($cache_key, $query, 'queries', HOUR_IN_SECONDS);
        
        return $query;
    }
    
    /**
     * Batched query processing
     */
    public static function batch_process_posts() {
        $batch_size = 50;
        $offset = 0;
        $results = [];
        
        do {
            $args = [
                'post_type' => 'product',
                'posts_per_page' => $batch_size,
                'offset' => $offset,
                'orderby' => 'ID',
                'order' => 'ASC',
                'fields' => 'ids',
                'no_found_rows' => true,
                'update_post_meta_cache' => false,
                'update_post_term_cache' => false
            ];
            
            $query = new WP_Query($args);
            
            if ($query->posts) {
                // Process batch
                foreach ($query->posts as $post_id) {
                    $results[] = self::process_single_post($post_id);
                }
                
                $offset += $batch_size;
                
                // Free memory
                wp_reset_postdata();
                unset($query);
            }
            
        } while (count($query->posts) === $batch_size);
        
        return $results;
    }
    
    /**
     * Pre-warm caches for complex queries
     */
    public static function prewarm_query_caches() {
        // Prime post caches
        $post_ids = get_posts([
            'post_type' => 'product',
            'posts_per_page' => 100,
            'fields' => 'ids',
            'orderby' => 'modified',
            'order' => 'DESC'
        ]);
        
        if ($post_ids) {
            // Prime post cache
            _prime_post_caches($post_ids);
            
            // Prime meta cache
            update_meta_cache('post', $post_ids);
            
            // Prime term cache
            wp_get_object_terms($post_ids, get_object_taxonomies('product'));
        }
    }
    
    /**
     * Database index hints
     */
    public static function use_index_hints() {
        global $wpdb;
        
        // Force specific index usage
        $sql = "
            SELECT p.ID, p.post_title
            FROM {$wpdb->posts} p USE INDEX (type_status_date)
            INNER JOIN {$wpdb->postmeta} pm USE INDEX (meta_key) 
                ON p.ID = pm.post_id
            WHERE p.post_type = 'product'
            AND p.post_status = 'publish'
            AND pm.meta_key = '_price'
            AND pm.meta_value BETWEEN 100 AND 500
            ORDER BY p.post_date DESC
            LIMIT 20
        ";
        
        return $wpdb->get_results($sql);
    }

Advanced Search Implementation

Full-Text Search with Relevance


// Advanced search functionality
class Advanced_Search {
    
    /**
     * Full-text search with relevance scoring
     */
    public static function relevance_search($search_term) {
        global $wpdb;
        
        $search_term = '%' . $wpdb->esc_like($search_term) . '%';
        
        $sql = $wpdb->prepare("
            SELECT 
                p.ID,
                p.post_title,
                p.post_content,
                p.post_excerpt,
                (
                    -- Title match (highest weight)
                    CASE WHEN p.post_title LIKE %s THEN 10 ELSE 0 END +
                    -- Exact title match (bonus)
                    CASE WHEN p.post_title = %s THEN 5 ELSE 0 END +
                    -- Content match
                    CASE WHEN p.post_content LIKE %s THEN 3 ELSE 0 END +
                    -- Excerpt match
                    CASE WHEN p.post_excerpt LIKE %s THEN 2 ELSE 0 END +
                    -- Meta match
                    (SELECT COUNT(*) FROM {$wpdb->postmeta} pm 
                     WHERE pm.post_id = p.ID 
                     AND pm.meta_value LIKE %s) +
                    -- Term match
                    (SELECT COUNT(DISTINCT t.term_id) 
                     FROM {$wpdb->terms} t
                     INNER JOIN {$wpdb->term_taxonomy} tt ON t.term_id = tt.term_id
                     INNER JOIN {$wpdb->term_relationships} tr ON tt.term_taxonomy_id = tr.term_taxonomy_id
                     WHERE tr.object_id = p.ID
                     AND t.name LIKE %s)
                ) as relevance_score
            FROM {$wpdb->posts} p
            WHERE p.post_status = 'publish'
            AND p.post_type IN ('post', 'page', 'product')
            HAVING relevance_score > 0
            ORDER BY relevance_score DESC, p.post_date DESC
            LIMIT 50
        ",
        $search_term,
        trim($search_term, '%'),
        $search_term,
        $search_term,
        $search_term,
        $search_term
        );
        
        return $wpdb->get_results($sql);
    }
    
    /**
     * Faceted search implementation
     */
    public static function faceted_search($search_params) {
        global $wpdb;
        
        $where_clauses = ["p.post_type = 'product'", "p.post_status = 'publish'"];
        $join_clauses = [];
        $having_clauses = [];
        
        // Text search
        if (!empty($search_params['q'])) {
            $search = '%' . $wpdb->esc_like($search_params['q']) . '%';
            $where_clauses[] = $wpdb->prepare(
                "(p.post_title LIKE %s OR p.post_content LIKE %s)",
                $search, $search
            );
        }
        
        // Price range
        if (!empty($search_params['min_price']) || !empty($search_params['max_price'])) {
            $join_clauses[] = "LEFT JOIN {$wpdb->postmeta} pm_price ON p.ID = pm_price.post_id AND pm_price.meta_key = '_price'";
            
            if (!empty($search_params['min_price'])) {
                $where_clauses[] = $wpdb->prepare("CAST(pm_price.meta_value AS DECIMAL) >= %f", $search_params['min_price']);
            }
            if (!empty($search_params['max_price'])) {
                $where_clauses[] = $wpdb->prepare("CAST(pm_price.meta_value AS DECIMAL) <= %f", $search_params['max_price']);
            }
        }
        
        // Categories
        if (!empty($search_params['categories'])) {
            $cat_placeholders = implode(',', array_fill(0, count($search_params['categories']), '%d'));
            $join_clauses[] = "
                INNER JOIN {$wpdb->term_relationships} tr_cat ON p.ID = tr_cat.object_id
                INNER JOIN {$wpdb->term_taxonomy} tt_cat ON tr_cat.term_taxonomy_id = tt_cat.term_taxonomy_id
            ";
            $where_clauses[] = $wpdb->prepare(
                "tt_cat.taxonomy = 'product_category' AND tt_cat.term_id IN ($cat_placeholders)",
                ...$search_params['categories']
            );
        }
        
        // Build final query
        $sql = "
            SELECT DISTINCT p.ID, p.post_title
            FROM {$wpdb->posts} p
            " . implode(' ', $join_clauses) . "
            WHERE " . implode(' AND ', $where_clauses) . "
            " . (!empty($having_clauses) ? 'HAVING ' . implode(' AND ', $having_clauses) : '') . "
            ORDER BY p.post_date DESC
            LIMIT 50
        ";
        
        $results = $wpdb->get_results($sql);
        
        // Get facet counts
        $facets = self::get_search_facets($search_params);
        
        return [
            'results' => $results,
            'facets' => $facets,
            'total' => count($results)
        ];
    }
    
    /**
     * Get facet counts for search filters
     */
    private static function get_search_facets($current_filters) {
        global $wpdb;
        
        $facets = [];
        
        // Category facets
        $facets['categories'] = $wpdb->get_results("
            SELECT 
                t.term_id,
                t.name,
                COUNT(DISTINCT p.ID) as count
            FROM {$wpdb->posts} p
            INNER JOIN {$wpdb->term_relationships} tr ON p.ID = tr.object_id
            INNER JOIN {$wpdb->term_taxonomy} tt ON tr.term_taxonomy_id = tt.term_taxonomy_id
            INNER JOIN {$wpdb->terms} t ON tt.term_id = t.term_id
            WHERE p.post_type = 'product'
            AND p.post_status = 'publish'
            AND tt.taxonomy = 'product_category'
            GROUP BY t.term_id
            ORDER BY count DESC
            LIMIT 20
        ");
        
        // Price range facets
        $facets['price_ranges'] = $wpdb->get_results("
            SELECT 
                CASE 
                    WHEN CAST(meta_value AS DECIMAL) < 100 THEN 'Under $100'
                    WHEN CAST(meta_value AS DECIMAL) BETWEEN 100 AND 500 THEN '$100-$500'
                    WHEN CAST(meta_value AS DECIMAL) BETWEEN 500 AND 1000 THEN '$500-$1000'
                    ELSE 'Over $1000'
                END as price_range,
                COUNT(*) as count
            FROM {$wpdb->posts} p
            INNER JOIN {$wpdb->postmeta} pm ON p.ID = pm.post_id
            WHERE p.post_type = 'product'
            AND p.post_status = 'publish'
            AND pm.meta_key = '_price'
            GROUP BY price_range
            ORDER BY 
                CASE price_range
                    WHEN 'Under $100' THEN 1
                    WHEN '$100-$500' THEN 2
                    WHEN '$500-$1000' THEN 3
                    ELSE 4
                END
        ");
        
        return $facets;
    }

  • WP_Query supports complex meta and tax queries

  • Custom SQL is necessary for some complex relationships
  • Query optimization is crucial for performance
  • Caching strategies reduce database load
  • Advanced search requires relevance scoring

Query Best Practices

  • Use WP_Query when possible, custom SQL when necessary
  • Always sanitize and prepare SQL queries
  • Cache expensive queries
  • Use selective field loading for large datasets
  • Implement pagination for better performance
  • Index custom tables and meta keys
  • Monitor query performance with Query Monitor plugin
