REST API Fundamentals

What is REST?

REST (Representational State Transfer) is an architectural style for designing networked applications. The WordPress REST API follows REST principles:

  • Stateless: Each request contains all information needed
  • Client-Server: Separation of concerns
  • Cacheable: Responses can be cached
  • Uniform Interface: Consistent resource identification
  • Layered System: Architecture can have multiple layers

Request Lifecycle

Understanding the Flow


/**
 * WordPress REST API Request Lifecycle
 * 
 * 1. Request arrives at server
 * 2. WordPress loads and initializes
 * 3. REST API is initialized (rest_api_init hook)
 * 4. Request is parsed and routed
 * 5. Permission checks are performed
 * 6. Endpoint callback is executed
 * 7. Response is prepared and sent
 */

// Hook into different stages of the lifecycle
class REST_Lifecycle_Handler {
    
    public function __construct() {
        // Stage 1: API Initialization
        add_action('rest_api_init', [$this, 'on_api_init'], 10);
        
        // Stage 2: Pre-serve request
        add_filter('rest_pre_serve_request', [$this, 'pre_serve'], 10, 4);
        
        // Stage 3: Pre-dispatch
        add_filter('rest_pre_dispatch', [$this, 'pre_dispatch'], 10, 3);
        
        // Stage 4: Request processing
        add_filter('rest_request_before_callbacks', [$this, 'before_callbacks'], 10, 3);
        add_filter('rest_request_after_callbacks', [$this, 'after_callbacks'], 10, 3);
        
        // Stage 5: Response filtering
        add_filter('rest_post_dispatch', [$this, 'post_dispatch'], 10, 3);
        
        // Stage 6: Final output
        add_filter('rest_pre_echo_response', [$this, 'pre_echo'], 10, 3);
    }
    
    public function on_api_init() {
        // Register routes, configure settings
        error_log('REST API initialized');
    }
    
    public function pre_serve($served, $result, $request, $server) {
        // Intercept before WordPress serves the request
        // Return true to prevent default serving
        error_log('Pre-serve: ' . $request->get_route());
        return $served;
    }
    
    public function pre_dispatch($result, $server, $request) {
        // Modify request before dispatch
        error_log('Pre-dispatch: ' . $request->get_method() . ' ' . $request->get_route());
        
        // Add custom headers
        $request->set_header('X-API-Version', '1.0');
        
        // Log request data
        $this->log_request($request);
        
        return $result;
    }
    
    public function before_callbacks($response, $handler, $request) {
        // Before endpoint callback execution
        error_log('Before callback: ' . $request->get_route());
        
        // Validate request data
        $validation = $this->validate_request($request);
        if (is_wp_error($validation)) {
            return $validation;
        }
        
        return $response;
    }
    
    public function after_callbacks($response, $handler, $request) {
        // After endpoint callback execution
        error_log('After callback: ' . $request->get_route());
        
        // Modify response data
        if (!is_wp_error($response)) {
            $data = $response->get_data();
            $data['processed_at'] = current_time('mysql');
            $response->set_data($data);
        }
        
        return $response;
    }
    
    public function post_dispatch($result, $server, $request) {
        // After dispatch, before output
        error_log('Post-dispatch: ' . $request->get_route());
        
        // Add response headers
        if ($result instanceof WP_REST_Response) {
            $result->header('X-Total-Time', timer_stop(0, 3));
            $result->header('X-Request-ID', wp_generate_uuid4());
        }
        
        return $result;
    }
    
    public function pre_echo($result, $server, $request) {
        // Final modification before output
        error_log('Pre-echo: ' . $request->get_route());
        
        // Cache the response
        $this->cache_response($request, $result);
        
        return $result;
    }
    
    private function log_request($request) {
        $log_data = [
            'timestamp' => current_time('mysql'),
            'method' => $request->get_method(),
            'route' => $request->get_route(),
            'params' => $request->get_params(),
            'headers' => $request->get_headers(),
            'ip' => $_SERVER['REMOTE_ADDR'] ?? '',
        ];
        
        error_log('API Request: ' . json_encode($log_data));
    }
    
    private function validate_request($request) {
        // Custom validation logic
        return true;
    }
    
    private function cache_response($request, $response) {
        if ($request->get_method() === 'GET' && !is_wp_error($response)) {
            $cache_key = 'rest_' . md5($request->get_route() . serialize($request->get_params()));
            set_transient($cache_key, $response, HOUR_IN_SECONDS);
        }
    }
}

class My_Custom_REST_Server extends WP_REST_Server {
    
    /**
     * Override serve_request to add custom logic
     */
    public function serve_request($path = null) {
        // Add CORS headers
        $this->send_cors_headers();
        
        // Add rate limiting
        if (!$this->check_rate_limit()) {
            $this->send_error(
                new WP_Error('rate_limit_exceeded', 'Too many requests', ['status' => 429])
            );
            return;
        }
        
        // Continue with normal processing
        return parent::serve_request($path);
    }
    
    protected function send_cors_headers() {
        $origin = get_http_origin();
        
        if ($origin) {
            header('Access-Control-Allow-Origin: ' . esc_url_raw($origin));
            header('Access-Control-Allow-Methods: GET, POST, PUT, DELETE, OPTIONS');
            header('Access-Control-Allow-Credentials: true');
            header('Access-Control-Allow-Headers: Authorization, Content-Type, X-WP-Nonce');
        }
    }
    
    protected function check_rate_limit() {
        $identifier = $this->get_client_identifier();
        $key = 'rate_limit_' . $identifier;
        $limit = 100; // 100 requests per hour
        
        $count = get_transient($key) ?: 0;
        
        if ($count >= $limit) {
            return false;
        }
        
        set_transient($key, $count + 1, HOUR_IN_SECONDS);
        return true;
    }
    
    protected function get_client_identifier() {
        if (is_user_logged_in()) {
            return 'user_' . get_current_user_id();
        }
        return 'ip_' . $_SERVER['REMOTE_ADDR'];
    }
}

// Working with REST requests
class REST_Request_Handler {
    
    /**

     */
    public function create_custom_request() {
        // Create a new request
        $request = new WP_REST_Request('GET', '/wp/v2/posts');

        $request->set_param('per_page', 10);

        $request->set_param('order', 'desc');

        $request->set_query_params([
            'meta_key' => '_featured',
            'meta_value' => 'yes'
        ]);
        
        // Set body parameters (for POST/PUT)
        $request->set_body_params([
            'title' => 'New Post Title',
            'content' => 'Post content here'
        ]);
        
        // Set headers
        $request->set_header('X-Custom-Header', 'value');
        $request->set_header('Authorization', 'Bearer token123');
        
        // Set JSON body
        $request->set_body(json_encode([
            'data' => 'value'
        ]));
        
        // Get merged parameters
        $all_params = $request->get_params();
        
        // Check if parameter exists
        if ($request->has_param('per_page')) {
            $per_page = $request->get_param('per_page');
        }
        
        // Sanitize and validate
        $request->sanitize_params();
        $valid = $request->has_valid_params();
        
        // Get route
        $route = $request->get_route();
        
        // Get method
        $method = $request->get_method();
        
        // Check for file uploads
        $files = $request->get_file_params();
        
        return $request;
    }
    
    /**
     * Process incoming request
     */
    public function process_request(WP_REST_Request $request) {
        // Get all data from request
        $data = [
            'method' => $request->get_method(),
            'route' => $request->get_route(),
            'params' => $request->get_params(),
            'headers' => $request->get_headers(),
            'body' => $request->get_body(),
            'content_type' => $request->get_content_type(),
        ];
        
        // Check authentication
        $user = wp_get_current_user();
        if ($user->ID === 0) {
            // Not authenticated
            $nonce = $request->get_header('X-WP-Nonce');
            if ($nonce && wp_verify_nonce($nonce, 'wp_rest')) {
                // Nonce is valid
                wp_set_current_user($request->get_param('user_id'));
            }
        }
        
        // Validate request against schema
        $schema = $this->get_endpoint_schema();
        $valid = rest_validate_value_from_schema(
            $request->get_params(),
            $schema
        );
        
        if (is_wp_error($valid)) {
            return $valid;
        }
        
        // Process the request
        return $this->handle_request($data);
    }
    
    private function get_endpoint_schema() {
        return [
            'type' => 'object',
            'properties' => [
                'title' => [
                    'type' => 'string',
                    'required' => true,
                    'minLength' => 1,
                    'maxLength' => 200,
                ],
                'content' => [
                    'type' => 'string',
                    'required' => true,
                ],
                'status' => [
                    'type' => 'string',
                    'enum' => ['publish', 'draft', 'private'],
                    'default' => 'draft',
                ],
            ],
        ];
    }
}

        $response = new WP_REST_Response([

        $created_response = new WP_REST_Response([

            'message' => 'Resource created successfully',
        ]);
        $created_response->set_status(201);
        $created_response->header('Location', rest_url('/wp/v2/posts/456'));
        
        // No content response (204)
        $no_content = new WP_REST_Response(null);
        $no_content->set_status(204);
        
        // Error response
        $error = new WP_Error(
            'invalid_data',
            'The provided data is invalid',
            ['status' => 400]
        );
        
        // Paginated response
        $paginated = $this->create_paginated_response();
        
        // Response with links
        $linked = $this->create_linked_response();
        
        return $response;
    }
    
    /**
     * Create paginated response with headers
     */
    public function create_paginated_response() {
        $data = [
            'items' => [
                ['id' => 1, 'title' => 'Item 1'],
                ['id' => 2, 'title' => 'Item 2'],
            ],
        ];
        
        $response = new WP_REST_Response($data);
        
        // Add pagination headers
        $response->header('X-WP-Total', 100);
        $response->header('X-WP-TotalPages', 10);
        
        // Add Link header for pagination
        $links = [];
        $links[] = '<' . rest_url('/wp/v2/posts?page=2') . '>; rel="next"';
        $links[] = '<' . rest_url('/wp/v2/posts?page=10') . '>; rel="last"';
        $response->header('Link', implode(', ', $links));
        
        return $response;
    }
    
    /**
     * Create response with HAL links
     */
    public function create_linked_response() {
        $data = [
            'id' => 123,
            'title' => 'Sample Post',
            'author' => 1,
        ];
        
        $response = rest_ensure_response($data);
        
        // Add links
        $response->add_links([
            'self' => [
                'href' => rest_url('/wp/v2/posts/123'),
            ],
            'collection' => [
                'href' => rest_url('/wp/v2/posts'),
            ],
            'author' => [
                'href' => rest_url('/wp/v2/users/1'),
                'embeddable' => true,
            ],
            'wp:attachment' => [
                'href' => rest_url('/wp/v2/media?parent=123'),
            ],
            'wp:term' => [
                [
                    'taxonomy' => 'category',
                    'embeddable' => true,
                    'href' => rest_url('/wp/v2/categories?post=123'),
                ],
                [
                    'taxonomy' => 'post_tag',
                    'embeddable' => true,
                    'href' => rest_url('/wp/v2/tags?post=123'),
                ],
            ],
        ]);
        
        return $response;
    }
    
    /**
     * Handle embedded data
     */
    public function add_embedded_data(WP_REST_Response $response, WP_REST_Request $request) {
        $embed = $request->get_param('_embed');
        
        if ($embed) {
            $data = $response->get_data();
            
            // Add embedded author data
            if (isset($data['author'])) {
                $author = get_user_by('id', $data['author']);
                $response->add_link('author', rest_url('/wp/v2/users/' . $data['author']), [
                    'embeddable' => true,
                ]);
                
                // Add to _embedded
                $embedded = $response->get_data();
                $embedded['_embedded'] = [
                    'author' => [
                        [
                            'id' => $author->ID,
                            'name' => $author->display_name,
                            'avatar' => get_avatar_url($author->ID),
                        ],
                    ],
                ];
                $response->set_data($embedded);
            }
        }
        
        return $response;
    }
}

// Define and use JSON Schema for validation
class REST_Schema_Handler {
    
    /**
     * Get item schema for custom endpoint
     */
    public function get_item_schema() {
        if ($this->schema) {
            return $this->schema;
        }
        
        $this->schema = [
            '$schema' => 'http://json-schema.org/draft-04/schema#',
            'title' => 'product',
            'type' => 'object',
            'properties' => [
                'id' => [
                    'description' => 'Unique identifier for the product.',
                    'type' => 'integer',
                    'context' => ['view', 'edit', 'embed'],
                    'readonly' => true,
                ],
                'title' => [
                    'description' => 'The title of the product.',
                    'type' => 'string',
                    'context' => ['view', 'edit', 'embed'],
                    'required' => true,
                    'minLength' => 1,
                    'maxLength' => 200,
                    'pattern' => '^[a-zA-Z0-9 ]+$',
                ],
                'price' => [
                    'description' => 'The price of the product.',
                    'type' => 'number',
                    'context' => ['view', 'edit'],
                    'required' => true,
                    'minimum' => 0,
                    'maximum' => 999999.99,
                ],
                'status' => [
                    'description' => 'Product availability status.',
                    'type' => 'string',
                    'context' => ['view', 'edit'],
                    'enum' => ['available', 'out_of_stock', 'discontinued'],
                    'default' => 'available',
                ],
                'categories' => [
                    'description' => 'Product categories.',
                    'type' => 'array',
                    'context' => ['view', 'edit'],
                    'items' => [
                        'type' => 'integer',
                    ],
                    'uniqueItems' => true,
                ],
                'metadata' => [
                    'description' => 'Additional product metadata.',
                    'type' => 'object',
                    'context' => ['view', 'edit'],
                    'properties' => [
                        'sku' => [
                            'type' => 'string',
                            'pattern' => '^[A-Z0-9-]+$',
                        ],
                        'weight' => [
                            'type' => 'number',
                            'minimum' => 0,
                        ],
                        'dimensions' => [
                            'type' => 'object',
                            'properties' => [
                                'length' => ['type' => 'number'],
                                'width' => ['type' => 'number'],
                                'height' => ['type' => 'number'],
                            ],
                            'additionalProperties' => false,
                        ],
                    ],
                ],
                'created_date' => [
                    'description' => 'The date the product was created.',
                    'type' => 'string',
                    'format' => 'date-time',
                    'context' => ['view', 'edit'],
                    'readonly' => true,
                ],
            ],
            'additionalProperties' => false,
        ];
        
        return $this->schema;
    }
    
    /**
     * Validate data against schema
     */
    public function validate_against_schema($data, $schema = null) {
        if (!$schema) {
            $schema = $this->get_item_schema();
        }
        
        // Use WordPress validation
        $valid = rest_validate_value_from_schema($data, $schema);
        
        if (is_wp_error($valid)) {
            return $valid;
        }
        
        // Additional custom validation
        $custom_valid = $this->custom_validation($data);
        
        if (is_wp_error($custom_valid)) {
            return $custom_valid;
        }
        
        return true;
    }
    
    /**
     * Custom validation logic
     */
    protected function custom_validation($data) {
        // Check SKU uniqueness
        if (isset($data['metadata']['sku'])) {
            $existing = $this->get_product_by_sku($data['metadata']['sku']);
            if ($existing && $existing->ID !== $data['id']) {
                return new WP_Error(
                    'duplicate_sku',
                    'A product with this SKU already exists',
                    ['status' => 400]
                );
            }
        }
        
        // Validate price and status relationship
        if (isset($data['price']) && isset($data['status'])) {
            if ($data['price'] <= 0 && $data['status'] === 'available') {
                return new WP_Error(
                    'invalid_price_status',
                    'Product cannot be available with zero price',
                    ['status' => 400]
                );
            }
        }
        
        return true;
    }
    
    /**
     * Sanitize data based on schema
     */
    public function sanitize_data($data, $schema = null) {
        if (!$schema) {
            $schema = $this->get_item_schema();
        }
        
        $sanitized = [];
        
        foreach ($schema['properties'] as $property => $property_schema) {
            if (!isset($data[$property])) {
                continue;
            }
            
            $sanitized[$property] = $this->sanitize_property(
                $data[$property],
                $property_schema
            );
        }
        
        return $sanitized;
    }
    
    protected function sanitize_property($value, $schema) {
        switch ($schema['type']) {
            case 'string':
                return sanitize_text_field($value);
                
            case 'integer':
                return (int) $value;
                
            case 'number':
                return (float) $value;
                
            case 'boolean':
                return (bool) $value;
                
            case 'array':
                if (!is_array($value)) {
                    return [];
                }
                return array_map(function($item) use ($schema) {
                    if (isset($schema['items'])) {
                        return $this->sanitize_property($item, $schema['items']);
                    }
                    return $item;
                }, $value);
                
            case 'object':
                if (!is_array($value)) {
                    return [];
                }
                if (isset($schema['properties'])) {
                    return $this->sanitize_data($value, $schema);
                }
                return $value;
                
            default:
                return $value;
        }
    }
}

Error Handling

Comprehensive Error Management


// Advanced error handling for REST API
class REST_Error_Handler {
    
    public function __construct() {
        add_filter('rest_request_after_callbacks', [$this, 'handle_errors'], 10, 3);
        add_filter('rest_post_dispatch', [$this, 'format_error_response'], 10, 3);
    }
    
    /**
     * Create structured error responses
     */
    public function create_error($code, $message, $status = 400, $additional_data = []) {
        $error = new WP_Error($code, $message, array_merge(
            ['status' => $status],
            $additional_data
        ));
        
        // Add additional error details
        $error->add_data([
            'timestamp' => current_time('c'),
            'path' => $_SERVER['REQUEST_URI'],
            'method' => $_SERVER['REQUEST_METHOD'],
        ], 'error_context');
        
        return $error;
    }
    
    /**
     * Handle different types of errors
     */
    public function handle_errors($response, $handler, $request) {
        if (is_wp_error($response)) {
            // Log the error
            $this->log_error($response, $request);
            
            // Add helpful information for developers
            if (WP_DEBUG) {
                $response->add_data([
                    'trace' => debug_backtrace(DEBUG_BACKTRACE_IGNORE_ARGS),
                    'request_data' => $request->get_params(),
                ], 'debug_info');
            }
            
            // Transform specific errors
            $response = $this->transform_error($response);
        }
        
        return $response;
    }
    
    /**
     * Transform WordPress errors to REST API errors
     */
    protected function transform_error(WP_Error $error) {
        $code = $error->get_error_code();
        $message = $error->get_error_message();
        $data = $error->get_error_data();
        
        // Map WordPress errors to HTTP status codes
        $status_map = [
            'rest_forbidden' => 403,
            'rest_not_found' => 404,
            'rest_invalid_param' => 400,
            'rest_missing_param' => 400,
            'rest_unauthorized' => 401,
            'rest_too_many_requests' => 429,
        ];
        
        $status = $status_map[$code] ?? ($data['status'] ?? 500);
        
        // Create structured error response
        $response_data = [
            'code' => $code,
            'message' => $message,
            'data' => [
                'status' => $status,
            ],
        ];
        
        // Add validation errors if present
        if ($code === 'rest_invalid_param') {
            $response_data['data']['params'] = $this->format_validation_errors($error);
        }
        
        // Add additional error information
        if (isset($data['additional_errors'])) {
            $response_data['additional_errors'] = $data['additional_errors'];
        }
        
        $new_error = new WP_Error($code, $message, $response_data);
        
        return $new_error;
    }
    
    /**
     * Format validation errors
     */
    protected function format_validation_errors(WP_Error $error) {
        $params = [];
        $data = $error->get_error_data();
        
        if (isset($data['params'])) {
            foreach ($data['params'] as $param => $param_error) {
                $params[$param] = [
                    'code' => 'invalid_param',
                    'message' => $param_error,
                ];
            }
        }
        
        return $params;
    }
    
    /**
     * Format error response for output
     */
    public function format_error_response($result, $server, $request) {
        if (is_wp_error($result)) {
            $error_data = $result->get_error_data();
            $status = isset($error_data['status']) ? $error_data['status'] : 500;
            
            $response = new WP_REST_Response([
                'code' => $result->get_error_code(),
                'message' => $result->get_error_message(),
                'data' => $error_data,
            ]);
            
            $response->set_status($status);
            
            // Add error-specific headers
            $response->header('X-WP-Error', 'true');
            $response->header('X-WP-Error-Code', $result->get_error_code());
            
            return $response;
        }
        
        return $result;
    }
    
    /**
     * Log errors for monitoring
     */
    protected function log_error(WP_Error $error, WP_REST_Request $request) {
        $log_entry = [
            'timestamp' => current_time('c'),
            'error_code' => $error->get_error_code(),
            'error_message' => $error->get_error_message(),
            'error_data' => $error->get_error_data(),
            'request' => [
                'method' => $request->get_method(),
                'route' => $request->get_route(),
                'params' => $request->get_params(),
            ],
            'user' => get_current_user_id(),
            'ip' => $_SERVER['REMOTE_ADDR'] ?? '',
        ];
        
        // Log to file or database
        error_log('REST API Error: ' . json_encode($log_entry));
        
        // Could also save to database for analysis
        $this->save_error_to_database($log_entry);
    }
    
    protected function save_error_to_database($log_entry) {
        global $wpdb;
        
        $table = $wpdb->prefix . 'rest_api_errors';
        
        $wpdb->insert($table, [
            'error_code' => $log_entry['error_code'],
            'error_message' => $log_entry['error_message'],
            'request_method' => $log_entry['request']['method'],
            'request_route' => $log_entry['request']['route'],
            'user_id' => $log_entry['user'],
            'ip_address' => $log_entry['ip'],
            'created_at' => $log_entry['timestamp'],
        ]);
    }
}

new REST_Error_Handler();

API Configuration and Settings

Configuring REST API Behavior


// Configure and customize REST API settings
class REST_API_Configuration {
    
    public function __construct() {
        $this->configure_api();
    }
    
    protected function configure_api() {
        // Change REST API prefix
        add_filter('rest_url_prefix', [$this, 'change_api_prefix']);
        
        // Disable REST API for non-authenticated users
        add_filter('rest_authentication_errors', [$this, 'restrict_api_access']);
        
        // Remove default endpoints
        add_filter('rest_endpoints', [$this, 'remove_default_endpoints']);
        
        // Configure allowed HTTP methods
        add_filter('rest_allowed_http_methods', [$this, 'set_allowed_methods']);
        
        // Set default per_page value
        add_filter('rest_post_collection_params', [$this, 'set_default_params'], 10, 2);
    }
    
    public function change_api_prefix() {
        return 'api'; // Changes /wp-json to /api
    }
    
    public function restrict_api_access($result) {
        if (!is_user_logged_in()) {
            return new WP_Error(
                'rest_not_logged_in',
                'You must be logged in to access the REST API.',
                ['status' => 401]
            );
        }
        return $result;
    }
    
    public function remove_default_endpoints($endpoints) {
        // Remove user endpoint for security
        unset($endpoints['/wp/v2/users']);
        unset($endpoints['/wp/v2/users/(?P[\d]+)']);
        
        // Remove comments if not needed
        unset($endpoints['/wp/v2/comments']);
        
        return $endpoints;
    }
    
    public function set_allowed_methods($methods) {
        // Only allow specific HTTP methods
        return ['GET', 'POST', 'PUT', 'DELETE'];
    }
    
    public function set_default_params($params, $post_type) {
        // Change default per_page from 10 to 20
        if (isset($params['per_page'])) {
            $params['per_page']['default'] = 20;
        }
        
        // Add custom parameters
        $params['custom_param'] = [
            'description' => 'Custom parameter for filtering',
            'type' => 'string',
            'sanitize_callback' => 'sanitize_text_field',
        ];
        
        return $params;
    }
}

new REST_API_Configuration();
