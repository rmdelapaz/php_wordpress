Dynamic Blocks and Patterns

Create server-rendered blocks and reusable design patterns

Learning Objectives

By the end of this lesson, you will be able to:

  • Build server-side rendered dynamic blocks
  • Create and register block patterns
  • Implement pattern categories
  • Use PHP render callbacks effectively
  • Create context-aware blocks
  • Build template part blocks

Dynamic Blocks Architecture

Complete Dynamic Block Implementation


<?php
/**
 * Dynamic block with server-side rendering
 */
class Dynamic_Block_Manager {
    
    private $block_name = 'myplugin/dynamic-content';
    
    public function __construct() {
        add_action('init', [$this, 'register_block']);
        add_action('rest_api_init', [$this, 'register_rest_routes']);
        add_action('wp_enqueue_scripts', [$this, 'enqueue_frontend_assets']);

    /**
     * Register the dynamic block
     */
    public function register_block() {
        // Register block script
        wp_register_script(
            'dynamic-block-editor',
            plugin_dir_url(__FILE__) . 'build/index.js',
            ['wp-blocks', 'wp-element', 'wp-editor', 'wp-components', 'wp-data'],

        // Register block styles
        wp_register_style(
            'dynamic-block-editor',
            plugin_dir_url(__FILE__) . 'build/editor.css',
            ['wp-edit-blocks'],
            filemtime(plugin_dir_path(__FILE__) . 'build/editor.css')
        );
        
        wp_register_style(
            'dynamic-block-style',
            plugin_dir_url(__FILE__) . 'build/style.css',
            [],
            filemtime(plugin_dir_path(__FILE__) . 'build/style.css')
        );
        
        // Register the block with render callback
        register_block_type($this->block_name, [
            'editor_script' => 'dynamic-block-editor',
            'editor_style' => 'dynamic-block-editor',
            'style' => 'dynamic-block-style',
            'render_callback' => [$this, 'render_block'],

            'uses_context' => ['postId', 'postType', 'queryId'],
            'provides_context' => [

            ]
        ]);
    }

        // Start output buffering
        ob_start();

        $classes = $this->get_block_classes($attributes);
        $wrapper_attributes = get_block_wrapper_attributes(['class' => $classes]);

    /**
     * REST API endpoint for preview
     */
    public function register_rest_routes() {
        register_rest_route('myplugin/v1', '/preview-dynamic-block', [
            'methods' => 'POST',
            'callback' => [$this, 'preview_block'],
            'permission_callback' => function() {
                return current_user_can('edit_posts');
            },
            'args' => [
                'attributes' => [
                    'type' => 'object',
                    'required' => true
                ]
            ]
        ]);
    }
    
    /**
     * Preview block via REST API
     */
    public function preview_block($request) {
        $attributes = $request->get_param('attributes');
        $block = [
            'blockName' => $this->block_name,
            'attrs' => $attributes,
            'context' => []
        ];
        
        $rendered = $this->render_block($attributes, '', (object)$block);
        
        return [
            'rendered' => $rendered,
            'success' => true
        ];
    }
}

    /**
     * Register dynamic patterns (generated from data)
     */
    public function register_dynamic_patterns() {
        // Get recent posts for dynamic pattern
        $recent_posts = get_posts([
            'numberposts' => 3,
            'post_status' => 'publish'
        ]);
        
        if (!empty($recent_posts)) {
            $content = $this->generate_posts_pattern($recent_posts);
            
            register_block_pattern(
                'myplugin/recent-posts-dynamic',
                [
                    'title' => __('Recent Posts (Dynamic)', 'myplugin'),
                    'description' => __('Automatically generated from recent posts', 'myplugin'),
                    'categories' => ['query'],
                    'content' => $content
                ]
            );
        }
        
        // Product showcase pattern (WooCommerce)
        if (class_exists('WooCommerce')) {
            $this->register_product_patterns();
        }
    }
    
    /**
     * Generate posts pattern from actual posts
     */
    private function generate_posts_pattern($posts) {
        $content = '

    ';
        
        foreach ($posts as $post) {

            $content .= '

            if ($thumbnail_url) {
                $content .= '
            
            
            ';
            }
            
            $content .= '

' . esc_html($excerpt) . '

            
            
            
            

                Read More
            

            
        

        
    

    ';
        }
        
        $content .= '
    


';
        
        return $content;
    }

    const categories = useSelect((select) => {

Summary

Key Takeaways

  • Dynamic blocks render on the server for fresh content
  • PHP render callbacks provide flexibility
  • Block patterns enable design reuse
  • Pattern categories organize designs
  • Template parts create reusable sections
  • Context awareness enables smart blocks

Dynamic Block Best Practices

  • Cache dynamic content when possible
  • Use ServerSideRender for preview
  • Implement proper error handling
  • Optimize database queries
  • Provide meaningful placeholders
  • Consider performance impact
  • Test with various data sets
