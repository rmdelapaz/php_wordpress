Homework: Architect a WordPress Application

Design and build a professional WordPress application structure

Assignment Overview

In this homework assignment, you'll architect and build a complete WordPress application using the patterns and principles learned in this session. You'll create a Event Management System that demonstrates professional WordPress development practices.

Learning Objectives

  • Apply MVC architecture to a WordPress application
  • Implement separation of concerns
  • Create reusable components
  • Use design patterns effectively
  • Structure a scalable WordPress project

Project Requirements

Functional Requirements

Build an Event Management System with the following features:

  • Event Management:
    • Create, read, update, delete events
    • Event details: title, description, date, time, location, capacity
    • Featured image and gallery
    • Event categories and tags
  • Registration System:
    • User registration for events
    • Capacity management
    • Waiting list functionality
    • Registration confirmation emails
  • User Dashboard:
    • View registered events
    • Cancel registrations
    • Download tickets/confirmations
  • Admin Interface:
    • Manage events
    • View registrations
    • Export attendee lists
    • Analytics dashboard

Technical Requirements

  • Implement MVC architecture
  • Use service layer for business logic
  • Create repository pattern for data access
  • Build at least 3 reusable components
  • Implement dependency injection
  • Use proper project structure
  • Include configuration management
  • Write clean, documented code

<?php
/**
 * Plugin Name: Event Management System
 * Description: Professional event management for WordPress
 * Version: 1.0.0
 * Author: Your Name
 * Text Domain: event-management
 */

namespace EventManagement;

// Prevent direct access
if (!defined('ABSPATH')) {
    exit;
}

// Define plugin constants
define('EM_PLUGIN_FILE', __FILE__);
define('EM_PLUGIN_DIR', plugin_dir_path(__FILE__));
define('EM_PLUGIN_URL', plugin_dir_url(__FILE__));
define('EM_PLUGIN_VERSION', '1.0.0');

// Autoloader
require_once EM_PLUGIN_DIR . 'vendor/autoload.php';

// Bootstrap application
require_once EM_PLUGIN_DIR . 'bootstrap/app.php';

// Initialize application
add_action('plugins_loaded', function() {
    $app = Application::getInstance();
    $app->boot();
});

// Activation hook
register_activation_hook(__FILE__, function() {
    Application::getInstance()->activate();
});

// Deactivation hook
register_deactivation_hook(__FILE__, function() {
    Application::getInstance()->deactivate();
});

Step 2: Create Application Class


// src/Application.php
namespace EventManagement;

use EventManagement\Services\Container;

class Application {
    private static $instance;
    private $container;
    private $providers = [];
    private $booted = false;
    
    private function __construct() {
        $this->container = new Container();
    }
    
    public static function getInstance() {
        if (!self::$instance) {
            self::$instance = new self();
        }
        return self::$instance;
    }
    
    public function boot() {
        if ($this->booted) {
            return;
        }
        
        $this->loadConfiguration();
        $this->registerServices();
        $this->registerProviders();
        $this->bootProviders();
        $this->registerHooks();
        
        $this->booted = true;
    }
    
    private function loadConfiguration() {
        $configs = glob(EM_PLUGIN_DIR . 'config/*.php');
        
        foreach ($configs as $config) {
            $key = basename($config, '.php');
            $this->container->singleton("config.{$key}", function() use ($config) {
                return require $config;
            });
        }
    }
    
    private function registerServices() {
        // Register core services
        $this->container->singleton('db', function() {
            global $wpdb;
            return $wpdb;
        });
        
        $this->container->singleton('view', function() {
            return new Views\View(EM_PLUGIN_DIR . 'resources/views');
        });
        
        // Register repositories
        $this->container->singleton(
            Repositories\EventRepository::class,
            Repositories\EventRepository::class
        );
        
        $this->container->singleton(
            Repositories\RegistrationRepository::class,
            Repositories\RegistrationRepository::class
        );
        
        // Register services
        $this->container->singleton(
            Services\EventService::class,
            Services\EventService::class
        );
        
        $this->container->singleton(
            Services\RegistrationService::class,
            Services\RegistrationService::class
        );
        
        $this->container->singleton(
            Services\EmailService::class,
            Services\EmailService::class
        );
    }
    
    private function registerProviders() {
        $providers = require EM_PLUGIN_DIR . 'bootstrap/providers.php';
        
        foreach ($providers as $provider) {
            $instance = new $provider($this->container);
            $instance->register();
            $this->providers[] = $instance;
        }
    }
    
    private function bootProviders() {
        foreach ($this->providers as $provider) {
            $provider->boot();
        }
    }
    
    private function registerHooks() {
        add_action('init', [$this, 'init']);
        add_action('admin_menu', [$this, 'adminMenu']);
        add_action('rest_api_init', [$this, 'registerApiRoutes']);
        add_action('wp_enqueue_scripts', [$this, 'enqueuePublicAssets']);
        add_action('admin_enqueue_scripts', [$this, 'enqueueAdminAssets']);
    }
    
    public function init() {
        // Register post types and taxonomies
        $this->registerPostTypes();
        $this->registerTaxonomies();
        
        // Load text domain
        load_plugin_textdomain(
            'event-management',
            false,
            dirname(plugin_basename(EM_PLUGIN_FILE)) . '/resources/languages'
        );
    }
    
    public function activate() {
        // Create database tables
        require_once EM_PLUGIN_DIR . 'database/migrations/create_tables.php';
        
        // Set default options
        add_option('event_management_version', EM_PLUGIN_VERSION);
        
        // Flush rewrite rules
        flush_rewrite_rules();
    }
    
    public function deactivate() {
        // Clean up
        flush_rewrite_rules();
    }
    
    public function get($abstract) {
        return $this->container->get($abstract);
    }
}

// Helper function
function app($abstract = null) {
    $app = Application::getInstance();
    
    if ($abstract) {
        return $app->get($abstract);
    }
    
    return $app;
}

Step 3: Implement Model Layer


// src/Models/Event.php
namespace EventManagement\Models;

class Event extends Model {
    protected $table = 'em_events';
    
    protected $fillable = [
        'title',
        'description',
        'start_date',
        'end_date',
        'location',
        'capacity',
        'price',
        'featured_image',
        'status'
    ];
    
    protected $casts = [
        'start_date' => 'datetime',
        'end_date' => 'datetime',
        'capacity' => 'integer',
        'price' => 'float'
    ];
    
    public function registrations() {
        return $this->hasMany(Registration::class, 'event_id');
    }
    
    public function getAvailableSeats() {
        $registered = $this->registrations()
            ->where('status', 'confirmed')
            ->count();
        
        return $this->capacity - $registered;
    }
    
    public function isFull() {
        return $this->getAvailableSeats() <= 0;
    }
    
    public function isUpcoming() {
        return $this->start_date > new \DateTime();
    }
    
    public function isPast() {
        return $this->end_date < new \DateTime();
    }
    
    public function isOngoing() {
        $now = new \DateTime();
        return $this->start_date <= $now && $this->end_date >= $now;
    }
}

Step 4: Create Service Layer


// src/Services/RegistrationService.php
namespace EventManagement\Services;

use EventManagement\Models\Event;
use EventManagement\Models\Registration;
use EventManagement\Repositories\EventRepository;
use EventManagement\Repositories\RegistrationRepository;

class RegistrationService {
    private $eventRepository;
    private $registrationRepository;
    private $emailService;
    
    public function __construct(
        EventRepository $eventRepository,
        RegistrationRepository $registrationRepository,
        EmailService $emailService
    ) {
        $this->eventRepository = $eventRepository;
        $this->registrationRepository = $registrationRepository;
        $this->emailService = $emailService;
    }
    
    public function register($eventId, $attendeeData) {
        $event = $this->eventRepository->find($eventId);
        
        if (!$event) {
            throw new \Exception('Event not found');
        }
        
        if ($event->isFull()) {
            return $this->addToWaitingList($event, $attendeeData);
        }
        
        // Check if already registered
        if ($this->isAlreadyRegistered($eventId, $attendeeData['email'])) {
            throw new \Exception('You are already registered for this event');
        }
        
        // Create registration
        $registration = $this->registrationRepository->create([
            'event_id' => $eventId,
            'user_id' => get_current_user_id(),
            'name' => $attendeeData['name'],
            'email' => $attendeeData['email'],
            'phone' => $attendeeData['phone'],
            'status' => 'confirmed',
            'registration_date' => current_time('mysql'),
            'ticket_code' => $this->generateTicketCode()
        ]);
        
        // Send confirmation email
        $this->emailService->sendRegistrationConfirmation($registration);
        
        // Trigger action for other plugins
        do_action('event_registration_completed', $registration, $event);
        
        return $registration;
    }
    
    public function cancel($registrationId, $reason = '') {
        $registration = $this->registrationRepository->find($registrationId);
        
        if (!$registration) {
            throw new \Exception('Registration not found');
        }
        
        // Check permissions
        if (!$this->canCancelRegistration($registration)) {
            throw new \Exception('You do not have permission to cancel this registration');
        }
        
        // Update status
        $registration->status = 'cancelled';
        $registration->cancellation_date = current_time('mysql');
        $registration->cancellation_reason = $reason;
        $registration->save();
        
        // Check waiting list
        $this->processWaitingList($registration->event_id);
        
        // Send cancellation email
        $this->emailService->sendCancellationConfirmation($registration);
        
        return $registration;
    }
    
    private function generateTicketCode() {
        return strtoupper(wp_generate_password(8, false));
    }
    
    private function isAlreadyRegistered($eventId, $email) {
        return $this->registrationRepository->exists([
            'event_id' => $eventId,
            'email' => $email,
            'status' => ['confirmed', 'waiting']
        ]);
    }
    
    private function canCancelRegistration($registration) {
        // Admin can always cancel
        if (current_user_can('manage_options')) {
            return true;
        }
        
        // User can cancel their own registration
        if (is_user_logged_in() && $registration->user_id == get_current_user_id()) {
            return true;
        }
        
        return false;
    }
    
    private function processWaitingList($eventId) {
        $event = $this->eventRepository->find($eventId);
        
        if ($event->getAvailableSeats() > 0) {
            $waitingRegistration = $this->registrationRepository->getNextWaiting($eventId);
            
            if ($waitingRegistration) {
                $waitingRegistration->status = 'confirmed';
                $waitingRegistration->save();
                
                $this->emailService->sendWaitingListPromotion($waitingRegistration);
            }
        }
    }
}

Step 5: Build Reusable Components

Event Card Component


// src/Components/EventCard/EventCard.php
namespace EventManagement\Components;

class EventCard extends Component {
    private $event;
    private $showRegistrationButton = true;
    private $showDetails = true;
    
    public function __construct($event, $options = []) {
        $this->event = $event;
        
        foreach (['showRegistrationButton', 'showDetails'] as $option) {
            if (isset($options[$option])) {
                $this->$option = $options[$option];
            }
        }
        
        parent::__construct();
    }
    
    protected function getTemplateData() {
        return [
            'event' => $this->event,
            'availableSeats' => $this->event->getAvailableSeats(),
            'statusClass' => $this->getStatusClass(),
            'statusLabel' => $this->getStatusLabel(),
            'showRegistrationButton' => $this->showRegistrationButton,
            'showDetails' => $this->showDetails,
            'registrationUrl' => $this->getRegistrationUrl()
        ];
    }
    
    private function getStatusClass() {
        if ($this->event->isFull()) {
            return 'event-full';
        } elseif ($this->event->isPast()) {
            return 'event-past';
        } elseif ($this->event->isOngoing()) {
            return 'event-ongoing';
        } else {
            return 'event-upcoming';
        }
    }
    
    private function getStatusLabel() {
        if ($this->event->isFull()) {
            return __('Sold Out', 'event-management');
        } elseif ($this->event->isPast()) {
            return __('Past Event', 'event-management');
        } elseif ($this->event->isOngoing()) {
            return __('Happening Now', 'event-management');
        } else {
            $days = $this->event->start_date->diff(new \DateTime())->days;
            if ($days == 0) {
                return __('Today', 'event-management');
            } elseif ($days == 1) {
                return __('Tomorrow', 'event-management');
            } else {
                return sprintf(__('In %d days', 'event-management'), $days);
            }
        }
    }
    
    private function getRegistrationUrl() {
        return add_query_arg([
            'action' => 'register',
            'event_id' => $this->event->id
        ], get_permalink(get_option('event_registration_page')));
    }
}

Registration Form Component


// src/Components/RegistrationForm/RegistrationForm.php
namespace EventManagement\Components;

class RegistrationForm extends Form {
    private $event;
    
    public function __construct($event) {
        $this->event = $event;
        
        parent::__construct(admin_url('admin-ajax.php'), [
            'ajax' => true
        ]);
        
        $this->buildForm();
    }
    
    private function buildForm() {
        $this->addHiddenField('action', 'event_registration');
        $this->addHiddenField('event_id', $this->event->id);
        $this->addHiddenField('_wpnonce', wp_create_nonce('event_registration'));
        
        $this->addTextField('name', __('Full Name', 'event-management'), [
            'required' => true,
            'validation' => 'required|min:3'
        ]);
        
        $this->addEmailField('email', __('Email Address', 'event-management'), [
            'required' => true,
            'validation' => 'required|email'
        ]);
        
        $this->addTextField('phone', __('Phone Number', 'event-management'), [
            'validation' => 'phone'
        ]);
        
        if ($this->event->hasCustomFields()) {
            foreach ($this->event->getCustomFields() as $field) {
                $this->addField($field['name'], $field['type'], $field['options']);
            }
        }
        
        $this->addCheckboxField('terms', __('I agree to the terms and conditions', 'event-management'), [
            'required' => true,
            'validation' => 'required'
        ]);
    }
}

// tests/Unit/EventServiceTest.php
namespace EventManagement\Tests\Unit;

use PHPUnit\Framework\TestCase;
use EventManagement\Services\EventService;
use EventManagement\Models\Event;

class EventServiceTest extends TestCase {
    private $eventService;
    private $eventRepository;
    
    protected function setUp(): void {
        $this->eventRepository = $this->createMock(EventRepository::class);
        $this->eventService = new EventService($this->eventRepository);
    }
    
    public function testCreateEvent() {
        $eventData = [
            'title' => 'Test Event',
            'description' => 'Test Description',
            'start_date' => '2025-12-01 10:00:00',
            'end_date' => '2025-12-01 18:00:00',
            'capacity' => 100
        ];
        
        $this->eventRepository
            ->expects($this->once())
            ->method('create')
            ->with($eventData)
            ->willReturn(new Event($eventData));
        
        $event = $this->eventService->createEvent($eventData);
        
        $this->assertInstanceOf(Event::class, $event);
        $this->assertEquals('Test Event', $event->title);
    }
    
    public function testCannotCreatePastEvent() {
        $this->expectException(\InvalidArgumentException::class);
        
        $eventData = [
            'title' => 'Past Event',
            'start_date' => '2020-01-01 10:00:00',
            'end_date' => '2020-01-01 18:00:00'
        ];
        
        $this->eventService->createEvent($eventData);
    }
}
