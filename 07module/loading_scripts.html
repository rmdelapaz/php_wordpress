Basic Script Loading

Enqueue Scripts Properly

// Basic script enqueueing
function enqueue_ajax_scripts() {
    wp_enqueue_script(
        'my-ajax-script',
        plugin_dir_url(__FILE__) . 'js/ajax.js',
        array('jquery'),
        '1.0.0',
        true
    );
    
    // Localize script
    wp_localize_script('my-ajax-script', 'ajax_object', array(
        'ajax_url' => admin_url('admin-ajax.php'),
        'nonce' => wp_create_nonce('ajax_nonce')
    ));
}
add_action('wp_enqueue_scripts', 'enqueue_ajax_scripts');

Script Dependencies

Managing Dependencies

// Register scripts with dependencies
wp_register_script('ajax-core', $url1, array('jquery'));
wp_register_script('ajax-forms', $url2, array('ajax-core'));
wp_register_script('ajax-search', $url3, array('ajax-core'));

// Enqueue with dependencies resolved
wp_enqueue_script('ajax-forms');

Conditional Loading

Load Scripts When Needed

function conditional_script_loading() {
    // Only on specific pages
    if (is_page('contact')) {
        wp_enqueue_script('ajax-contact-form');
    }
    
    // Only when shortcode exists
    global $post;
    if (has_shortcode($post->post_content, 'ajax_form')) {
        wp_enqueue_script('ajax-forms');
    }
}
add_action('wp_enqueue_scripts', 'conditional_script_loading');

Best Practices

  • Always use wp_enqueue_script()
  • Register scripts before enqueueing
  • Use wp_localize_script() for PHP data
  • Load scripts conditionally
  • Version scripts properly
  • Load in footer when possible
  • Minify scripts in production
