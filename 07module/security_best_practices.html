WordPress Security Best Practices

Protect your WordPress sites from vulnerabilities and attacks

Learning Objectives

By the end of this lesson, you will be able to:

  • Protect against common vulnerabilities

Comprehensive Security Hardening

WordPress Security Manager


<?php
/**
 * WordPress Security Manager
 * Comprehensive security hardening system
 */
class WP_Security_Manager {
    
    private static $instance = null;
    private $security_log_table;
    private $lockout_table;
    
    public static function get_instance() {
        if (null === self::$instance) {
            self::$instance = new self();
        }
        return self::$instance;
    }
    
    private function __construct() {
        global $wpdb;
        $this->security_log_table = $wpdb->prefix . 'security_logs';
        $this->lockout_table = $wpdb->prefix . 'login_lockouts';
        
        // Initialize security features
        add_action('init', [$this, 'init_security']);
        
        // File system protection
        add_action('init', [$this, 'protect_filesystem']);
        
        // Login security
        add_action('wp_login_failed', [$this, 'log_failed_login']);
        add_filter('authenticate', [$this, 'check_lockout'], 30, 3);
        add_action('wp_login', [$this, 'log_successful_login'], 10, 2);
        
        // Hide WordPress version
        remove_action('wp_head', 'wp_generator');
        add_filter('the_generator', '__return_empty_string');
        
        // Disable XML-RPC
        add_filter('xmlrpc_enabled', '__return_false');
        add_action('init', [$this, 'block_xmlrpc_requests']);
        
        // Security headers
        add_action('send_headers', [$this, 'add_security_headers']);
        
        // File upload validation
        add_filter('wp_handle_upload_prefilter', [$this, 'validate_file_upload']);
        
        // REST API security
        add_filter('rest_authentication_errors', [$this, 'restrict_rest_api']);
        
        // Admin security
        add_action('admin_init', [$this, 'admin_security']);
        
        // Database security
        add_action('init', [$this, 'database_security']);
        
        // Scheduled security scans
        add_action('wp', [$this, 'schedule_security_scan']);
        add_action('security_scan_event', [$this, 'run_security_scan']);
    }
    
    /**
     * Initialize security features
     */
    public function init_security() {
        // Create security tables
        $this->create_security_tables();
        
        // Disable file editing
        if (!defined('DISALLOW_FILE_EDIT')) {
            define('DISALLOW_FILE_EDIT', true);
        }
        
        // Force SSL for admin
        if (!defined('FORCE_SSL_ADMIN')) {
            define('FORCE_SSL_ADMIN', true);
        }
        
        // Limit login attempts
        $this->cleanup_old_lockouts();
        
        // Change default login URL
        $this->custom_login_url();
        
        // Disable directory browsing
        $this->disable_directory_browsing();
    }
    
    /**
     * Create security database tables
     */
    private function create_security_tables() {
        global $wpdb;
        $charset_collate = $wpdb->get_charset_collate();
        
        // Security logs table
        $sql1 = "CREATE TABLE IF NOT EXISTS {$this->security_log_table} (
            id bigint(20) NOT NULL AUTO_INCREMENT,
            event_type varchar(50) NOT NULL,
            event_details text,
            ip_address varchar(45),
            user_agent text,
            user_id bigint(20),
            timestamp datetime DEFAULT CURRENT_TIMESTAMP,
            PRIMARY KEY (id),
            KEY event_type (event_type),
            KEY timestamp (timestamp),
            KEY ip_address (ip_address)
        ) $charset_collate;";
        
        // Login lockouts table
        $sql2 = "CREATE TABLE IF NOT EXISTS {$this->lockout_table} (
            id bigint(20) NOT NULL AUTO_INCREMENT,
            ip_address varchar(45) NOT NULL,
            username varchar(255),
            attempts int(11) DEFAULT 0,
            lockout_until datetime,
            created_at datetime DEFAULT CURRENT_TIMESTAMP,
            PRIMARY KEY (id),
            UNIQUE KEY ip_address (ip_address),
            KEY lockout_until (lockout_until)
        ) $charset_collate;";
        
        require_once(ABSPATH . 'wp-admin/includes/upgrade.php');
        dbDelta($sql1);
        dbDelta($sql2);
    }
    
    /**
     * Protect filesystem
     */
    public function protect_filesystem() {
        // Protect wp-config.php
        $this->protect_file(ABSPATH . 'wp-config.php');
        
        // Protect .htaccess
        $this->protect_file(ABSPATH . '.htaccess');
        
        // Protect wp-content/uploads
        $this->protect_uploads_directory();
        
        // Check file permissions
        $this->check_file_permissions();
    }
    
    /**
     * Protect individual file
     */
    private function protect_file($file) {
        if (file_exists($file)) {
            // Set restrictive permissions
            @chmod($file, 0444);
            
            // Add to .htaccess protection
            $htaccess = ABSPATH . '.htaccess';
            $filename = basename($file);
            $rules = "\n# Protect $filename\n";
            $rules .= "\n";
            $rules .= "Order allow,deny\n";
            $rules .= "Deny from all\n";
            $rules .= "\n";
            
            // Check if rule already exists
            if (file_exists($htaccess)) {
                $content = file_get_contents($htaccess);
                if (strpos($content, "Protect $filename") === false) {
                    file_put_contents($htaccess, $rules, FILE_APPEND);
                }
            }
        }
    }
    
    /**
     * Protect uploads directory
     */
    private function protect_uploads_directory() {
        $upload_dir = wp_upload_dir();
        $htaccess_file = $upload_dir['basedir'] . '/.htaccess';
        
        $rules = "# Protect uploads directory\n";
        $rules .= "\n";
        $rules .= "deny from all\n";
        $rules .= "\n";
        $rules .= "\n";
        $rules .= "deny from all\n";
        $rules .= "\n";
        $rules .= "\n";
        $rules .= "deny from all\n";
        $rules .= "\n";
        
        if (!file_exists($htaccess_file) || strpos(file_get_contents($htaccess_file), 'Protect uploads') === false) {
            file_put_contents($htaccess_file, $rules, FILE_APPEND);
        }
    }
    
    /**
     * Check file permissions
     */
    private function check_file_permissions() {
        $critical_files = [
            ABSPATH . 'wp-config.php' => 0444,
            ABSPATH . '.htaccess' => 0444,
            ABSPATH . 'index.php' => 0444
        ];
        
        $critical_dirs = [
            ABSPATH => 0755,
            WP_CONTENT_DIR => 0755,
            WP_CONTENT_DIR . '/themes' => 0755,
            WP_CONTENT_DIR . '/plugins' => 0755,
            WP_CONTENT_DIR . '/uploads' => 0755
        ];
        
        foreach ($critical_files as $file => $recommended) {
            if (file_exists($file)) {
                $current = substr(sprintf('%o', fileperms($file)), -4);
                if ($current != $recommended) {
                    $this->log_security_event('file_permission_issue', [
                        'file' => $file,
                        'current' => $current,
                        'recommended' => $recommended
                    ]);
                }
            }
        }
        
        foreach ($critical_dirs as $dir => $recommended) {
            if (is_dir($dir)) {
                $current = substr(sprintf('%o', fileperms($dir)), -4);
                if ($current != $recommended) {
                    $this->log_security_event('directory_permission_issue', [
                        'directory' => $dir,
                        'current' => $current,
                        'recommended' => $recommended
                    ]);
                }
            }
        }
    }
    
    /**
     * Add security headers
     */
    public function add_security_headers() {
        // Content Security Policy
        header("Content-Security-Policy: default-src 'self' https: data: 'unsafe-inline' 'unsafe-eval'");
        
        // X-Frame-Options
        header('X-Frame-Options: SAMEORIGIN');
        
        // X-Content-Type-Options
        header('X-Content-Type-Options: nosniff');
        
        // X-XSS-Protection
        header('X-XSS-Protection: 1; mode=block');
        
        // Referrer Policy
        header('Referrer-Policy: strict-origin-when-cross-origin');
        
        // Feature Policy
        header("Feature-Policy: camera 'none'; microphone 'none'; geolocation 'self'");
        
        // Strict Transport Security
        if (is_ssl()) {
            header('Strict-Transport-Security: max-age=31536000; includeSubDomains; preload');
        }
    }
    
    /**
     * Log failed login attempt
     */
    public function log_failed_login($username) {
        global $wpdb;
        
        $ip = $this->get_client_ip();
        $user_agent = $_SERVER['HTTP_USER_AGENT'] ?? '';
        
        // Log the attempt
        $this->log_security_event('failed_login', [
            'username' => $username,
            'ip' => $ip,
            'user_agent' => $user_agent
        ]);
        
        // Update lockout table
        $existing = $wpdb->get_row($wpdb->prepare(
            "SELECT * FROM {$this->lockout_table} WHERE ip_address = %s",
            $ip
        ));
        
        if ($existing) {
            $attempts = $existing->attempts + 1;
            $lockout_until = null;
            
            // Lock out after 5 attempts
            if ($attempts >= 5) {
                $lockout_until = date('Y-m-d H:i:s', strtotime('+30 minutes'));
                
                // Send alert for lockout
                $this->send_security_alert('account_locked', [
                    'ip' => $ip,
                    'username' => $username,
                    'attempts' => $attempts
                ]);
            }
            
            $wpdb->update(
                $this->lockout_table,
                [
                    'attempts' => $attempts,
                    'lockout_until' => $lockout_until,
                    'username' => $username
                ],
                ['ip_address' => $ip]
            );
        } else {
            $wpdb->insert(
                $this->lockout_table,
                [
                    'ip_address' => $ip,
                    'username' => $username,
                    'attempts' => 1
                ]
            );
        }
    }
    
    /**
     * Check if IP is locked out
     */
    public function check_lockout($user, $username, $password) {
        global $wpdb;
        
        if (empty($username) || empty($password)) {
            return $user;
        }
        
        $ip = $this->get_client_ip();
        
        $lockout = $wpdb->get_row($wpdb->prepare(
            "SELECT * FROM {$this->lockout_table} 
             WHERE ip_address = %s AND lockout_until > NOW()",
            $ip
        ));
        
        if ($lockout) {
            return new WP_Error('locked_out', 
                sprintf(__('Too many failed login attempts. Please try again in %d minutes.'), 
                    ceil((strtotime($lockout->lockout_until) - time()) / 60)
                )
            );
        }
        
        return $user;
    }
    
    /**
     * Log successful login
     */
    public function log_successful_login($user_login, $user) {
        global $wpdb;
        
        $ip = $this->get_client_ip();
        
        // Clear lockout for this IP
        $wpdb->delete($this->lockout_table, ['ip_address' => $ip]);
        
        // Log successful login
        $this->log_security_event('successful_login', [
            'username' => $user_login,
            'user_id' => $user->ID,
            'ip' => $ip
        ]);
        
        // Check for suspicious login
        $this->check_suspicious_login($user);
    }
    
    /**
     * Check for suspicious login activity
     */
    private function check_suspicious_login($user) {
        global $wpdb;
        
        $ip = $this->get_client_ip();
        $user_agent = $_SERVER['HTTP_USER_AGENT'] ?? '';
        
        // Get last successful login
        $last_login = $wpdb->get_row($wpdb->prepare(
            "SELECT * FROM {$this->security_log_table} 
             WHERE event_type = 'successful_login' 
             AND user_id = %d 
             AND id != LAST_INSERT_ID()
             ORDER BY timestamp DESC 
             LIMIT 1",
            $user->ID
        ));
        
        if ($last_login) {
            $last_ip = json_decode($last_login->event_details)->ip ?? '';
            
            // Check if login is from different location
            if ($last_ip && $last_ip !== $ip) {
                // Check if IPs are from different countries
                if ($this->is_different_country($last_ip, $ip)) {
                    $this->send_security_alert('suspicious_login', [
                        'user' => $user->user_login,
                        'new_ip' => $ip,
                        'last_ip' => $last_ip
                    ]);
                }
            }
        }
    }
    
    /**
     * Validate file uploads
     */
    public function validate_file_upload($file) {
        // Check file extension
        $allowed_extensions = [
            'jpg', 'jpeg', 'png', 'gif', 'pdf', 'doc', 'docx', 
            'xls', 'xlsx', 'ppt', 'pptx', 'zip', 'txt', 'csv'
        ];
        
        $extension = strtolower(pathinfo($file['name'], PATHINFO_EXTENSION));
        
        if (!in_array($extension, $allowed_extensions)) {
            $file['error'] = __('File type not allowed for security reasons.');
            return $file;
        }
        
        // Check MIME type
        $finfo = finfo_open(FILEINFO_MIME_TYPE);
        $mime_type = finfo_file($finfo, $file['tmp_name']);
        finfo_close($finfo);
        
        $allowed_mimes = get_allowed_mime_types();
        
        if (!in_array($mime_type, $allowed_mimes)) {
            $file['error'] = __('File MIME type not allowed.');
            return $file;
        }
        
        // Check for PHP code in file
        $content = file_get_contents($file['tmp_name']);
        if (preg_match('/<\?php|<\?=/i', $content)) {
            $file['error'] = __('File contains PHP code and cannot be uploaded.');
            
            // Log security event
            $this->log_security_event('malicious_upload_attempt', [
                'filename' => $file['name'],
                'user_id' => get_current_user_id()
            ]);
            
            return $file;
        }
        
        // Scan for malware signatures
        if ($this->scan_for_malware($file['tmp_name'])) {
            $file['error'] = __('File appears to contain malicious code.');
            
            $this->log_security_event('malware_upload_blocked', [
                'filename' => $file['name'],
                'user_id' => get_current_user_id()
            ]);
            
            return $file;
        }
        
        return $file;
    }
    
    /**
     * Scan file for malware
     */
    private function scan_for_malware($file) {
        $signatures = [
            'eval\s*\(',
            'base64_decode\s*\(',
            'gzinflate\s*\(',
            'str_rot13\s*\(',
            'system\s*\(',
            'exec\s*\(',
            'shell_exec\s*\(',
            'passthru\s*\(',
            'assert\s*\(',
            '\$_GET\[.*\]\s*\(',
            '\$_POST\[.*\]\s*\(',
            '\$_REQUEST\[.*\]\s*\(',
            'file_get_contents\s*\(\s*[\'"]https?://',
            'fopen\s*\(\s*[\'"]https?://',
            'curl_exec\s*\(',
            'preg_replace\s*\(.*\/e[\'"]'
        ];
        
        $content = file_get_contents($file);
        
        foreach ($signatures as $signature) {
            if (preg_match('/' . $signature . '/i', $content)) {
                return true;
            }
        }
        
        return false;
    }
    
    /**
     * Run security scan
     */
    public function run_security_scan() {
        $issues = [];
        
        // Check core file integrity
        $core_issues = $this->check_core_integrity();
        if (!empty($core_issues)) {
            $issues['core_integrity'] = $core_issues;
        }
        
        // Check for vulnerable plugins
        $plugin_issues = $this->check_vulnerable_plugins();
        if (!empty($plugin_issues)) {
            $issues['vulnerable_plugins'] = $plugin_issues;
        }
        
        // Check for suspicious files
        $file_issues = $this->scan_for_suspicious_files();
        if (!empty($file_issues)) {
            $issues['suspicious_files'] = $file_issues;
        }
        
        // Check database for suspicious content
        $db_issues = $this->scan_database();
        if (!empty($db_issues)) {
            $issues['database'] = $db_issues;
        }
        
        // Check user accounts
        $user_issues = $this->audit_user_accounts();
        if (!empty($user_issues)) {
            $issues['users'] = $user_issues;
        }
        
        // Save scan results
        if (!empty($issues)) {
            $this->log_security_event('security_scan_issues', $issues);
            $this->send_security_alert('security_scan_issues', $issues);
        } else {
            $this->log_security_event('security_scan_clean', ['message' => 'No issues found']);
        }
        
        return $issues;
    }
    
    /**
     * Check core file integrity
     */
    private function check_core_integrity() {
        $issues = [];
        
        // Get core checksums
        $version = get_bloginfo('version');
        $locale = get_locale();
        
        $response = wp_remote_get(
            "https://api.wordpress.org/core/checksums/1.0/?version={$version}&locale={$locale}"
        );
        
        if (!is_wp_error($response)) {
            $body = wp_remote_retrieve_body($response);
            $checksums = json_decode($body, true);
            
            if (isset($checksums['checksums']) && is_array($checksums['checksums'])) {
                foreach ($checksums['checksums'] as $file => $checksum) {
                    $file_path = ABSPATH . $file;
                    
                    if (file_exists($file_path)) {
                        $actual_checksum = md5_file($file_path);
                        
                        if ($actual_checksum !== $checksum) {
                            $issues[] = [
                                'file' => $file,
                                'expected' => $checksum,
                                'actual' => $actual_checksum
                            ];
                        }
                    }
                }
            }
        }
        
        return $issues;
    }
    
    /**
     * Log security event
     */
    private function log_security_event($event_type, $details = []) {
        global $wpdb;
        
        $wpdb->insert(
            $this->security_log_table,
            [
                'event_type' => $event_type,
                'event_details' => json_encode($details),
                'ip_address' => $this->get_client_ip(),
                'user_agent' => $_SERVER['HTTP_USER_AGENT'] ?? '',
                'user_id' => get_current_user_id()
            ]
        );
    }
    
    /**
     * Send security alert
     */
    private function send_security_alert($alert_type, $details) {
        $admin_email = get_option('admin_email');
        $site_name = get_bloginfo('name');
        
        $subject = "[Security Alert] {$alert_type} on {$site_name}";
        $message = "Security alert detected:\n\n";
        $message .= "Type: {$alert_type}\n";
        $message .= "Time: " . current_time('mysql') . "\n";
        $message .= "Details: " . print_r($details, true) . "\n";
        
        wp_mail($admin_email, $subject, $message);
    }
    
    /**
     * Get client IP address
     */
    private function get_client_ip() {
        $ip_keys = ['HTTP_CF_CONNECTING_IP', 'HTTP_X_FORWARDED_FOR', 'REMOTE_ADDR'];
        
        foreach ($ip_keys as $key) {
            if (isset($_SERVER[$key]) && filter_var($_SERVER[$key], FILTER_VALIDATE_IP)) {
                return $_SERVER[$key];
            }
        }
        
        return '0.0.0.0';
    }
}

Web Application Firewall (WAF)

WordPress WAF Implementation


<?php
/**
 * WordPress Web Application Firewall
 */
class WP_WAF {
    
    private $rules = [];
    private $whitelist = [];
    private $blacklist = [];
    
    public function __construct() {
        // Load early in WordPress
        add_action('muplugins_loaded', [$this, 'init_waf'], 1);
        
        // Load rules
        $this->load_rules();
        
        // Check request
        $this->check_request();
    }
    
    /**
     * Initialize WAF
     */
    public function init_waf() {
        // Check if WAF is enabled
        if (!get_option('waf_enabled', true)) {
            return;
        }
        
        // Run WAF checks
        $this->check_ip_blacklist();
        $this->check_rate_limiting();
        $this->check_request_method();
        $this->check_user_agent();
        $this->check_sql_injection();
        $this->check_xss_attacks();
        $this->check_file_inclusion();
        $this->check_command_injection();
        $this->check_directory_traversal();
        $this->check_request_size();
    }
    
    /**
     * Load WAF rules
     */
    private function load_rules() {
        // SQL Injection patterns
        $this->rules['sql_injection'] = [
            '/\bunion\b.*\bselect\b/i',
            '/\bselect\b.*\bfrom\b.*\bwhere\b/i',
            '/\binsert\b.*\binto\b.*\bvalues\b/i',
            '/\bdelete\b.*\bfrom\b/i',
            '/\bdrop\b.*\btable\b/i',
            '/\bupdate\b.*\bset\b/i',
            '/\bexec\b.*\(/i',
            '/\bscript\b.*\>/i',
            '/\-\-$/m',
            '/\bor\b.*\=.*\bor\b/i',
            '/\bwaitfor\b.*\bdelay\b/i',
            '/\bbenchmark\b\s*\(/i',
            '/\bsleep\b\s*\(/i'
        ];
        
        // XSS patterns
        $this->rules['xss'] = [
            '/]*>.*?<\/script>/is',
            '/on\w+\s*=\s*["\'].*?["\']/i',
            '/javascript\s*:/i',
            '/]*>/i',
            '/]*>/i',
            '/]*>/i',
            '/document\.\w+/i',
            '/window\.\w+/i',
            '/alert\s*\(/i',
            '/prompt\s*\(/i',
            '/confirm\s*\(/i'
        ];
        
        // File inclusion patterns
        $this->rules['file_inclusion'] = [
            '/\.\.[\/\\\]/',
            '/\/etc\/passwd/',
            '/\/etc\/shadow/',
            '/proc\/self\/environ/',
            '/\bfile_get_contents\b/i',
            '/\binclude\b/i',
            '/\brequire\b/i',
            '/\binclude_once\b/i',
            '/\brequire_once\b/i',
            '/\bfopen\b/i',
            '/\bfread\b/i',
            '/\bfile\b/i'
        ];
        
        // Command injection patterns
        $this->rules['command_injection'] = [
            '/;\s*\w+/i',
            '/\|\s*\w+/i',
            '/`[^`]*`/',
            '/\$\([^)]*\)/',
            '/\bsystem\b\s*\(/i',
            '/\bexec\b\s*\(/i',
            '/\bshell_exec\b\s*\(/i',
            '/\bpassthru\b\s*\(/i',
            '/\beval\b\s*\(/i',
            '/\bpopen\b\s*\(/i'
        ];
        
        // Load IP lists
        $this->whitelist = get_option('waf_ip_whitelist', []);
        $this->blacklist = get_option('waf_ip_blacklist', []);
    }
    
    /**
     * Check IP blacklist
     */
    private function check_ip_blacklist() {
        $client_ip = $this->get_client_ip();
        
        // Check whitelist first
        if (in_array($client_ip, $this->whitelist)) {
            return;
        }
        
        // Check blacklist
        if (in_array($client_ip, $this->blacklist)) {
            $this->block_request('IP Blacklisted', ['ip' => $client_ip]);
        }
        
        // Check for known bad IPs (could integrate with external services)
        if ($this->is_known_bad_ip($client_ip)) {
            $this->block_request('Known Malicious IP', ['ip' => $client_ip]);
        }
    }
    
    /**
     * Check rate limiting
     */
    private function check_rate_limiting() {
        $client_ip = $this->get_client_ip();
        $cache_key = 'waf_rate_limit_' . md5($client_ip);
        
        $requests = get_transient($cache_key);
        
        if ($requests === false) {
            set_transient($cache_key, 1, 60); // 1 minute window
        } else {
            $requests++;
            set_transient($cache_key, $requests, 60);
            
            // Block if more than 60 requests per minute
            if ($requests > 60) {
                $this->block_request('Rate Limit Exceeded', [
                    'ip' => $client_ip,
                    'requests' => $requests
                ]);
            }
        }
    }
    
    /**
     * Check request method
     */
    private function check_request_method() {
        $allowed_methods = ['GET', 'POST', 'HEAD'];
        $method = $_SERVER['REQUEST_METHOD'] ?? '';
        
        if (!in_array($method, $allowed_methods)) {
            $this->block_request('Invalid Request Method', ['method' => $method]);
        }
        
        // Check for method override attempts
        if (isset($_POST['_method']) || isset($_GET['_method'])) {
            $this->block_request('Method Override Attempt');
        }
    }
    
    /**
     * Check user agent
     */
    private function check_user_agent() {
        $user_agent = $_SERVER['HTTP_USER_AGENT'] ?? '';
        
        // Block empty user agents (optional)
        if (empty($user_agent) && get_option('waf_block_empty_ua', false)) {
            $this->block_request('Empty User Agent');
        }
        
        // Block known bad user agents
        $bad_agents = [
            'sqlmap',
            'nikto',
            'nmap',
            'masscan',
            'metasploit',
            'arachni',
            'acunetix',
            'nessus',
            'openvas',
            'havij',
            'wget',
            'curl' // Be careful with this
        ];
        
        foreach ($bad_agents as $agent) {
            if (stripos($user_agent, $agent) !== false) {
                $this->block_request('Malicious User Agent', ['agent' => $user_agent]);
            }
        }
    }
    
    /**
     * Check for SQL injection
     */
    private function check_sql_injection() {
        $request_data = array_merge($_GET, $_POST, $_COOKIE);
        
        foreach ($request_data as $key => $value) {
            if (is_array($value)) {
                $value = json_encode($value);
            }
            
            foreach ($this->rules['sql_injection'] as $pattern) {
                if (preg_match($pattern, $value) || preg_match($pattern, $key)) {
                    $this->block_request('SQL Injection Attempt', [
                        'pattern' => $pattern,
                        'value' => substr($value, 0, 100)
                    ]);
                }
            }
        }
        
        // Check URI
        $uri = $_SERVER['REQUEST_URI'] ?? '';
        foreach ($this->rules['sql_injection'] as $pattern) {
            if (preg_match($pattern, urldecode($uri))) {
                $this->block_request('SQL Injection in URI', ['uri' => $uri]);
            }
        }
    }
    
    /**
     * Check for XSS attacks
     */
    private function check_xss_attacks() {
        $request_data = array_merge($_GET, $_POST);
        
        foreach ($request_data as $key => $value) {
            if (is_array($value)) {
                $value = json_encode($value);
            }
            
            // Decode value
            $decoded = html_entity_decode(urldecode($value));
            
            foreach ($this->rules['xss'] as $pattern) {
                if (preg_match($pattern, $decoded) || preg_match($pattern, $key)) {
                    $this->block_request('XSS Attack Attempt', [
                        'pattern' => $pattern,
                        'value' => substr($value, 0, 100)
                    ]);
                }
            }
        }
    }
    
    /**
     * Check for file inclusion attempts
     */
    private function check_file_inclusion() {
        $request_data = array_merge($_GET, $_POST, $_COOKIE);
        $uri = $_SERVER['REQUEST_URI'] ?? '';
        
        foreach ($request_data as $key => $value) {
            if (is_array($value)) {
                continue;
            }
            
            foreach ($this->rules['file_inclusion'] as $pattern) {
                if (preg_match($pattern, $value) || preg_match($pattern, urldecode($uri))) {
                    $this->block_request('File Inclusion Attempt', [
                        'pattern' => $pattern,
                        'value' => substr($value, 0, 100)
                    ]);
                }
            }
        }
    }
    
    /**
     * Check for command injection
     */
    private function check_command_injection() {
        $request_data = array_merge($_GET, $_POST);
        
        foreach ($request_data as $key => $value) {
            if (is_array($value)) {
                continue;
            }
            
            foreach ($this->rules['command_injection'] as $pattern) {
                if (preg_match($pattern, $value)) {
                    $this->block_request('Command Injection Attempt', [
                        'pattern' => $pattern,
                        'value' => substr($value, 0, 100)
                    ]);
                }
            }
        }
    }
    
    /**
     * Check for directory traversal
     */
    private function check_directory_traversal() {
        $paths = array_merge($_GET, $_POST);
        
        foreach ($paths as $key => $value) {
            if (is_array($value)) {
                continue;
            }
            
            // Check for directory traversal patterns
            if (preg_match('/\.\.[\/\\\]/', $value) || 
                preg_match('/%2e%2e[%2f%5c]/i', $value) ||
                preg_match('/\.\.\%/', $value)) {
                
                $this->block_request('Directory Traversal Attempt', [
                    'value' => substr($value, 0, 100)
                ]);
            }
        }
    }
    
    /**
     * Check request size
     */
    private function check_request_size() {
        // Check POST size
        if ($_SERVER['REQUEST_METHOD'] === 'POST') {
            $max_size = 2 * 1024 * 1024; // 2MB default
            $content_length = $_SERVER['CONTENT_LENGTH'] ?? 0;
            
            if ($content_length > $max_size) {
                $this->block_request('Request Size Exceeded', [
                    'size' => $content_length,
                    'max' => $max_size
                ]);
            }
        }
        
        // Check number of parameters
        $total_params = count($_GET) + count($_POST);
        if ($total_params > 100) {
            $this->block_request('Too Many Parameters', ['count' => $total_params]);
        }
    }
    
    /**
     * Block request
     */
    private function block_request($reason, $details = []) {
        // Log the block
        $this->log_blocked_request($reason, $details);
        
        // Send 403 Forbidden
        header('HTTP/1.1 403 Forbidden');
        
        // Display error page
        ?>

        prefix . 'waf_logs';
        
        $wpdb->insert($table, [
            'reason' => $reason,
            'details' => json_encode($details),
            'ip_address' => $this->get_client_ip(),
            'user_agent' => $_SERVER['HTTP_USER_AGENT'] ?? '',
            'request_uri' => $_SERVER['REQUEST_URI'] ?? '',
            'request_method' => $_SERVER['REQUEST_METHOD'] ?? '',
            'timestamp' => current_time('mysql')
        ]);
        
        // Send alert for critical blocks
        if (get_option('waf_send_alerts', true)) {
            $this->send_security_alert($reason, $details);
        }
    }
    
    /**
     * Get client IP
     */
    private function get_client_ip() {
        $ip_keys = ['HTTP_CF_CONNECTING_IP', 'HTTP_X_FORWARDED_FOR', 'REMOTE_ADDR'];
        
        foreach ($ip_keys as $key) {
            if (isset($_SERVER[$key])) {
                $ip = explode(',', $_SERVER[$key])[0];
                if (filter_var(trim($ip), FILTER_VALIDATE_IP)) {
                    return trim($ip);
                }
            }
        }
        
        return '0.0.0.0';
    }
    
    /**
     * Check if IP is known bad
     */
    private function is_known_bad_ip($ip) {
        // This could integrate with services like:
        // - Project Honey Pot
        // - Stop Forum Spam
        // - AbuseIPDB
        // - Spamhaus
        
        // Example check (would need actual API integration)
        $bad_ip_ranges = [
            '10.0.0.0/8',     // Private
            '172.16.0.0/12',  // Private
            '192.168.0.0/16', // Private
            // Add known bad ranges
        ];
        
        foreach ($bad_ip_ranges as $range) {
            if ($this->ip_in_range($ip, $range)) {
                return true;
            }
        }
        
        return false;
    }
    
    /**
     * Check if IP is in range
     */
    private function ip_in_range($ip, $range) {
        if (strpos($range, '/') === false) {
            return $ip === $range;
        }
        
        list($subnet, $bits) = explode('/', $range);
        $ip = ip2long($ip);
        $subnet = ip2long($subnet);
        $mask = -1 << (32 - $bits);
        $subnet &= $mask;
        
        return ($ip & $mask) == $subnet;
    }
