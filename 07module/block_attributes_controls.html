Advanced Attribute Patterns


/**
 * Advanced attribute patterns and helpers

import { useState, useEffect } from '@wordpress/element';

// Custom hook for attribute management
function useAttributes(attributes, setAttributes) {
    // Batch updates
    const updateAttributes = (updates) => {
        setAttributes(updates);
    };
    
    // Single attribute updater
    const updateAttribute = (key, value) => {
        setAttributes({ [key]: value });
    };
    
    // Nested object updater
    const updateNestedAttribute = (path, value) => {
        const keys = path.split('.');
        const newAttributes = { ...attributes };
        let current = newAttributes;
        
        for (let i = 0; i < keys.length - 1; i++) {
            if (!current[keys[i]]) {
                current[keys[i]] = {};
            }
            current = current[keys[i]];
        }
        
        current[keys[keys.length - 1]] = value;
        setAttributes(newAttributes);
    };
    
    // Array attribute helpers
    const arrayHelpers = {
        add: (arrayKey, item) => {
            const array = attributes[arrayKey] || [];
            setAttributes({ [arrayKey]: [...array, item] });
        },
        
        remove: (arrayKey, index) => {
            const array = attributes[arrayKey] || [];
            setAttributes({
                [arrayKey]: array.filter((_, i) => i !== index)
            });
        },
        
        update: (arrayKey, index, item) => {
            const array = attributes[arrayKey] || [];
            const newArray = [...array];
            newArray[index] = item;
            setAttributes({ [arrayKey]: newArray });
        },
        
        move: (arrayKey, fromIndex, toIndex) => {
            const array = attributes[arrayKey] || [];
            const newArray = [...array];
            const [removed] = newArray.splice(fromIndex, 1);
            newArray.splice(toIndex, 0, removed);
            setAttributes({ [arrayKey]: newArray });
        }
    };
    
    return {
        updateAttributes,
        updateAttribute,
        updateNestedAttribute,
        arrayHelpers
    };
}

// Attribute validation
function validateAttributes(attributes, schema) {
    const errors = {};
    
    Object.keys(schema).forEach(key => {
        const rule = schema[key];
        const value = attributes[key];
        
        // Required check
        if (rule.required && !value) {
            errors[key] = 'This field is required';
        }
        
        // Type check
        if (value && rule.type && typeof value !== rule.type) {
            errors[key] = `Expected ${rule.type}, got ${typeof value}`;
        }
        
        // Min/Max for numbers
        if (rule.type === 'number' && value) {
            if (rule.min !== undefined && value < rule.min) {
                errors[key] = `Minimum value is ${rule.min}`;
            }
            if (rule.max !== undefined && value > rule.max) {
                errors[key] = `Maximum value is ${rule.max}`;
            }
        }
        
        // Pattern for strings
        if (rule.type === 'string' && rule.pattern && value) {
            const regex = new RegExp(rule.pattern);
            if (!regex.test(value)) {
                errors[key] = 'Invalid format';
            }
        }
        
        // Custom validation
        if (rule.validate && typeof rule.validate === 'function') {
            const error = rule.validate(value, attributes);
            if (error) {
                errors[key] = error;
            }
        }
    });
    
    return errors;
}

// Helper function to build category tree
function buildCategoryTree(categories) {
    if (!categories) return [];
    
    const tree = [];
    const map = {};
    
    // Create map
    categories.forEach(cat => {
        map[cat.id] = {
            id: cat.id.toString(),
            label: cat.name,
            children: []
        };
    });
    
    // Build tree
    categories.forEach(cat => {
        if (cat.parent === 0) {
            tree.push(map[cat.id]);
        } else if (map[cat.parent]) {
            map[cat.parent].children.push(map[cat.id]);
        }
    });
    
    return tree;
}
