Building Related Taxonomies


// Create relationships between taxonomies
class Taxonomy_Relationships {
    
    public function __construct() {
        add_action('init', [$this, 'register_related_taxonomies']);
        add_action('save_post', [$this, 'save_taxonomy_relationships']);
        add_filter('get_terms_args', [$this, 'filter_terms_by_relationship'], 10, 2);
    }
    
    public function register_related_taxonomies() {
        // Brand taxonomy
        register_taxonomy('brand', ['product'], [
            'labels' => [
                'name' => 'Brands',
                'singular_name' => 'Brand',
            ],
            'hierarchical' => false,
            'show_ui' => true,
            'show_admin_column' => true,
            'rewrite' => ['slug' => 'brand'],
        ]);
        
        // Model taxonomy (dependent on brand)
        register_taxonomy('model', ['product'], [
            'labels' => [
                'name' => 'Models',
                'singular_name' => 'Model',
            ],
            'hierarchical' => false,
            'show_ui' => true,
            'show_admin_column' => true,
            'rewrite' => ['slug' => 'model'],
        ]);
        
        // Year taxonomy
        register_taxonomy('year', ['product'], [
            'labels' => [
                'name' => 'Years',
                'singular_name' => 'Year',
            ],
            'hierarchical' => false,
            'show_ui' => true,
            'show_admin_column' => true,
            'rewrite' => ['slug' => 'year'],
        ]);
    }
    
    public function save_taxonomy_relationships($post_id) {
        if (get_post_type($post_id) !== 'product') {
            return;
        }
        
        // Get selected terms
        $brands = wp_get_post_terms($post_id, 'brand', ['fields' => 'ids']);
        $models = wp_get_post_terms($post_id, 'model', ['fields' => 'ids']);
        
        // Store brand-model relationships
        foreach ($models as $model_id) {
            foreach ($brands as $brand_id) {
                $this->create_term_relationship($brand_id, $model_id, 'brand_model');
            }
        }
    }
    
    private function create_term_relationship($parent_term_id, $child_term_id, $relationship_type) {
        global $wpdb;
        
        $table_name = $wpdb->prefix . 'term_relationships_custom';
        
        // Create table if not exists
        $this->create_relationships_table();
        
        // Check if relationship exists
        $exists = $wpdb->get_var($wpdb->prepare(
            "SELECT COUNT(*) FROM $table_name 
            WHERE parent_term_id = %d 
            AND child_term_id = %d 
            AND relationship_type = %s",
            $parent_term_id,
            $child_term_id,
            $relationship_type
        ));
        
        if (!$exists) {
            $wpdb->insert(
                $table_name,
                [
                    'parent_term_id' => $parent_term_id,
                    'child_term_id' => $child_term_id,
                    'relationship_type' => $relationship_type,
                    'created_at' => current_time('mysql')
                ],
                ['%d', '%d', '%s', '%s']
            );
        }
    }
    
    private function create_relationships_table() {
        global $wpdb;
        
        $table_name = $wpdb->prefix . 'term_relationships_custom';
        $charset_collate = $wpdb->get_charset_collate();
        
        $sql = "CREATE TABLE IF NOT EXISTS $table_name (
            id bigint(20) UNSIGNED NOT NULL AUTO_INCREMENT,
            parent_term_id bigint(20) UNSIGNED NOT NULL,
            child_term_id bigint(20) UNSIGNED NOT NULL,
            relationship_type varchar(50) NOT NULL,
            created_at datetime DEFAULT CURRENT_TIMESTAMP,
            PRIMARY KEY (id),
            KEY parent_term_idx (parent_term_id),
            KEY child_term_idx (child_term_id),
            KEY relationship_type_idx (relationship_type)
        ) $charset_collate;";
        
        require_once(ABSPATH . 'wp-admin/includes/upgrade.php');
        dbDelta($sql);
    }
    
    public function get_related_terms($term_id, $relationship_type, $direction = 'children') {
        global $wpdb;
        
        $table_name = $wpdb->prefix . 'term_relationships_custom';
        
        if ($direction === 'children') {
            $results = $wpdb->get_col($wpdb->prepare(
                "SELECT child_term_id FROM $table_name 
                WHERE parent_term_id = %d 
                AND relationship_type = %s",
                $term_id,
                $relationship_type
            ));
        } else {
            $results = $wpdb->get_col($wpdb->prepare(
                "SELECT parent_term_id FROM $table_name 
                WHERE child_term_id = %d 
                AND relationship_type = %s",
                $term_id,
                $relationship_type
            ));
        }
        
        return $results;
    }
    
    public function filter_terms_by_relationship($args, $taxonomies) {
        // Filter models based on selected brand
        if (in_array('model', $taxonomies) && isset($_GET['brand_filter'])) {
            $brand_id = intval($_GET['brand_filter']);
            $related_models = $this->get_related_terms($brand_id, 'brand_model', 'children');
            
            if (!empty($related_models)) {
                $args['include'] = $related_models;
            }
        }
        
        return $args;
    }
}

// Complex taxonomy queries
class Complex_Tax_Queries {
    
    // Query posts with multiple taxonomy relationships
    public static function query_with_relationships($args = []) {
        $defaults = [
            'post_type' => 'product',
            'posts_per_page' => 12,
            'tax_query' => [
                'relation' => 'AND',
            ],
        ];
        
        $args = wp_parse_args($args, $defaults);
        
        // Example: Get products in specific category AND with specific features
        if (!empty($args['category']) && !empty($args['features'])) {
            $args['tax_query'][] = [
                'taxonomy' => 'product_category',
                'field' => 'slug',
                'terms' => $args['category'],
            ];
            
            $args['tax_query'][] = [
                'taxonomy' => 'product_features',
                'field' => 'slug',
                'terms' => $args['features'],
                'operator' => 'IN',
            ];
        }
        
        return new WP_Query($args);
    }
    
    // Query with nested tax queries
    public static function complex_nested_query() {
        $args = [
            'post_type' => 'product',
            'posts_per_page' => 20,
            'tax_query' => [
                'relation' => 'OR',
                [
                    'relation' => 'AND',
                    [
                        'taxonomy' => 'product_category',
                        'field' => 'slug',
                        'terms' => 'electronics',
                    ],
                    [
                        'taxonomy' => 'brand',
                        'field' => 'slug',
                        'terms' => ['apple', 'samsung'],
                        'operator' => 'IN',
                    ],
                ],
                [
                    'relation' => 'AND',
                    [
                        'taxonomy' => 'product_category',
                        'field' => 'slug',
                        'terms' => 'accessories',
                    ],
                    [
                        'taxonomy' => 'product_features',
                        'field' => 'slug',
                        'terms' => 'wireless',
                    ],
                ],
            ],
        ];
        
        return new WP_Query($args);
    }
    
    // Query excluding certain terms
    public static function exclude_terms_query() {
        $args = [
            'post_type' => 'product',
            'posts_per_page' => 10,
            'tax_query' => [
                [
                    'taxonomy' => 'product_category',
                    'field' => 'term_id',
                    'terms' => [5, 10, 15], // Exclude these category IDs
                    'operator' => 'NOT IN',
                ],
            ],
        ];
        
        return new WP_Query($args);
    }
    
    // Query posts with term count
    public static function query_by_term_count() {
        global $wpdb;
        
        // Get posts that have at least 3 different features
        $sql = "
            SELECT p.ID, p.post_title, COUNT(DISTINCT tr.term_taxonomy_id) as term_count
            FROM {$wpdb->posts} p
            INNER JOIN {$wpdb->term_relationships} tr ON p.ID = tr.object_id
            INNER JOIN {$wpdb->term_taxonomy} tt ON tr.term_taxonomy_id = tt.term_taxonomy_id
            WHERE p.post_type = 'product'
            AND p.post_status = 'publish'
            AND tt.taxonomy = 'product_features'
            GROUP BY p.ID
            HAVING term_count >= 3
            ORDER BY term_count DESC
        ";
        
        return $wpdb->get_results($sql);
    }
    
    // Get posts in hierarchical category tree
    public static function get_category_tree_posts($parent_category_id) {
        // Get all child categories
        $child_categories = get_term_children($parent_category_id, 'product_category');
        
        // Include parent category
        $all_categories = array_merge([$parent_category_id], $child_categories);
        
        $args = [
            'post_type' => 'product',
            'posts_per_page' => -1,
            'tax_query' => [
                [
                    'taxonomy' => 'product_category',
                    'field' => 'term_id',
                    'terms' => $all_categories,
                    'include_children' => false, // We've already included children
                ],
            ],
        ];
        
        return new WP_Query($args);
    }
    
    // Cross-taxonomy relationship query
    public static function cross_taxonomy_query() {
        global $wpdb;
        
        // Get products that share terms across different taxonomies
        $sql = "
            SELECT DISTINCT p.ID, p.post_title
            FROM {$wpdb->posts} p
            INNER JOIN {$wpdb->term_relationships} tr1 ON p.ID = tr1.object_id
            INNER JOIN {$wpdb->term_taxonomy} tt1 ON tr1.term_taxonomy_id = tt1.term_taxonomy_id
            INNER JOIN {$wpdb->terms} t1 ON tt1.term_id = t1.term_id
            INNER JOIN {$wpdb->term_relationships} tr2 ON p.ID = tr2.object_id
            INNER JOIN {$wpdb->term_taxonomy} tt2 ON tr2.term_taxonomy_id = tt2.term_taxonomy_id
            INNER JOIN {$wpdb->terms} t2 ON tt2.term_id = t2.term_id
            WHERE p.post_type = 'product'
            AND p.post_status = 'publish'
            AND tt1.taxonomy = 'brand'
            AND tt2.taxonomy = 'model'
            AND t1.slug = %s
            AND t2.slug IN (
                SELECT t3.slug
                FROM {$wpdb->terms} t3
                INNER JOIN {$wpdb->prefix}term_relationships_custom trc 
                ON t3.term_id = trc.child_term_id
                WHERE trc.parent_term_id = t1.term_id
                AND trc.relationship_type = 'brand_model'
            )
        ";
        
        return $wpdb->get_results($wpdb->prepare($sql, 'apple'));
    }
}

Custom Taxonomy Templates

Dynamic Template Selection


// Custom taxonomy template system
class Taxonomy_Templates {
    
    public function __construct() {
        add_filter('template_include', [$this, 'custom_taxonomy_template']);
        add_action('pre_get_posts', [$this, 'modify_taxonomy_query']);
    }
    
    public function custom_taxonomy_template($template) {
        if (!is_tax()) {
            return $template;
        }
        
        $term = get_queried_object();
        $taxonomy = $term->taxonomy;
        
        // Check for term-specific template
        $term_template = get_term_meta($term->term_id, 'category_template', true);
        
        if ($term_template) {
            $custom_template = locate_template([
                "taxonomy-{$taxonomy}-{$term->slug}-{$term_template}.php",
                "taxonomy-{$taxonomy}-{$term_template}.php",
                "taxonomy-{$term_template}.php",
            ]);
            
            if ($custom_template) {
                return $custom_template;
            }
        }
        
        // Check for parent category template (for hierarchical taxonomies)
        if (is_taxonomy_hierarchical($taxonomy) && $term->parent) {
            $parent_template = get_term_meta($term->parent, 'category_template', true);
            
            if ($parent_template) {
                $parent_custom = locate_template([
                    "taxonomy-{$taxonomy}-parent-{$parent_template}.php",
                    "taxonomy-parent-{$parent_template}.php",
                ]);
                
                if ($parent_custom) {
                    return $parent_custom;
                }
            }
        }
        
        return $template;
    }
    
    public function modify_taxonomy_query($query) {
        if (!is_admin() && $query->is_main_query() && is_tax()) {
            $term = get_queried_object();
            
            // Custom posts per page for specific terms
            $posts_per_page = get_term_meta($term->term_id, 'posts_per_page', true);
            if ($posts_per_page) {
                $query->set('posts_per_page', intval($posts_per_page));
            }
            
            // Custom ordering
            $order_by = get_term_meta($term->term_id, 'order_by', true);
            if ($order_by) {
                $query->set('orderby', $order_by);
            }
            
            // Include child term posts for hierarchical taxonomies
            if (get_term_meta($term->term_id, 'include_children', true)) {
                $child_terms = get_term_children($term->term_id, $term->taxonomy);
                $all_terms = array_merge([$term->term_id], $child_terms);
                
                $query->set('tax_query', [
                    [
                        'taxonomy' => $term->taxonomy,
                        'field' => 'term_id',
                        'terms' => $all_terms,
                        'include_children' => false,
                    ],
                ]);
            }
        }
    }
}

Taxonomy Performance Optimization

Caching and Optimization


// Optimize taxonomy queries
class Taxonomy_Performance {
    
    public function __construct() {
        add_action('init', [$this, 'setup_caching']);
        add_filter('get_terms', [$this, 'cache_term_queries'], 10, 4);
        add_action('edited_term', [$this, 'clear_term_cache'], 10, 3);
    }
    
    public function setup_caching() {
        // Cache expensive term queries
        if (false === ($featured_categories = get_transient('featured_product_categories'))) {
            $featured_categories = get_terms([
                'taxonomy' => 'product_category',
                'meta_query' => [
                    [
                        'key' => 'featured_category',
                        'value' => '1',
                        'compare' => '=',
                    ],
                ],
                'hide_empty' => false,
            ]);
            
            set_transient('featured_product_categories', $featured_categories, HOUR_IN_SECONDS);
        }
    }
    
    public function cache_term_queries($terms, $taxonomies, $args, $term_query) {
        // Cache complex term queries
        $cache_key = 'terms_' . md5(serialize([$taxonomies, $args]));
        
        if (false === ($cached_terms = wp_cache_get($cache_key, 'term_queries'))) {
            wp_cache_set($cache_key, $terms, 'term_queries', 3600);
        }
        
        return $terms;
    }
    
    public function clear_term_cache($term_id, $tt_id, $taxonomy) {
        // Clear specific caches when terms are updated
        delete_transient('featured_product_categories');
        wp_cache_delete('term_queries', 'terms');
        
        // Clear term relationship caches
        clean_term_cache($term_id, $taxonomy);
    }
    
    // Preload term meta for performance
    public static function preload_term_meta($term_ids) {
        if (empty($term_ids)) {
            return;
        }
        
        update_termmeta_cache($term_ids);
    }
    
    // Optimized term count update
    public static function update_term_counts($taxonomy) {
        global $wpdb;
        
        $terms = $wpdb->get_results($wpdb->prepare("
            SELECT tt.term_taxonomy_id, tt.term_id, tt.taxonomy, COUNT(tr.object_id) as count
            FROM {$wpdb->term_taxonomy} tt
            LEFT JOIN {$wpdb->term_relationships} tr ON tt.term_taxonomy_id = tr.term_taxonomy_id
            LEFT JOIN {$wpdb->posts} p ON tr.object_id = p.ID
            WHERE tt.taxonomy = %s
            AND (p.post_status = 'publish' OR p.ID IS NULL)
            GROUP BY tt.term_taxonomy_id
        ", $taxonomy));
        
        foreach ($terms as $term) {
            $wpdb->update(
                $wpdb->term_taxonomy,
                ['count' => $term->count],
                ['term_taxonomy_id' => $term->term_taxonomy_id]
            );
        }
        
        clean_term_cache([], $taxonomy);
    }
}
