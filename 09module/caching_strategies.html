WordPress Transients API

Using Transients

// Set transient
set_transient('my_data', $data, 12 * HOUR_IN_SECONDS);

// Get transient
$data = get_transient('my_data');
if (false === $data) {
    // Cache miss - regenerate data
    $data = expensive_operation();
    set_transient('my_data', $data, 12 * HOUR_IN_SECONDS);
}

// Delete transient
delete_transient('my_data');

// Example: Cache API response
function get_api_data() {
    $cache_key = 'api_response_' . md5(serialize(func_get_args()));
    
    if (false === ($data = get_transient($cache_key))) {
        $response = wp_remote_get('https://api.example.com/data');
        
        if (!is_wp_error($response)) {
            $data = wp_remote_retrieve_body($response);
            set_transient($cache_key, $data, HOUR_IN_SECONDS);
        }
    }
    
    return $data;
}

Object Caching

WordPress Object Cache

// Object cache (memory cache for current request)
wp_cache_set('my_key', $data, 'my_group', 3600);
$data = wp_cache_get('my_key', 'my_group');
wp_cache_delete('my_key', 'my_group');

// Cache implementation
class Cache_Manager {
    private $cache_group = 'my_plugin';
    
    public function get($key) {
        // Try object cache first
        $data = wp_cache_get($key, $this->cache_group);
        
        if (false === $data) {
            // Try transient
            $data = get_transient($this->cache_group . '_' . $key);
            
            if (false !== $data) {
                // Store in object cache for this request
                wp_cache_set($key, $data, $this->cache_group);
            }
        }
        
        return $data;
    }
    
    public function set($key, $data, $expiration = 3600) {
        // Set both caches
        wp_cache_set($key, $data, $this->cache_group, $expiration);
        set_transient($this->cache_group . '_' . $key, $data, $expiration);
    }
    
    public function delete($key) {
        wp_cache_delete($key, $this->cache_group);
        delete_transient($this->cache_group . '_' . $key);
    }
}

Query Result Caching

Cache Database Queries

function get_popular_posts($limit = 10) {
    $cache_key = 'popular_posts_' . $limit;
    
    // Try cache first
    $posts = get_transient($cache_key);
    
    if (false === $posts) {
        global $wpdb;
        
        $posts = $wpdb->get_results($wpdb->prepare("
            SELECT p.*, COUNT(pm.meta_id) as view_count
            FROM {$wpdb->posts} p
            LEFT JOIN {$wpdb->postmeta} pm ON p.ID = pm.post_id
            WHERE pm.meta_key = 'post_views'
            AND p.post_status = 'publish'
            GROUP BY p.ID
            ORDER BY view_count DESC
            LIMIT %d
        ", $limit));
        
        // Cache for 6 hours
        set_transient($cache_key, $posts, 6 * HOUR_IN_SECONDS);
    }
    
    return $posts;
}

// Clear cache when posts are updated
add_action('save_post', function($post_id) {
    // Clear all popular posts caches
    delete_transient('popular_posts_5');
    delete_transient('popular_posts_10');
});

Fragment Caching

Cache HTML Fragments

function get_cached_widget($widget_id) {
    $cache_key = 'widget_html_' . $widget_id;
    $html = get_transient($cache_key);
    
    if (false === $html) {
        ob_start();
        // Generate widget HTML
        render_widget($widget_id);
        $html = ob_get_clean();
        
        // Cache for 1 day
        set_transient($cache_key, $html, DAY_IN_SECONDS);
    }
    
    return $html;
}

Cache Invalidation

Smart Cache Clearing

class Smart_Cache {
    
    public function invalidate_post_cache($post_id) {
        // Clear specific post cache
        delete_transient('post_data_' . $post_id);
        
        // Clear related caches
        $categories = wp_get_post_categories($post_id);
        foreach ($categories as $cat_id) {
            delete_transient('category_posts_' . $cat_id);
        }
        
        // Clear archive caches
        delete_transient('recent_posts');
        delete_transient('popular_posts');
    }
    
    public function setup_hooks() {
        // Clear cache on post save
        add_action('save_post', array($this, 'invalidate_post_cache'));
        
        // Clear cache on comment
        add_action('comment_post', function($comment_id) {
            $comment = get_comment($comment_id);
            $this->invalidate_post_cache($comment->comment_post_ID);
        });
    }
}

Best Practices

  • Use appropriate expiration times
  • Implement cache warming strategies
  • Clear cache on relevant events
  • Use unique cache keys
  • Monitor cache hit rates
  • Consider external caching solutions
  • Don't cache user-specific data globally
