Menu Position System

WordPress uses a numeric position system to determine where menu items appear. Understanding these positions helps you place your plugin's menu appropriately.

Position Default Menu Item Description
2 Dashboard Main dashboard menu
4 Separator LINE First separator line
5 Posts Post management
10 Media Media library
15 Links Link manager (if enabled)
20 Pages Page management
25 Comments Comment moderation
59 Separator LINE Second separator
60 Appearance Themes and customization
65 Plugins Plugin management
70 Users User management
75 Tools WordPress tools
80 Settings Site settings
99 Separator LINE Third separator
Use decimal positions (like 26.5) to place your menu between existing items without conflicts.

Choosing Menu Position

<?php
// Safe positions for custom menus
add_action( 'admin_menu', 'register_custom_menu_position' );

function register_custom_menu_position() {
    // After Comments (25) but before separator (59)
    add_menu_page(
        'My Plugin',           // Page title
        'My Plugin',           // Menu title
        'manage_options',      // Capability
        'my-plugin',          // Menu slug
        'my_plugin_page',     // Callback
        'dashicons-chart-pie', // Icon
        30                    // Position after Comments
    );
    
    // Using decimal for precise placement
    add_menu_page(
        'Another Plugin',
        'Another Plugin',
        'manage_options',
        'another-plugin',
        'another_plugin_page',
        'dashicons-admin-generic',
        25.5  // Between Comments (25) and position 26
    );
    
    // Avoid conflicts by checking if position is taken
    global $menu;
    $position = 27;
    while ( isset( $menu[$position] ) ) {
        $position++;
    }
    
    add_menu_page(
        'Smart Position',
        'Smart Position',
        'manage_options',
        'smart-plugin',
        'smart_plugin_page',
        'dashicons-lightbulb',
        $position
    );
}

Menu Capabilities and Permissions

Understanding Capabilities

Each menu item requires a specific capability to be visible and accessible. This ensures only authorized users can access certain areas.

Common Menu Capabilities

<?php
// Capability requirements for different user roles

// Super Admin (Multisite)
'manage_network'     // Network administration

// Administrator
'manage_options'     // General admin access
'activate_plugins'   // Plugin activation
'edit_theme_options' // Theme customization
'edit_users'        // User management

// Editor
'edit_pages'        // Page editing
'edit_others_posts' // Edit all posts
'publish_pages'     // Publish pages
'moderate_comments' // Moderate comments

// Author
'edit_posts'        // Edit own posts
'publish_posts'     // Publish posts
'upload_files'      // Upload media

// Contributor
'edit_posts'        // Edit own posts (can't publish)
'read'             // Basic read access

// Subscriber
'read'             // Only read access

// Example: Different menus for different roles
add_action( 'admin_menu', 'role_based_menus' );

function role_based_menus() {
    // Admin only menu
    add_menu_page(
        'Admin Settings',
        'Admin Settings',
        'manage_options',  // Only administrators
        'admin-settings',
        'admin_settings_page'
    );
    
    // Editor and above
    add_menu_page(
        'Content Manager',
        'Content Manager',
        'edit_pages',      // Editors and administrators
        'content-manager',
        'content_manager_page'
    );
    
    // All logged-in users
    add_menu_page(
        'User Dashboard',
        'User Dashboard',
        'read',            // All users including subscribers
        'user-dashboard',
        'user_dashboard_page'
    );
}

Menu-Related Hooks

WordPress provides several hooks for customizing the admin menu:

  • admin_menu: Add menu items
  • admin_init: Initialize admin features
  • admin_bar_menu: Customize admin bar
  • network_admin_menu: Network admin menus
  • user_admin_menu: User admin menus

Menu Hook Examples

<?php
// Standard admin menu
add_action( 'admin_menu', 'standard_admin_menu' );

function standard_admin_menu() {
    add_menu_page( /* ... */ );
}

// Network admin menu (multisite)
add_action( 'network_admin_menu', 'network_admin_menu' );

function network_admin_menu() {
    add_menu_page(
        'Network Settings',
        'Network Settings',
        'manage_network',
        'network-settings',
        'network_settings_page'
    );
}

// Modify existing menu items
add_action( 'admin_menu', 'modify_menu_items', 999 );

function modify_menu_items() {
    // Remove menu items
    remove_menu_page( 'edit-comments.php' );
    
    // Remove submenu items
    remove_submenu_page( 'themes.php', 'theme-editor.php' );
    
    // Rename menu items
    global $menu, $submenu;
    
    // Rename Posts to Articles
    $menu[5][0] = 'Articles';
    $submenu['edit.php'][5][0] = 'All Articles';
    $submenu['edit.php'][10][0] = 'Add Article';
}

// Add to admin bar
add_action( 'admin_bar_menu', 'add_admin_bar_items', 100 );

function add_admin_bar_items( $admin_bar ) {
    $admin_bar->add_menu( array(
        'id'    => 'my-plugin-bar',
        'title' => 'My Plugin',
        'href'  => admin_url( 'admin.php?page=my-plugin' ),
        'meta'  => array(
            'title' => 'My Plugin Quick Access'
        )
    ) );
}

Menu Structure Best Practices

  • Group related features: Use submenus for related functionality
  • Choose appropriate positions: Place menus logically based on function
  • Use proper capabilities: Ensure correct user access levels
  • Consistent naming: Use clear, descriptive menu titles
  • Avoid menu pollution: Don't create too many top-level menus
  • Consider existing menus: Add to Settings or Tools when appropriate
  • Use Dashicons: Choose appropriate icons for visual consistency
  • Test with different roles: Verify menu visibility for each user role
Avoid hardcoding menu positions below 100 - these are reserved for future WordPress use.
For simple settings, consider adding a submenu to the existing Settings menu instead of creating a new top-level menu.

Planning Your Plugin's Menu Structure

Complete Menu Structure Example

<?php
/**
 * Well-structured plugin menu system
 */
class My_Plugin_Admin_Menu {
    
    public function __construct() {
        add_action( 'admin_menu', array( $this, 'register_menus' ) );
    }
    
    public function register_menus() {
        // Main menu
        add_menu_page(
            __( 'My Plugin Dashboard', 'my-plugin' ),
            __( 'My Plugin', 'my-plugin' ),
            'manage_options',
            'my-plugin',
            array( $this, 'dashboard_page' ),
            'dashicons-superhero',
            26
        );
        
        // Dashboard submenu (replaces default first item)
        add_submenu_page(
            'my-plugin',
            __( 'Dashboard', 'my-plugin' ),
            __( 'Dashboard', 'my-plugin' ),
            'manage_options',
            'my-plugin',
            array( $this, 'dashboard_page' )
        );
        
        // Settings submenu
        add_submenu_page(
            'my-plugin',
            __( 'Settings', 'my-plugin' ),
            __( 'Settings', 'my-plugin' ),
            'manage_options',
            'my-plugin-settings',
            array( $this, 'settings_page' )
        );
        
        // Reports submenu
        add_submenu_page(
            'my-plugin',
            __( 'Reports', 'my-plugin' ),
            __( 'Reports', 'my-plugin' ),
            'view_reports',  // Custom capability
            'my-plugin-reports',
            array( $this, 'reports_page' )
        );
        
        // Tools submenu
        add_submenu_page(
            'my-plugin',
            __( 'Tools', 'my-plugin' ),
            __( 'Tools', 'my-plugin' ),
            'manage_options',
            'my-plugin-tools',
            array( $this, 'tools_page' )
        );
        
        // Hidden page (not in menu but accessible)
        add_submenu_page(
            null,  // No parent = hidden
            __( 'Secret Page', 'my-plugin' ),
            __( 'Secret', 'my-plugin' ),
            'manage_options',
            'my-plugin-secret',
            array( $this, 'secret_page' )
        );
    }
    
    // Page callbacks
    public function dashboard_page() {
        echo '<div class="wrap">';
        echo '<h1>' . esc_html( get_admin_page_title() ) . '</h1>';
        echo '<p>Welcome to My Plugin Dashboard!</p>';
        echo '</div>';
    }
    
    public function settings_page() {
        // Settings page content
    }
    
    public function reports_page() {
        // Reports page content
    }
    
    public function tools_page() {
        // Tools page content
    }
    
    public function secret_page() {
        // Hidden page content
    }
}

// Initialize
new My_Plugin_Admin_Menu();
