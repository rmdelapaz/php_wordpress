Complete Attribute Example

Team Member Card Shortcode

function team_member_shortcode( $atts ) {
    // Parse attributes with comprehensive defaults
    $atts = shortcode_atts( array(
        'name' => '',
        'title' => '',
        'email' => '',
        'phone' => '',
        'image' => '',
        'bio' => '',
        'social_linkedin' => '',
        'social_twitter' => '',
        'layout' => 'card', // card, list, minimal
        'color_scheme' => 'default', // default, dark, light, custom
        'custom_color' => '',
        'show_contact' => 'yes',
        'show_social' => 'yes',
    ), $atts, 'team_member' );
    
    // Validate required fields
    if ( empty( $atts['name'] ) ) {
        return '
Team member name is required.
';
    }
    
    // Start building output
    $classes = 'team-member team-layout-' . esc_attr( $atts['layout'] );
    $classes .= ' team-scheme-' . esc_attr( $atts['color_scheme'] );
    
    $output = '
';
    
    // Add image if provided
    if ( ! empty( $atts['image'] ) ) {
        $output .= '
';
        $output .= '';
        $output .= '
';
    }
    
    $output .= '
';
    $output .= '
' . esc_html( $atts['name'] ) . '
';
    
    if ( ! empty( $atts['title'] ) ) {
        $output .= '
' . esc_html( $atts['title'] ) . '
';
    }
    
    if ( ! empty( $atts['bio'] ) ) {
        $output .= '
' . wp_kses_post( $atts['bio'] ) . '
';
    }
    
    // Contact information
    if ( $atts['show_contact'] === 'yes' && ( ! empty( $atts['email'] ) || ! empty( $atts['phone'] ) ) ) {
        $output .= '
';
        
        if ( ! empty( $atts['email'] ) ) {
            $output .= '';
            $output .= esc_html( $atts['email'] ) . '';
        }
        
        if ( ! empty( $atts['phone'] ) ) {
            $output .= '';
            $output .= esc_html( $atts['phone'] ) . '';
        }
        
        $output .= '
';
    }
    
    // Social links
    if ( $atts['show_social'] === 'yes' && ( ! empty( $atts['social_linkedin'] ) || ! empty( $atts['social_twitter'] ) ) ) {
        $output .= '
';
        
        if ( ! empty( $atts['social_linkedin'] ) ) {
            $output .= 'LinkedIn';
        }
        
        if ( ! empty( $atts['social_twitter'] ) ) {
            $output .= 'Twitter';
        }
        
        $output .= '
';
    }
    
    $output .= '
'; // .team-member-info
    $output .= '
'; // .team-member
    
    return $output;
}
add_shortcode( 'team_member', 'team_member_shortcode' );

// Usage examples:
// [team_member name="John Doe" title="CEO" email="john@example.com"]
// [team_member name="Jane Smith" title="CTO" layout="minimal" show_contact="no"]

Validation and Sanitization

‚ö†Ô∏è Always Validate and Sanitize Attributes

User input from shortcode attributes should never be trusted. Always validate and sanitize before using.

Validation Functions

function validated_shortcode( $atts ) {
    $atts = shortcode_atts( array(
        'id' => 0,
        'count' => 5,
        'email' => '',
        'url' => '',
        'color' => '#ffffff',
        'date' => '',
        'choice' => 'option1',
        'show' => 'yes',
    ), $atts );
    
    // Validate and sanitize each attribute type
    
    // Integer validation
    $id = absint( $atts['id'] );
    if ( $id <= 0 ) {
        return '
Invalid ID provided.
';
    }
    
    // Number range validation
    $count = intval( $atts['count'] );
    if ( $count < 1 || $count > 100 ) {
        $count = 5; // Reset to default if out of range
    }
    
    // Email validation
    if ( ! empty( $atts['email'] ) && ! is_email( $atts['email'] ) ) {
        return '
Invalid email address.
';
    }
    $email = sanitize_email( $atts['email'] );
    
    // URL validation
    $url = esc_url( $atts['url'] );
    if ( ! empty( $atts['url'] ) && ! filter_var( $url, FILTER_VALIDATE_URL ) ) {
        return '
Invalid URL provided.
';
    }
    
    // Color validation (hex)
    $color = sanitize_hex_color( $atts['color'] );
    if ( empty( $color ) ) {
        $color = '#ffffff'; // Default if invalid
    }
    
    // Date validation
    $date = sanitize_text_field( $atts['date'] );
    if ( ! empty( $date ) ) {
        $timestamp = strtotime( $date );
        if ( $timestamp === false ) {
            return '
Invalid date format.
';
        }
        $date = date( 'Y-m-d', $timestamp );
    }
    
    // Choice validation (whitelist)
    $valid_choices = array( 'option1', 'option2', 'option3' );
    $choice = sanitize_text_field( $atts['choice'] );
    if ( ! in_array( $choice, $valid_choices, true ) ) {
        $choice = 'option1'; // Default if invalid
    }
    
    // Boolean validation
    $show = ( $atts['show'] === 'yes' || $atts['show'] === 'true' || $atts['show'] === '1' );
    
    // Build output with validated data...
    $output = '
';
    $output .= '
All attributes validated!
';
    $output .= '
';
    
    return $output;
}
add_shortcode( 'validated', 'validated_shortcode' );

Best Practices

‚úÖ Attribute Best Practices

  • Always provide sensible defaults
  • Use consistent naming conventions (lowercase, underscores)
  • Validate all user input
  • Document all available attributes
  • Keep attribute names short but descriptive
  • Use the third parameter in shortcode_atts() for filters
  • Consider backwards compatibility when changing attributes

‚ö†Ô∏è Common Mistakes

  • Not setting default values
  • Trusting user input without validation
  • Using spaces in attribute names
  • Not escaping output
  • Making attributes case-sensitive
  • Not handling empty or missing attributes
