Database Optimization

Efficient Database Queries

// Bad: Multiple queries
foreach ($post_ids as $id) {
    $post = get_post($id);
    // Process post
}

// Good: Single query
$posts = get_posts(array(
    'post__in' => $post_ids,
    'posts_per_page' => -1
));

// Use caching for expensive queries
$results = wp_cache_get('expensive_query');
if (false === $results) {
    global $wpdb;
    $results = $wpdb->get_results($query);
    wp_cache_set('expensive_query', $results, '', 3600);
}

// Batch operations
global $wpdb;
$values = array();
foreach ($data as $row) {
    $values[] = $wpdb->prepare("(%d, %s)", $row['id'], $row['name']);
}
$query = "INSERT INTO {$wpdb->prefix}custom_table (id, name) VALUES " . implode(',', $values);

Asset Optimization

Optimize Scripts and Styles

// Load assets only when needed
add_action('wp_enqueue_scripts', function() {
    // Conditional loading
    if (is_page('contact')) {
        wp_enqueue_script('contact-form');
    }
    
    // Use minified versions
    $suffix = defined('SCRIPT_DEBUG') && SCRIPT_DEBUG ? '' : '.min';
    wp_enqueue_script(
        'my-script',
        plugin_dir_url(__FILE__) . "js/script{$suffix}.js",
        array('jquery'),
        '1.0.0',
        true // Load in footer
    );
});

Query Optimization

Optimize WP_Query

// Optimize queries
$args = array(
    'posts_per_page' => 10,
    'no_found_rows' => true, // Skip pagination count
    'fields' => 'ids', // Get only IDs if that's all you need
    'update_post_meta_cache' => false, // Skip meta cache
    'update_post_term_cache' => false, // Skip term cache
);

$query = new WP_Query($args);

Best Practices

  • Use WordPress built-in functions when possible
  • Cache expensive operations
  • Load assets only when needed
  • Minimize database queries
  • Use transients for temporary data
  • Profile and monitor performance
