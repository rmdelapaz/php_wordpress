/**
 * Handles the plugin activation process.
 *
 * Creates database tables, sets default options, and schedules
 * cron jobs necessary for the plugin to function properly.
 *
 * @since 1.0.0
 * @access public
 *
 * @global wpdb $wpdb WordPress database abstraction object.
 *
 * @param bool $network_wide Whether to enable the plugin for all sites
 *                          in the network or just the current site.
 * @return void
 */
public function activate( $network_wide = false ) {
    global $wpdb;
    
    if ( $network_wide && is_multisite() ) {
        // Activate for all sites in network
        $sites = get_sites();
        foreach ( $sites as $site ) {
            switch_to_blog( $site->blog_id );
            $this->single_activate();
            restore_current_blog();
        }
    } else {
        $this->single_activate();
    }
}

API Documentation

Documenting Hooks

## Available Hooks

### Actions

#### `my_plugin_init`
Fires after the plugin has been initialized.

**Since:** 1.0.0

**Example:**
```php
add_action( 'my_plugin_init', function() {
    // Your code here
});
```

#### `my_plugin_after_save`
Fires after data has been saved to the database.

**Since:** 1.0.0

**Parameters:**
- `$post_id` (int) - The post ID
- `$data` (array) - The saved data

### Filters

#### `my_plugin_settings`
Filters the plugin settings before they are saved.

**Since:** 1.0.0

**Parameters:**
- `$settings` (array) - The settings array

**Return:** (array) Modified settings

**Example:**
```php
add_filter( 'my_plugin_settings', function( $settings ) {
    $settings['custom_option'] = 'value';
    return $settings;
});
```

Code Examples

Providing Clear Examples

## Usage Examples

### Basic Usage

```php
// Initialize the plugin
$my_plugin = new My_Plugin();

// Add custom data
$my_plugin->add_item( array(
    'title' => 'Sample Item',
    'content' => 'Item content',
    'status' => 'published'
) );
```

### Advanced Usage

```php
// Hook into plugin events
add_action( 'my_plugin_before_save', function( $data ) {
    // Validate data before saving
    if ( empty( $data['title'] ) ) {
        return new WP_Error( 'missing_title', 'Title is required' );
    }
    return $data;
} );

// Extend functionality
class My_Plugin_Extension extends My_Plugin {
    public function custom_method() {
        // Custom implementation
    }
}
```
