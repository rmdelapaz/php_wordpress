Options Page Architecture

A well-designed options page makes your plugin easy to configure and professional. Let's explore how to create comprehensive settings interfaces.

Typical Options Page Structure

  • Page Header: Title, description, and navigation
  • Settings Form: Organized input fields and controls
  • Sidebar: Help text, quick links, or promotional content
  • Action Buttons: Save, reset, import/export controls
  • Feedback Messages: Success, error, and warning notices

Complete Options Page Implementation

Professional Options Page Class

<?php
/**
 * Plugin Options Page
 */
class My_Plugin_Options_Page {
    
    private $options;
    private $plugin_name = 'my-plugin';
    
    public function __construct() {
        add_action( 'admin_menu', array( $this, 'add_options_page' ) );
        add_action( 'admin_init', array( $this, 'init_settings' ) );
        add_action( 'admin_enqueue_scripts', array( $this, 'enqueue_admin_assets' ) );
        
        $this->options = get_option( $this->plugin_name . '_options' );
    }
    
    /**
     * Add options page to admin menu
     */
    public function add_options_page() {
        $page_hook = add_menu_page(
            __( 'My Plugin Settings', 'my-plugin' ),
            __( 'My Plugin', 'my-plugin' ),
            'manage_options',
            $this->plugin_name,
            array( $this, 'render_options_page' ),
            'dashicons-admin-generic',
            30
        );
        
        // Add help tab
        add_action( "load-{$page_hook}", array( $this, 'add_help_tab' ) );
    }
    
    /**
     * Render the options page
     */
    public function render_options_page() {
        // Get active tab
        $active_tab = isset( $_GET['tab'] ) ? $_GET['tab'] : 'general';
        ?>
        
        <div class="wrap">
            <!-- Page Header -->
            <h1>
                <span class="dashicons dashicons-admin-generic"></span>
                <?php echo esc_html( get_admin_page_title() ); ?>
            </h1>
            
            <!-- Plugin Description -->
            <div class="notice notice-info inline">
                <p>
                    <strong>Welcome to My Plugin!</strong> 
                    Configure your settings below to customize the plugin's behavior.
                </p>
            </div>
            
            <!-- Settings Updated Notice -->
            <?php settings_errors( 'my_plugin_messages' ); ?>
            
            <!-- Tab Navigation -->
            <h2 class="nav-tab-wrapper">
                <a href="?page=<?php echo $this->plugin_name; ?>&tab=general" 
                   class="nav-tab <?php echo $active_tab == 'general' ? 'nav-tab-active' : ''; ?>">
                    <span class="dashicons dashicons-admin-settings"></span> General
                </a>

                    <?php

                        case 'general':
                            $this->render_general_tab();
                            break;
                        case 'display':
                            $this->render_display_tab();
                            break;

        <?php

    /**
     * Add help tab
     */
    public function add_help_tab() {
        $screen = get_current_screen();
        
        // Overview tab
        $screen->add_help_tab( array(
            'id'      => 'overview',
            'title'   => __( 'Overview', 'my-plugin' ),
            'content' => '<p>' . __( 'This screen provides access to all plugin settings. Use the tabs to navigate between different sections.', 'my-plugin' ) . '</p>'
        ) );
        
        // General settings help
        $screen->add_help_tab( array(
            'id'      => 'general_help',
            'title'   => __( 'General Settings', 'my-plugin' ),
            'content' => '<p>' . __( 'The General tab contains the main configuration options for the plugin.', 'my-plugin' ) . '</p>'
        ) );
        
        // Set help sidebar
        $screen->set_help_sidebar(
            '<p><strong>' . __( 'For more information:', 'my-plugin' ) . '</strong></p>' .
            '<p><a href="https://example.com/docs" target="_blank">' . __( 'Documentation', 'my-plugin' ) . '</a></p>' .
            '<p><a href="https://example.com/support" target="_blank">' . __( 'Support Forum', 'my-plugin' ) . '</a></p>'
        );
    }
    
    /**
     * Enqueue admin assets
     */
    public function enqueue_admin_assets( $hook ) {
        if ( strpos( $hook, $this->plugin_name ) === false ) {
            return;
        }
        
        // Color picker
        wp_enqueue_style( 'wp-color-picker' );
        wp_enqueue_script( 'wp-color-picker' );
        
        // Media uploader
        wp_enqueue_media();
        
        // Custom admin script
        wp_enqueue_script( 
            'my-plugin-admin',
            plugin_dir_url( __FILE__ ) . 'assets/admin.js',
            array( 'jquery', 'wp-color-picker' ),
            '1.0.0',
            true
        );
        
        // Localize script
        wp_localize_script( 'my-plugin-admin', 'myPluginAdmin', array(
            'ajaxUrl' => admin_url( 'admin-ajax.php' ),
            'nonce' => wp_create_nonce( 'my_plugin_admin' ),
            'strings' => array(
                'confirmReset' => __( 'Are you sure you want to reset all settings?', 'my-plugin' ),
                'processing' => __( 'Processing...', 'my-plugin' ),
                'success' => __( 'Success!', 'my-plugin' ),
                'error' => __( 'An error occurred.', 'my-plugin' )
            )
        ) );
    }
}

jQuery(document).ready(function($) {
    // Initialize color picker
    $('.color-field').wpColorPicker();
    
    // Handle clear cache button
    $('#clear-cache').on('click', function(e) {
        e.preventDefault();
        
        var $button = $(this);
        $button.prop('disabled', true).text(myPluginAdmin.strings.processing);
        
        $.post(myPluginAdmin.ajaxUrl, {
            action: 'my_plugin_clear_cache',
            nonce: myPluginAdmin.nonce
        })
        .done(function(response) {
            if (response.success) {
                alert(myPluginAdmin.strings.success);
            } else {
                alert(response.data.message);
            }
        })
        .fail(function() {
            alert(myPluginAdmin.strings.error);
        })
        .always(function() {
            $button.prop('disabled', false).text('Clear Cache');
        });
    });
    
    // Handle reset settings
    $('#reset-settings').on('click', function(e) {
        e.preventDefault();
        
        if (!confirm(myPluginAdmin.strings.confirmReset)) {
            return;
        }
        
        // Perform reset via AJAX
        $.post(myPluginAdmin.ajaxUrl, {
            action: 'my_plugin_reset_settings',
            nonce: myPluginAdmin.nonce
        })
        .done(function(response) {
            if (response.success) {
                window.location.reload();
            }
        });
    });
    
    // Export settings
    $('#export-settings').on('click', function(e) {
        e.preventDefault();
        
        $.post(myPluginAdmin.ajaxUrl, {
            action: 'my_plugin_export_settings',
            nonce: myPluginAdmin.nonce
        })
        .done(function(response) {
            if (response.success) {
                // Create download link
                var blob = new Blob([JSON.stringify(response.data)], {type: 'application/json'});
                var url = URL.createObjectURL(blob);
                var a = document.createElement('a');
                a.href = url;
                a.download = 'my-plugin-settings.json';
                a.click();
                URL.revokeObjectURL(url);
            }
        });
    });
    
    // Media uploader
    var mediaUploader;
    
    $('.upload-button').on('click', function(e) {
        e.preventDefault();
        
        var $button = $(this);
        var $input = $button.siblings('input[type="hidden"]');
        var $preview = $button.siblings('#preview-image');
        
        // If uploader exists, open it
        if (mediaUploader) {
            mediaUploader.open();
            return;
        }
        
        // Create media uploader
        mediaUploader = wp.media({
            title: 'Choose Image',
            button: {
                text: 'Use this image'
            },
            multiple: false
        });
        
        // When image is selected
        mediaUploader.on('select', function() {
            var attachment = mediaUploader.state().get('selection').first().toJSON();
            $input.val(attachment.id);
            $preview.html('<img src="' + attachment.url + '" style="max-width: 200px;">');
        });
        
        mediaUploader.open();
    });
    
    // Remove image
    $('.remove-button').on('click', function(e) {
        e.preventDefault();
        
        var $button = $(this);
        var $input = $button.siblings('input[type="hidden"]');
        var $preview = $button.siblings('#preview-image');
        
        $input.val('');
        $preview.html('');
    });
    
    // Form validation
    $('form').on('submit', function() {
        var $form = $(this);
        var hasError = false;
        
        // Validate required fields
        $form.find('[required]').each(function() {
            if (!$(this).val()) {
                $(this).addClass('error');
                hasError = true;
            }
        });
        
        if (hasError) {
            alert('Please fill in all required fields.');
            return false;
        }
    });
});

Options Page Best Practices

  • Consistent UI: Use WordPress UI components for familiarity
  • Clear organization: Group related settings with tabs or sections
  • Helpful descriptions: Provide context for each setting
  • Visual feedback: Show success/error messages clearly
  • Progressive disclosure: Hide advanced options by default
  • Responsive design: Ensure settings work on mobile
  • Accessibility: Use proper labels and ARIA attributes
  • Import/Export: Allow users to backup settings
  • Reset option: Provide way to restore defaults
  • Contextual help: Add help tabs for complex features
Always validate and sanitize user input on the server side, never rely solely on JavaScript validation!
Use WordPress transients API to cache expensive operations in your options page for better performance.
