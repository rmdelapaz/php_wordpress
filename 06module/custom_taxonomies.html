Common Issues and Solutions

Issue: Taxonomy Not Showing in Admin

Possible Causes:

  • Registration happening too late (not on 'init' hook)
  • show_ui set to false
  • Post type not properly associated

Solution:

// Ensure proper registration
add_action( 'init', 'register_taxonomies', 0 ); // Priority 0
// Check show_ui is true
'show_ui' => true,
// Verify post type exists
register_taxonomy( 'my_tax', array( 'existing_post_type' ), $args );

Issue: 404 Errors on Taxonomy Archives

Solution: Flush rewrite rules

// On plugin activation
register_activation_hook( __FILE__, 'my_plugin_activate' );
function my_plugin_activate() {
    register_taxonomies(); // Register your taxonomies
    flush_rewrite_rules();
}

// Or manually: Settings > Permalinks > Save Changes

Issue: Terms Not Saving

Possible Causes:

  • Capability issues
  • JavaScript conflicts
  • Missing nonce verification

Solution: Check capabilities and conflicts

// Check user capabilities
if ( ! current_user_can( 'manage_categories' ) ) {
    return;
}

// Verify nonce in custom save functions
if ( ! wp_verify_nonce( $_POST['my_nonce'], 'save_taxonomy' ) ) {
    return;
}

Taxonomy Term Meta

Store additional data with taxonomy terms:

// Add term meta
add_term_meta( $term_id, 'custom_image', $image_url );
add_term_meta( $term_id, 'featured', 'yes' );
add_term_meta( $term_id, 'order', 5 );

// Get term meta
$image = get_term_meta( $term_id, 'custom_image', true );
$is_featured = get_term_meta( $term_id, 'featured', true );

// Update term meta
update_term_meta( $term_id, 'custom_image', $new_image_url );

// Delete term meta
delete_term_meta( $term_id, 'custom_image' );

// Query by term meta
$terms = get_terms( array(
    'taxonomy' => 'product_category',
    'meta_key' => 'featured',
    'meta_value' => 'yes',

AJAX Taxonomy Filtering

// JavaScript for AJAX filtering
jQuery(document).ready(function($) {
    $('.taxonomy-filter').on('change', function() {
        var taxonomy = $(this).data('taxonomy');
        var term = $(this).val();
        
        $.ajax({
            url: ajax_object.ajax_url,
            type: 'POST',
            data: {
                action: 'filter_posts',
                taxonomy: taxonomy,
                term: term,
                nonce: ajax_object.nonce
            },
            success: function(response) {
                $('.results-container').html(response);
            }
        });
    });
});
// PHP handler for AJAX
function handle_taxonomy_filter() {
    check_ajax_referer( 'filter_nonce', 'nonce' );
    
    $taxonomy = sanitize_text_field( $_POST['taxonomy'] );
    $term = sanitize_text_field( $_POST['term'] );
    
    $args = array(
        'post_type' => 'product',
        'tax_query' => array(
            array(
                'taxonomy' => $taxonomy,
                'field'    => 'slug',
                'terms'    => $term,
            ),
        ),
    );
    
    $query = new WP_Query( $args );
    
    if ( $query->have_posts() ) {
        while ( $query->have_posts() ) {
            $query->the_post();
            // Output HTML
            get_template_part( 'template-parts/content', 'product' );
        }
    } else {
        echo '
No products found.
';
    }
    
    wp_die();
}
add_action( 'wp_ajax_filter_posts', 'handle_taxonomy_filter' );
add_action( 'wp_ajax_nopriv_filter_posts', 'handle_taxonomy_filter' );
