Plugin Lifecycle Overview

WordPress provides specific hooks that fire when a plugin is activated, deactivated, or uninstalled. These hooks allow you to perform setup and cleanup operations.

Plugin Lifecycle

Install Activate Running Deactivate Uninstall

The Activation Hook

The activation hook runs when a plugin is activated through the WordPress admin interface.

Basic Activation Hook

<?php
// Register activation hook
register_activation_hook( __FILE__, 'my_plugin_activate' );

function my_plugin_activate() {
    // Activation code here
    
    // Add default options
    add_option( 'my_plugin_version', '1.0.0' );
    add_option( 'my_plugin_db_version', '1.0' );
    add_option( 'my_plugin_activated', time() );
    
    // Create database tables
    my_plugin_create_tables();
    
    // Schedule events
    if ( ! wp_next_scheduled( 'my_plugin_daily_event' ) ) {
        wp_schedule_event( time(), 'daily', 'my_plugin_daily_event' );
    }
    
    // Flush rewrite rules
    flush_rewrite_rules();
    
    // Set a transient to show admin notice
    set_transient( 'my_plugin_activation_notice', true, 5 );
}

// Show activation notice
add_action( 'admin_notices', 'my_plugin_activation_notice' );

function my_plugin_activation_notice() {
    if ( get_transient( 'my_plugin_activation_notice' ) ) {
        ?>
        <div class="notice notice-success is-dismissible">
            <p>Thank you for activating My Plugin!</p>
        </div>
        <?php
        delete_transient( 'my_plugin_activation_notice' );
    }
}

  • Create database tables

Creating Database Tables

Database Table Creation on Activation

<?php
function my_plugin_create_tables() {
    global $wpdb;
    
    $table_name = $wpdb->prefix . 'my_plugin_data';
    $charset_collate = $wpdb->get_charset_collate();
    
    $sql = "CREATE TABLE $table_name (
        id mediumint(9) NOT NULL AUTO_INCREMENT,
        time datetime DEFAULT '0000-00-00 00:00:00' NOT NULL,
        name tinytext NOT NULL,
        text text NOT NULL,
        url varchar(100) DEFAULT '' NOT NULL,
        PRIMARY KEY  (id),
        KEY time (time)
    ) $charset_collate;";
    
    require_once( ABSPATH . 'wp-admin/includes/upgrade.php' );
    dbDelta( $sql );
    
    // Store database version for future upgrades
    add_option( 'my_plugin_db_version', '1.0' );
}

// Check and upgrade database
function my_plugin_check_db_version() {
    $installed_version = get_option( 'my_plugin_db_version' );
    
    if ( $installed_version != MY_PLUGIN_DB_VERSION ) {
        my_plugin_create_tables();
        update_option( 'my_plugin_db_version', MY_PLUGIN_DB_VERSION );
    }
}
add_action( 'plugins_loaded', 'my_plugin_check_db_version' );

The Deactivation Hook

The deactivation hook runs when a plugin is deactivated. Use it for temporary cleanup, not permanent data deletion.

Basic Deactivation Hook

<?php
// Register deactivation hook
register_deactivation_hook( __FILE__, 'my_plugin_deactivate' );

function my_plugin_deactivate() {
    // Unschedule events
    $timestamp = wp_next_scheduled( 'my_plugin_daily_event' );
    if ( $timestamp ) {
        wp_unschedule_event( $timestamp, 'my_plugin_daily_event' );
    }
    
    // Clear all scheduled hooks for the plugin
    wp_clear_scheduled_hook( 'my_plugin_hourly_event' );
    
    // Flush cache
    wp_cache_flush();
    
    // Remove temporary data
    delete_transient( 'my_plugin_cache' );
    
    // Flush rewrite rules
    flush_rewrite_rules();
    
    // Log deactivation
    error_log( 'My Plugin was deactivated on ' . current_time( 'mysql' ) );
}
Don't delete user data or settings in the deactivation hook! Users expect their data to remain if they reactivate the plugin.

Uninstall Methods

WordPress provides two methods for handling plugin uninstallation:

Method File/Hook When to Use
uninstall.php Separate file Complex uninstall logic
Uninstall Hook register_uninstall_hook() Simple uninstall logic

Method 1: uninstall.php

<?php
// File: uninstall.php
// This file is called when the plugin is deleted

// If uninstall not called from WordPress, exit
if ( ! defined( 'WP_UNINSTALL_PLUGIN' ) ) {
    exit;
}

// Delete options
delete_option( 'my_plugin_version' );
delete_option( 'my_plugin_db_version' );
delete_option( 'my_plugin_settings' );

// Delete user meta
delete_metadata( 'user', 0, 'my_plugin_user_preference', '', true );

// Delete posts of custom post type
$posts = get_posts( array(
    'post_type' => 'my_custom_type',
    'numberposts' => -1,
    'post_status' => 'any'
) );

foreach ( $posts as $post ) {
    wp_delete_post( $post->ID, true );
}

// Drop custom database tables
global $wpdb;
$wpdb->query( "DROP TABLE IF EXISTS {$wpdb->prefix}my_plugin_data" );

// Delete custom files
$upload_dir = wp_upload_dir();
$plugin_dir = $upload_dir['basedir'] . '/my-plugin';
if ( is_dir( $plugin_dir ) ) {
    // Recursively delete directory
    my_plugin_delete_directory( $plugin_dir );
}

// Clear any cached data
wp_cache_flush();

Method 2: Uninstall Hook

<?php
// Register uninstall hook
register_uninstall_hook( __FILE__, 'my_plugin_uninstall' );

function my_plugin_uninstall() {
    // Only works for single site
    if ( ! is_multisite() ) {
        // Delete options
        delete_option( 'my_plugin_settings' );
        
        // Delete database table
        global $wpdb;
        $wpdb->query( "DROP TABLE IF EXISTS {$wpdb->prefix}my_plugin_data" );
    } else {
        // For multisite
        global $wpdb;
        $blog_ids = $wpdb->get_col( "SELECT blog_id FROM $wpdb->blogs" );
        $original_blog_id = get_current_blog_id();
        
        foreach ( $blog_ids as $blog_id ) {
            switch_to_blog( $blog_id );
            delete_option( 'my_plugin_settings' );
            $wpdb->query( "DROP TABLE IF EXISTS {$wpdb->prefix}my_plugin_data" );
        }
        
        switch_to_blog( $original_blog_id );
    }
}

Multisite Activation

Network Activation Handling

<?php
// Activation for multisite
function my_plugin_activate( $network_wide ) {
    if ( is_multisite() && $network_wide ) {
        // Network activation
        global $wpdb;
        
        // Get all blog ids
        $blog_ids = $wpdb->get_col( "SELECT blog_id FROM $wpdb->blogs" );
        
        foreach ( $blog_ids as $blog_id ) {
            switch_to_blog( $blog_id );
            my_plugin_activate_single_site();
            restore_current_blog();
        }
    } else {
        // Single site activation
        my_plugin_activate_single_site();
    }
}
register_activation_hook( __FILE__, 'my_plugin_activate' );

// Single site activation logic
function my_plugin_activate_single_site() {
    // Create tables
    my_plugin_create_tables();
    
    // Add options
    add_option( 'my_plugin_activated', true );
    
    // Set up roles
    my_plugin_setup_roles();
}

// Handle new site creation in multisite
add_action( 'wp_initialize_site', 'my_plugin_new_site', 10, 2 );

function my_plugin_new_site( $new_site, $args ) {
    if ( is_plugin_active_for_network( 'my-plugin/my-plugin.php' ) ) {
        switch_to_blog( $new_site->blog_id );
        my_plugin_activate_single_site();
        restore_current_blog();
    }
}

// Handle site deletion
add_action( 'wp_delete_site', 'my_plugin_delete_site' );

function my_plugin_delete_site( $old_site ) {
    global $wpdb;
    
    switch_to_blog( $old_site->blog_id );
    
    // Delete options
    delete_option( 'my_plugin_settings' );
    
    // Delete tables
    $wpdb->query( "DROP TABLE IF EXISTS {$wpdb->prefix}my_plugin_data" );
    
    restore_current_blog();
}

Error Handling in Activation

Activation with Error Checking

<?php
class My_Plugin_Activator {
    
    public static function activate() {
        // Check PHP version
        if ( version_compare( PHP_VERSION, '7.2', '<' ) ) {
            deactivate_plugins( plugin_basename( __FILE__ ) );
            wp_die( 
                'This plugin requires PHP version 7.2 or higher.',
                'Plugin Activation Error',
                array( 'back_link' => true )
            );
        }
        
        // Check WordPress version
        if ( version_compare( get_bloginfo( 'version' ), '5.0', '<' ) ) {
            deactivate_plugins( plugin_basename( __FILE__ ) );
            wp_die(
                'This plugin requires WordPress version 5.0 or higher.',
                'Plugin Activation Error',
                array( 'back_link' => true )
            );
        }
        
        // Check for required plugin
        if ( ! is_plugin_active( 'required-plugin/required-plugin.php' ) ) {
            deactivate_plugins( plugin_basename( __FILE__ ) );
            wp_die(
                'This plugin requires "Required Plugin" to be installed and activated.',
                'Plugin Activation Error',
                array( 'back_link' => true )
            );
        }
        
        // Check for required PHP extensions
        if ( ! extension_loaded( 'curl' ) ) {
            deactivate_plugins( plugin_basename( __FILE__ ) );
            wp_die(
                'This plugin requires the PHP CURL extension.',
                'Plugin Activation Error',
                array( 'back_link' => true )
            );
        }
        
        // If all checks pass, proceed with activation
        self::create_tables();
        self::set_default_options();
        self::create_roles();
        
        // Set flag for activation redirect
        add_option( 'my_plugin_do_activation_redirect', true );
    }
    
    private static function create_tables() {
        // Table creation logic
    }
    
    private static function set_default_options() {
        // Set default options
    }
    
    private static function create_roles() {
        // Create custom roles
    }
}

register_activation_hook( __FILE__, array( 'My_Plugin_Activator', 'activate' ) );

// Redirect after activation
add_action( 'admin_init', 'my_plugin_activation_redirect' );

function my_plugin_activation_redirect() {
    if ( get_option( 'my_plugin_do_activation_redirect', false ) ) {
        delete_option( 'my_plugin_do_activation_redirect' );
        
        // Don't redirect on bulk activation
        if ( isset( $_GET['activate-multi'] ) ) {
            return;
        }
        
        wp_safe_redirect( admin_url( 'admin.php?page=my-plugin-welcome' ) );
        exit;
    }
}

Do's and Don'ts

‚úÖ Do's

  • Check requirements before activation
  • Use dbDelta() for table creation
  • Store version numbers
  • Use transients for temporary data
  • Handle multisite properly
  • Clean up scheduled events
  • Provide user feedback
  • Log important actions

‚ùå Don'ts

  • Don't delete user data on deactivation
  • Don't perform heavy operations
  • Don't echo output during activation
  • Don't use exit or die (except wp_die)
  • Don't assume single site
  • Don't ignore error handling
  • Don't flush rewrite rules often
  • Don't modify core tables

Complete Activation/Deactivation Example

Full Implementation

<?php
/**
 * Plugin Name: Complete Lifecycle Example
 * Version: 1.0.0
 */

// Prevent direct access
if ( ! defined( 'ABSPATH' ) ) {
    exit;
}

// Define constants
define( 'MY_PLUGIN_VERSION', '1.0.0' );
define( 'MY_PLUGIN_DB_VERSION', '1.0' );

// Activation
register_activation_hook( __FILE__, 'my_plugin_activate' );

function my_plugin_activate( $network_wide ) {
    // Minimum requirements
    $min_php = '7.2';
    $min_wp = '5.0';
    
    // Check PHP version
    if ( version_compare( PHP_VERSION, $min_php, '<' ) ) {
        deactivate_plugins( plugin_basename( __FILE__ ) );
        wp_die( sprintf( 
            'This plugin requires PHP %s or higher. You are running PHP %s.',
            $min_php,
            PHP_VERSION
        ) );
    }
    
    // Check WordPress version
    if ( version_compare( $GLOBALS['wp_version'], $min_wp, '<' ) ) {
        deactivate_plugins( plugin_basename( __FILE__ ) );
        wp_die( sprintf(
            'This plugin requires WordPress %s or higher. You are running WordPress %s.',
            $min_wp,
            $GLOBALS['wp_version']
        ) );
    }
    
    // Handle multisite
    if ( is_multisite() && $network_wide ) {
        $sites = get_sites();
        foreach ( $sites as $site ) {
            switch_to_blog( $site->blog_id );
            my_plugin_activate_single();
            restore_current_blog();
        }
    } else {
        my_plugin_activate_single();
    }
}

function my_plugin_activate_single() {
    // Create database tables
    global $wpdb;
    $table_name = $wpdb->prefix . 'my_plugin_data';
    $charset_collate = $wpdb->get_charset_collate();
    
    $sql = "CREATE TABLE $table_name (
        id mediumint(9) NOT NULL AUTO_INCREMENT,
        created_at datetime DEFAULT CURRENT_TIMESTAMP,
        data text NOT NULL,
        PRIMARY KEY (id)
    ) $charset_collate;";
    
    require_once( ABSPATH . 'wp-admin/includes/upgrade.php' );
    dbDelta( $sql );
    
    // Add options
    add_option( 'my_plugin_version', MY_PLUGIN_VERSION );
    add_option( 'my_plugin_db_version', MY_PLUGIN_DB_VERSION );
    add_option( 'my_plugin_settings', array(
        'enabled' => true,
        'mode' => 'standard'
    ) );
    
    // Add capabilities
    $role = get_role( 'administrator' );
    if ( $role ) {
        $role->add_cap( 'manage_my_plugin' );
    }
    
    // Schedule cron
    if ( ! wp_next_scheduled( 'my_plugin_cron' ) ) {
        wp_schedule_event( time(), 'daily', 'my_plugin_cron' );
    }
    
    // Set activation flag
    set_transient( 'my_plugin_activated', true, 30 );
}

// Deactivation
register_deactivation_hook( __FILE__, 'my_plugin_deactivate' );

function my_plugin_deactivate() {
    // Clear scheduled hooks
    wp_clear_scheduled_hook( 'my_plugin_cron' );
    
    // Remove capabilities
    $role = get_role( 'administrator' );
    if ( $role ) {
        $role->remove_cap( 'manage_my_plugin' );
    }
    
    // Clear cache
    wp_cache_flush();
    
    // Clear transients
    delete_transient( 'my_plugin_activated' );
    delete_transient( 'my_plugin_cache' );
}

// Show welcome notice after activation
add_action( 'admin_notices', 'my_plugin_activation_notice' );

function my_plugin_activation_notice() {
    if ( get_transient( 'my_plugin_activated' ) ) {
        ?>
        <div class="notice notice-success is-dismissible">
            <p><strong>Plugin activated successfully!</strong></p>
            <p>Thank you for installing our plugin. <a href="<?php echo admin_url( 'admin.php?page=my-plugin' ); ?>">Get started here</a>.</p>
        </div>
        <?php
        delete_transient( 'my_plugin_activated' );
    }
}

Lifecycle Hook Best Practices

  • Keep it light: Don't perform heavy operations during activation
  • Check requirements: Verify system requirements before activation
  • Use dbDelta: For database table creation and updates
  • Version everything: Track plugin and database versions
  • Handle errors gracefully: Use wp_die() with helpful messages
  • Clean up properly: Remove scheduled tasks on deactivation
  • Respect user data: Only delete data on uninstall
  • Test thoroughly: Test on single and multisite
