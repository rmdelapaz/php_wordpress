Writing Unit Tests

Basic Test Example

class Test_Plugin_Functions extends WP_UnitTestCase {
    
    public function setUp(): void {
        parent::setUp();
        // Setup test environment
    }
    
    public function test_plugin_activation() {
        activate_plugin('my-plugin/my-plugin.php');
        $this->assertTrue(is_plugin_active('my-plugin/my-plugin.php'));
    }
    
    public function test_custom_post_type_registration() {
        $this->assertTrue(post_type_exists('my_custom_type'));
    }
    
    public function test_shortcode_output() {
        $output = do_shortcode('[my_shortcode]');
        $this->assertStringContainsString('expected content', $output);
    }
    
    public function test_ajax_handler() {
        $_POST['action'] = 'my_ajax_action';
        $_POST['nonce'] = wp_create_nonce('my_nonce');
        
        $this->_handleAjax('my_ajax_action');
        
        $response = json_decode($this->_last_response);
        $this->assertTrue($response->success);
    }
}

Integration Testing

Testing Database Operations

class Test_Database_Operations extends WP_UnitTestCase {
    
    public function test_create_custom_table() {
        global $wpdb;
        
        // Run table creation
        my_plugin_create_tables();
        
        // Check if table exists
        $table_name = $wpdb->prefix . 'my_custom_table';
        $this->assertEquals($table_name, $wpdb->get_var(
            "SHOW TABLES LIKE '$table_name'"
        ));
    }
    
    public function test_data_insertion() {
        global $wpdb;
        
        $data = array(
            'name' => 'Test Item',
            'value' => 'Test Value'
        );
        
        $result = $wpdb->insert(
            $wpdb->prefix . 'my_custom_table',
            $data
        );
        
        $this->assertNotFalse($result);
        $this->assertEquals(1, $wpdb->insert_id);
    }
}

Automated Testing

GitHub Actions CI/CD

# .github/workflows/tests.yml
name: Plugin Tests
on: [push, pull_request]

jobs:
  test:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        wordpress: ['5.9', '6.0', '6.1']
        php: ['7.4', '8.0', '8.1']
    
    steps:
      - uses: actions/checkout@v2
      
      - name: Setup PHP
        uses: shivammathur/setup-php@v2
        with:
          php-version: ${{ matrix.php }}
      
      - name: Install dependencies
        run: composer install
      
      - name: Run tests
        run: ./vendor/bin/phpunit
