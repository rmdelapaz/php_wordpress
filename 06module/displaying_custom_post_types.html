    // Get current post's categories
    $terms = wp_get_post_terms( $post_id, 'product_category', array( 'fields' => 'ids' ) );

        'tax_query' => array(
            array(
                'taxonomy' => 'product_category',
                'field' => 'term_id',
                'terms' => $terms,
                'operator' => 'IN'
            )
        ),
        'orderby' => 'rand'
    );
    
    return new WP_Query( $args );
}

    function get_featured_products_cached() {
    $cache_key = 'featured_products_' . get_current_blog_id();
    $products = get_transient( $cache_key );
    
    if ( false === $products ) {
        $args = array(
            'post_type' => 'product',
            'posts_per_page' => 6,
            'meta_query' => array(
                array(
                    'key' => '_featured',
                    'value' => 'yes',
                    'compare' => '='
                )
            ),
            'orderby' => 'modified',
            'order' => 'DESC'
        );
        
        $products = new WP_Query( $args );
        
        // Cache for 12 hours
        set_transient( $cache_key, $products, 12 * HOUR_IN_SECONDS );
    }
    
    return $products;
}

// Clear cache when products are updated
function clear_featured_products_cache( $post_id ) {
    if ( get_post_type( $post_id ) === 'product' ) {
        delete_transient( 'featured_products_' . get_current_blog_id() );
    }
}

    // Infinite scroll for products
(function($) {
    var page = 2;
    var loading = false;
    var $container = $('#products-container');
    var $window = $(window);
    
    function loadMoreProducts() {
        if (loading) return;
        
        loading = true;
        
        $.ajax({
            url: ajax_object.ajax_url,
            type: 'POST',
            data: {
                action: 'infinite_scroll_products',
                page: page,
                nonce: ajax_object.nonce
            },
            success: function(response) {
                if (response.success && response.data.html) {
                    $container.append(response.data.html);
                    page++;
                    loading = false;
                    
                    if (!response.data.has_more) {
                        $window.off('scroll', checkScroll);
                    }
                } else {
                    $window.off('scroll', checkScroll);
                }
            },
            error: function() {
                loading = false;
            }
        });
    }
    
    function checkScroll() {
        if ($window.scrollTop() + $window.height() > $(document).height() - 100) {
            loadMoreProducts();
        }
    }
    
    $window.on('scroll', checkScroll);
})(jQuery);

    // Complex filtering system
class Product_Filter {
    
    public function __construct() {
        add_action( 'wp_ajax_filter_products', array( $this, 'handle_filter' ) );
        add_action( 'wp_ajax_nopriv_filter_products', array( $this, 'handle_filter' ) );
    }
    
    public function handle_filter() {
        check_ajax_referer( 'filter_nonce', 'nonce' );
        
        $args = array(
            'post_type' => 'product',
            'posts_per_page' => 12,
            'paged' => $_POST['page'] ?? 1
        );
        
        // Price range filter
        if ( ! empty( $_POST['min_price'] ) || ! empty( $_POST['max_price'] ) ) {
            $args['meta_query'][] = array(
                'key' => '_price',
                'value' => array( 
                    floatval( $_POST['min_price'] ?? 0 ), 
                    floatval( $_POST['max_price'] ?? 999999 ) 
                ),
                'type' => 'NUMERIC',
                'compare' => 'BETWEEN'
            );
        }
        
        // Category filter
        if ( ! empty( $_POST['categories'] ) ) {
            $args['tax_query'][] = array(
                'taxonomy' => 'product_category',
                'field' => 'slug',
                'terms' => array_map( 'sanitize_text_field', $_POST['categories'] )
            );
        }
        
        // Brand filter
        if ( ! empty( $_POST['brands'] ) ) {
            $args['tax_query'][] = array(
                'taxonomy' => 'product_brand',
                'field' => 'slug',
                'terms' => array_map( 'sanitize_text_field', $_POST['brands'] )
            );
        }
        
        // Features filter
        if ( ! empty( $_POST['features'] ) ) {
            foreach ( $_POST['features'] as $feature ) {
                $args['meta_query'][] = array(
                    'key' => '_feature_' . sanitize_key( $feature ),
                    'value' => 'yes',
                    'compare' => '='
                );
            }
        }
        
        // Sorting
        if ( ! empty( $_POST['orderby'] ) ) {
            switch ( $_POST['orderby'] ) {
                case 'price_asc':
                    $args['orderby'] = 'meta_value_num';
                    $args['meta_key'] = '_price';
                    $args['order'] = 'ASC';
                    break;
                case 'price_desc':
                    $args['orderby'] = 'meta_value_num';
                    $args['meta_key'] = '_price';
                    $args['order'] = 'DESC';
                    break;
                case 'name':
                    $args['orderby'] = 'title';
                    $args['order'] = 'ASC';
                    break;
                case 'newest':
                    $args['orderby'] = 'date';
                    $args['order'] = 'DESC';
                    break;
            }
        }
        
        // Search term
        if ( ! empty( $_POST['search'] ) ) {
            $args['s'] = sanitize_text_field( $_POST['search'] );
        }
        
        $query = new WP_Query( $args );
        
        ob_start();
        
        if ( $query->have_posts() ) {
            while ( $query->have_posts() ) {
                $query->the_post();
                get_template_part( 'template-parts/content', 'product' );
            }
        } else {
            echo '
    No products found matching your criteria.
    ';
        }
        
        $html = ob_get_clean();
        
        wp_send_json_success( array(
            'html' => $html,
            'found_posts' => $query->found_posts,
            'max_pages' => $query->max_num_pages
        ) );
    }
}
